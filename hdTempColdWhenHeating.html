(ahu,dates,tempThreshold:2,minTime:30min,heatvalveThreshold:2%,maxTime:60min,freq:2,hdTempMinVal:6,hdTempMaxVal:60,hdTempSPMinVal:6,hdTempSPMaxVal:60,damperMatch:10,check:true,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

  equipWorkOrder:toEquipWorkOrder(ahu)
  occupiedperiods:[].toGrid
  spark:[].toGrid
  spark1:[].toGrid
  spark2:[].toGrid

  heatElement:read(heat and cmd and equipRef == ahu->id)
  heatElementHis:heatElement.hisRead(dates).hisFindPeriods(v=>v!=na() and v>heatvalveThreshold)
  occupancyPoint:read(occupied and cmd and equipRef == ahu->id,false)
  hdTemp:read(hotDeck and sensor and temp and equipRef == ahu->id)
  hdTempSp:read(hotDeck and sp and temp and equipRef == ahu->id)
  hdDamper:read(hotDeck and damper and cmd and equipRef == ahu->id,false)
  maintanancePoint:ahu.tomaintenancePoint()
  ahuOn:ahu.ahuRunPeriods(dates)

  hdTempHis:hdTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>hdTempMinVal and v<hdTempMaxVal))
  issueTime:comparepoints(hdTempSp,hdTemp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
  hdTempSpHis:hdTempSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>hdTempSPMinVal and v<hdTempSPMaxVal))
  spark=hisPeriodIntersection([issueTime,ahuOn,hdTempHis,heatElementHis,hdTempSpHis])
  spark_inter:spark

  if(maintanancePoint!=null)
    do
    maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
    spark_inter= hisPeriodIntersection([spark,maintanceHis])
    end

  if(occupancyPoint!=null and check== true )
    do
    occupiedperiods=togetOccupancy(ahu,dates)
    spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
    end
  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)
  if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
