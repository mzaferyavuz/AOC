(vav, dates,tempdeviation:3,minTime:30min,dischargeTempThreshold:2,reheatValveThreshold:5,damperThreshold:20,maxTime:60min,freq:2,znTmpLowLimit:10,znTmpHighLimit:35,pressureSpdeviation:50,ahuvfdspeed:20,boilerTempsetpoint:71,boilererror:2,znTmpspLowLimit:18,znTmpspHighLimit:28,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

vavdamp:toVavDamper(vav)
ahu:vav.toAHUofVAV()
temp:vav.toAhuZoneTemp()
tempSP:vav.toAhuZoneSetpoint()
eqp:readById(vav->ahuRef)
boilerPlant:vav["boilerPlantRef"]
equipWorkOrder:toEquipWorkOrder(vav)
if(boilerPlant!=null) boilerPlant=readById(boilerPlant)
pressurePoint:read(discharge and pressure and sensor and equipRef == ahu->id,false)
pressureSPPoint:read(discharge and pressure and sp and equipRef == ahu->id,false)

ahuStatus:read(discharge and fan and run and sensor and equipRef == ahu->id,false)
if(ahuStatus==null) throw "AHU fan status point missing"

ps:hisRead([temp,tempSP,ahuStatus,vavdamp],dates).hisInterpolate().map row => do
if(row["v0"]!=na() and row["v1"]!=na() and row["v2"]!=na() and row["v3"]!=na()) {ts: row->ts,v0:((row["v1"] - row["v0"])>tempdeviation)and(row["v2"]==true) and row["v0"]>=znTmpLowLimit and row["v0"]<=znTmpHighLimit and row["v1"]>=znTmpspLowLimit and row["v1"]<=znTmpspHighLimit and row["v3"]>damperThreshold}
else {ts: row->ts,v0:na()}
end.hisFindPeriods(v=>(v!=na() and v==true))

overCoolingPeriod:ahudischargecheck(eqp,dates,dischargeTempThreshold)
reheatPeriods:vav.togetreheatPeriods(dates,reheatValveThreshold,true)

if(boilerPlant!=null) do
boilerrunperiods:boilerPlant.anyboileron(dates)
boilerTempsensor:read(leaving and water and temp and sensor and equipRef==boilerPlant->id,false)

if(boilerTempsensor!=null) do
boilertemp:boilerPlant.boilerLowTemp_inter(dates,boilerTempsetpoint,boilererror)
ps=hisPeriodIntersection([ps,boilertemp,boilerrunperiods,overCoolingPeriod,reheatPeriods])
end
else ps=hisPeriodIntersection([ps,boilerrunperiods,overCoolingPeriod,reheatPeriods])

end
else ps=hisPeriodIntersection([ps,overCoolingPeriod,reheatPeriods])


if((pressurePoint!=null)and(pressureSPPoint!=null)) do
pressuremismatch:eqp.dischargeFanStatic_inter(dates,pressureSpdeviation,ahuvfdspeed)
ps=hisPeriodIntersection([ps,pressuremismatch])
end

ps=occupancyandMaintenance(vav,dates,ps)

spark1:ps.findAll(ts => ts->v0 > minTime)
spark2:ps.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
