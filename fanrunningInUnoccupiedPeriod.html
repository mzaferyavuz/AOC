src

(ahu, dates, minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Energy",execute:true) => do

equipWorkOrder:toEquipWorkOrder(ahu)
occupancyPoint:read(occupied and cmd and equipRef == ahu->id,false)
cmd:read(discharge and fan and cmd and run and equipRef==ahu->id,false)
fanStatus:read(discharge and fan and sensor and run and equipRef==ahu->id,false)

if(fanStatus==null) throw "No Fan status in the equipment"
statusOnPeriods:fanStatus.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))

if(occupancyPoint!=null and occupancyPoint["kind"]=="Bool") unOccPeriods:occupancyPoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
else if(occupancyPoint!=null and occupancyPoint["kind"]=="Str") unOccPeriods:occupancyPoint.hisRead(dates).hisFindPeriods(matchPointVal(_,"Unoccupied" ))
else throw "No Occupied point found"

spark:hisPeriodIntersection([statusOnPeriods,unOccPeriods])

  enablecmd:readAll(fan and enable and cmd and equipRef==ahu->id)

  if(enablecmd.size==1) do
  enbl:enablecmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
  spark=hisPeriodIntersection([spark,enbl])
  end

  else if(enablecmd.size>1) do
  enbl:[].toGrid

  enablecmd.each r=>do
  a:r.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
  if(enbl.size==0 and a.size>0) enbl==a else if(enbl.size>0 and a.size>0) enbl==hisPeriodUnion([enbl,a])
  end

  if(enbl.size>0) spark=hisPeriodIntersection([spark,enbl])
  end


if(cmd!=null) do
  cmdOnPeriods:cmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))
  spark= hisPeriodIntersection([spark,cmdOnPeriods])
end

spark=occupancyandMaintenance(ahu,dates,spark,false)

spark1:spark.findAll(ts => ts->v0 > minTime)
spark2:spark.findAll(ts => ts->v0 > maxTime)

ruleCost:costCalc(ahu,spark1,"supply",pf:1,execute)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type),spark_opexCost:ruleCost})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type),spark_opexCost:ruleCost})
end
