(eqp,dates,threshold:25,ruleWorkOrder:"None",priority:"Medium",type:"Gas") => do

  equipWorkOrder:toEquipWorkOrder(eqp)
  maintanancePoint:tomaintenancePoint(eqp)

  pnthis:[].toGrid
  pnt:read(point and gas and energy and equipRef==eqp->id,false)
  if(pnt==null) throw "No Gas Energy Point"

  ptime1:dates-(7day)
  ptime2:dates-(14day)

  aveg:na()
  if(dates.weekday == 0) do
  avg1:pnt.hisRead(ptime1).hisFindAll((val, ts) => ts.weekday == 0).hisFindAll(v=> v!=na())
  if(avg1.size>0) avg1=avg1.hisRollup(sum,1day)
  if(avg1.size>0) avg1=avg1.toRec->v0
  else avg1=na()

  avg2:pnt.hisRead(ptime2).hisFindAll((val, ts) => ts.weekday == 0).hisFindAll(v=> v!=na())
  if(avg2.size>0) avg2=avg2.hisRollup(sum,1day)
  if(avg2.size>0) avg2=avg2.toRec->v0
  else avg2=na()

  if(avg1!=na() and avg2!=na() and avg1>0 and avg2>0) aveg=((avg2+avg1)/2)
  else if(avg1!=na() and avg2!=na() and avg1==0 and avg2>0) aveg=avg2
  else if(avg1!=na() and avg2!=na() and avg2==0 and avg1>0) aveg=avg1
  end

  if(dates.weekday == 1) do

  avg1:pnt.hisRead(ptime1).hisFindAll((val, ts) => ts.weekday == 1).hisFindAll(v=> v!=na())
  if(avg1.size>0) avg1=avg1.hisRollup(sum,1day)
  if(avg1.size>0) avg1=avg1.toRec->v0
  else avg1=na()

  avg2:pnt.hisRead(ptime2).hisFindAll((val, ts) => ts.weekday == 1).hisFindAll(v=> v!=na())

  if(avg1.size>0) avg1=avg1.hisRollup(sum,1day)

  avg1:pnt.hisRead(ptime1).hisFindAll((val, ts) => ts.weekday == 3).hisFindAll(v=> v!=na())
  if(avg1.size>0) avg1=avg1.hisRollup(sum,1day)

  if(avg1.size>0) avg1=avg1.hisRollup(sum,1day)

  end

  if(dates.weekday == 6) do
  avg1:pnt.hisRead(ptime1).hisFindAll((val, ts) => ts.weekday == 6).hisFindAll(v=> v!=na())
  if(avg1.size>0) avg1=avg1.hisRollup(sum,1day)
  if(avg1.size>0) avg1=avg1.toRec->v0
  else avg1=na()

  avg2:pnt.hisRead(ptime2).hisFindAll((val, ts) => ts.weekday == 6).hisFindAll(v=> v!=na())
  if(avg2.size>0) avg2=avg2.hisRollup(sum,1day)

  end

  //aveg

  his:pnt.hisRead(dates).hisFindAll(v=> v!=na()).hisRollup(sum,1day)

  if(his.size>0) pnthis=his.map row=>do
  if(aveg!=na() and row["v0"]>0 and aveg>0) {ts:row->ts, v0:(row["v0"]>(aveg+(aveg*(threshold/100))))}
  else {ts:row->ts,v0:na()}
  end.hisFindPeriods(v=> v!=na() and v==true)

  if(pnthis.size>0) return pnthis.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
