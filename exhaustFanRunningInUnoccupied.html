(equipment,dates,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Energy") => do

      equipWorkOrder:toEquipWorkOrder(equipment)

          occupancyPoint:read(occupied and cmd and equipRef == equipment->id,false)
          maintanancePoint:equipment.tomaintenancePoint()
          occupiedperiods:[].toGrid

      cmd:read(exhaust and fan and cmd and run and equipRef==equipment->id,false)
      fanStatus:read(exhaust and fan and sensor and run and equipRef==equipment->id,false)

       if(fanStatus==null) throw "No Status Point Available"

      statusOnPeriods:fanStatus.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))



       if(occupancyPoint!=null and occupancyPoint["kind"]=="Bool") occupiedperiods=occupancyPoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
        else if(occupancyPoint!=null and occupancyPoint["kind"]=="Str") occupiedperiods=occupancyPoint.hisRead(dates).hisFindPeriods(matchPointVal(_,"Unoccupied" ))
        else throw "No Occupancy Point Available"


      spark:hisPeriodIntersection([statusOnPeriods,occupiedperiods])


      if(cmd!=null)
        do
          cmdOnPeriods:cmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))
          spark= hisPeriodIntersection([spark,cmdOnPeriods])
          end

        if(maintanancePoint!=null)
          do
            maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
            spark= hisPeriodIntersection([spark,maintanceHis])
          end

        spark1:spark.findAll(ts => ts->v0 > minTime)
        spark2:spark.findAll(ts => ts->v0 > maxTime)

        if(spark1.size>=freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
      else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
    end
