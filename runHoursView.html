src

(systemtype,site) => do

    readAll(equip and eTypeRef->dis==systemtype and siteRef==site).map row =>do

    //Read equip run sensor

    waterEquip:read(dis==row->dis and equip and (chiller or boiler or pump or coolingTower or chillerPlant or boilerPlant or fan),false)

    if(waterEquip!=null) do

    runSensor:read(run and sensor and equipRef->dis==row->dis,false)

    end

    //Read equip run sensor
   else runSensor:read(discharge and fan and run and sensor and equipRef->dis==row->dis,false)

     occupancy:read(occupied and cmd and equipRef->dis==row->dis,false)

      if(runSensor!=null)do

      zeroHrTM:runSensor.hisRead(thisMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true))
      zeroHrLM:runSensor.hisRead(lastMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true))
      zeroHrBLM:runSensor.hisRead(toDateSpan((lastMonth().start-1mo)..lastMonth().start-1day)).hisClip().hisFindPeriods(v=>(v!=na() and v==true))

      //Current month run hours calculation
      if(zeroHrTM.size==0)runSensorCurr:"0hr"
      else runSensorCurr:runSensor.hisRead(thisMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

      //Last month run hours calculation
      if(zeroHrLM.size==0)runSensorLast:"0hr"
      else runSensorLast:runSensor.hisRead(lastMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

      //Month before last month run hours calculation
      if(zeroHrBLM.size==0)runSensorBefore:"0hr"
      else runSensorBefore:runSensor.hisRead(toDateSpan((lastMonth().start-1mo)..lastMonth().start-1day)).hisClip().hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

      end

      else if(runSensor==null) do

      runSensorCurr:"No Run Sensor"
      runSensorLast:"No Run Sensor"
      runSensorBefore:"No Run Sensor"
      end

      if(waterEquip!=null) do

      occHrCurr:"Not Applicable"
      occHrLast:"Not Applicable"
      occHrBefore:"Not Applicable"
      end

      else

      if(occupancy!=null)do

      occHrTM:occupancy.hisRead(thisMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true))
      occHrLM:occupancy.hisRead(lastMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true))
      occHrBLM:occupancy.hisRead(toDateSpan((lastMonth().start-1mo)..lastMonth().start-1day)).hisClip().hisFindPeriods(v=>(v!=na() and v==true))

      //Current month occupancy hours calculation
      if(occHrTM.size==0)occHrCurr:"0hr"
      else occHrCurr:occupancy.hisRead(thisMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

      //Last month occupancy hours calculation
      if(occHrLM.size==0)occHrLast:"0hr"
      else occHrLast:occupancy.hisRead(lastMonth).hisClip().hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

      //Month before last month occupancy hours calculation
      if(occHrBLM.size==0)occHrBefore:"0hr"
      else occHrBefore:occupancy.hisRead(toDateSpan((lastMonth().start-1mo)..lastMonth().start-1day)).hisClip().hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

      end

      else if(occupancy==null) do

      occHrCurr:"No Occupancy Sensor"
      occHrLast:"No Occupancy Sensor"
      occHrBefore:"No Occupancy Sensor"
      end

    //runSensorLast:read(run and sensor and equipRef->dis==row->dis,false).hisRead(lastMonth).hisFindPeriods(v=>(v!=na() and v==true)).hisRollup(sum,1mo).colToList("v0")

   {equipName:row->dis, thisMonthRH:runSensorCurr, thisMonthOcc:occHrCurr, lastMonthRH:runSensorLast,lastMonthOcc:occHrLast, beforeLastMonthRH:runSensorBefore, beforeLastMonthOcc:occHrBefore}

   end
   end
