(workOrder) => do
   if(workOrder.lower == "and" or workOrder.lower == "a")
    return "And"
   else if(workOrder.lower == "dont" or workOrder.lower=="d")
     return "Dont"
   else if(workOrder.lower == "or" or workOrder.lower =="o")
     return "Or"
   else
     return "None"
end
