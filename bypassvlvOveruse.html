(system,dates,spdeviation:10,valveThreshold:10,minTime:30min,maxTime:60min,minVal:0,maxVal:1000,spMinVal:0,spMaxVal:1000,freq:2,ruleWorkOrder:"None",priority:"High",type:"Comfort") => do

 equipWorkOrder:toEquipWorkOrder(system)

 maintanancePoint:system.tomaintenancePoint()

  spark1:[].toGrid
  spark2:[].toGrid

  dp:system.toSystemPressureSensor()
  dpsp:system.toSystemPressureSetpoint()

  dpHis:dp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>minVal and v<maxVal))
  dpspHis:dpsp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>spMinVal and v<spMaxVal))
       dev:hisRead([dp,dpsp],dates).map row=>do

          if(row["v0"]!=na() and row["v1"]!=na())
               {ts:row->ts,v0:(row["v1"] - row["v0"])>(row["v1"]*(spdeviation/100))}
          else {ts:row->ts,v0:na()}

              end

      issue:dev.hisFindPeriods(v=>(v!=na() and v==true))
     valveOnPeriods:system.toSystemBypassValve.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>valveThreshold))
spark:hisPeriodIntersection([issue,valveOnPeriods,dpHis,dpspHis])
spark_inter:spark

   if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter=hisPeriodIntersection([spark,maintanceHis])
    end

  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

 if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  end
