(ahu, dates, valveThreshold:10) => do

  htgPointBoolPeriods:[].toGrid
  htgPointNumPeriods:[].toGrid

  htgPointBool:readAll(cmd and (heat or reheat) and kind=="Bool" and equipRef==ahu->id)
  htgPointNum:readAll(cmd and (heat or reheat) and kind=="Number" and equipRef==ahu->id)

  if(htgPointBool.size>0)
  do
  htgPointBoolPeriods=htgPointBool.hisRead(dates).hisMap(v=> if(v!=na() and v == true) 1 else 0).hisFoldCols(sum).hisFindPeriods(v=>(v!=na() and v==0))
  end
  if(htgPointNum.size>0)
  do
  htgPointNumPeriods=htgPointNum.hisRead(dates).hisMap(v=> if(v!=na() and v > valveThreshold) 1 else 0).hisFoldCols(sum).hisFindPeriods(v=>(v!=na() and v==0))
  end

  if((htgPointBool.size>0) and (htgPointNum.size>0)) return hisPeriodIntersection([htgPointBoolPeriods, htgPointNumPeriods])

  else if((htgPointBool.size>0) and (htgPointNum.size==0))  return htgPointBoolPeriods

  else if((htgPointBool.size==0) and (htgPointNum.size>0))  return htgPointNumPeriods

  else if((htgPointBool.size==0) and (htgPointNum.size==0))  return throw "No 'heat' points defined: " + dis(ahu)

end
