(ahu,dates,diff:2,chilledWaterTempSP:7,chilledWaterTempThreshold:1,ahuChilledWaterDiffTemp:5,coolValveThreshold:98,tempThreshold:2,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

       ahuRunTime:ahu.ahuRunPeriods(dates)
       spark:[].toGrid()


       equipWorkOrder:toEquipWorkOrder(ahu)

        chillerPlant:ahu["chillerPlantRef"]

        if(chillerPlant!=null ) do

          chilleronperiod: chillerPlant.anyChilleron(dates)
          chilledWaterTemp:read(chilled and leaving and water and temp and sensor and equipRef==chillerPlant->id)
          chilledWaterTempsp:read(chilled and leaving and water and temp and sp and equipRef==chillerPlant->id,false)
             if(chilledWaterTempsp!=null) do
               issue:hisRead([chilledWaterTemp,chilledWaterTempsp],dates).hisInterpolate().map row=>do
              if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:abs(row["v0"]-row["v1"])<chilledWaterTempThreshold}
               else {ts:row->ts,v0:na()}
             end.hisFindPeriods(v=>(v!=na() and v==true))
            end
          else do
           issue:chilledWaterTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v<chilledWaterTempSP))
         end
      else throw "No Chiller Plant Ref"


       coolPointNum:readAll(cmd and cool and position and kind=="Number" and equipRef==ahu->id).hisRead(dates).hisMap(v=> if(v!=na() and v > coolValveThreshold) 1 else 0).hisFoldCols(sum).hisFindPeriods(v=>(v!=na() and v>0))

       chilledEnteringTempSensor:read(chilled and water and temp and sensor and entering and kind=="Number" and equipRef==ahu->id)//.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>ahuChilledWaterEnteringTemp))
       chilledLeavingTempSensor:read(chilled and water and temp and sensor and leaving and kind=="Number" and equipRef==ahu->id)
       issue1:comparepoints(chilledEnteringTempSensor,chilledLeavingTempSensor,dates,ahuChilledWaterDiffTemp).hisFindPeriods(matchPointVal(_,"lesser"))


       datsp:read(discharge and temp and sp and equipRef==ahu->id,false)
      zoneTempSp:read(zone and temp and sp and equipRef==ahu->id,false)

      if(datsp!=null) do
      dat:read(discharge and temp and sensor and equipRef==ahu->id)
      issue2:comparepoints(dat,datsp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
      end
      else if(zoneTempSp!=null) do
      zoneTemp:read(zone and temp and sensor and equipRef==ahu->id)
      issue2:comparepoints(zoneTemp,zoneTempSp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
      end

       spark=hisPeriodIntersection([issue,coolPointNum,issue1,chilleronperiod,ahuRunTime,issue2])

        maintanancePoint:ahu.tomaintenancePoint()
       occupancyPoint:read(occupied and equipRef == ahu->id,false)

   if(maintanancePoint!=null)
     do
       maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
       spark=hisPeriodIntersection([spark,maintanceHis])
     end
   if(occupancyPoint!=null)
     do
        occupiedperiods:togetOccupancy(ahu,dates)
          if(occupiedperiods.size>0)
            do
              spark=hisPeriodIntersection([spark,occupiedperiods])
            end
     end

        spark1:spark.findAll(ts => ts->v0 > minTime)
        spark2:spark.findAll(ts => ts->v0 > maxTime)

     if(spark1.size>freq)
       do
       return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
       end
       else if(spark2.size>0)
       do
      return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
     end

    end
