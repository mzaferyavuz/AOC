(ahu,dates,match:10,minTime:30min,maxTime:60min,freq:2,checked:true,ruleWorkOrder:"None",priority:"High",type:"Energy",execute:true) => do

    heatTime:ahu.ahuHeatPeriods_inter(dates,match)
    ahuOffTime:[].toGrid()
  boilerPlant:ahu["boilerPlantRef"]
    equipWorkOrder:toEquipWorkOrder(ahu)
    spark:[].toGrid
    spark1:[].toGrid
    spark2:[].toGrid
    occupancyPoint:read(occupied and cmd and equipRef == ahu->id,false)
    maintanancePoint:ahu.tomaintenancePoint()
    cmd:read(run and discharge and fan and cmd and equipRef == ahu->id,false)
      sensor:read(run and discharge and fan and sensor and equipRef == ahu->id,false)

   if(cmd!=null and sensor!=null) do
      ahuOffTime=hisPeriodIntersection([sensor.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false)),cmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))])
      end
   else if(cmd!=null and sensor==null) do
      ahuOffTime=cmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      end
       else if(cmd==null and sensor!=null) do
      ahuOffTime=sensor.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      end
       else if(cmd==null and sensor==null) do
      throw "No status or cmd"
      end


  if(boilerPlant==null)
 spark=hisPeriodIntersection([heatTime,ahuOffTime])
  if(boilerPlant!=null ) do

  boileronperiod: boilerPlant.anyboileron(dates)

  if(boileronperiod!=null)
 spark=hisPeriodIntersection([heatTime,ahuOffTime,boileronperiod])
else if(boileronperiod==null)
 spark=hisPeriodIntersection([heatTime,ahuOffTime])
  end




  spark_inter:spark

frzStat:read(freezeStat and sensor and equipRef==ahu->id,false)

if(frzStat!=null and checked==true) do

frzStatStatus:frzStat.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
spark_inter= hisPeriodIntersection([spark_inter,frzStatStatus])
end

if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter= hisPeriodIntersection([spark_inter,maintanceHis])
    end


  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

  ruleCost:costCalc(ahu,spark1,"elecHeater",pf:1,execute)

    if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type),spark_opexCost:ruleCost})
      else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type),spark_opexCost:ruleCost})

    end
