(ahu,dates,threshold:5,minTime:30min,maxTime:60min,saPreHWTempMin:6,saPreHWTempMax:60,saPosHWTempMin:6,saPosHWTempMax:60,eaPreHWTempMin:10,eaPreHWTempMax:35,eaPosHWTempMin:10,eaPosHWTempMax:35,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Energy") => do

  spark1:[].toGrid
  spark2:[].toGrid
  spark:[].toGrid

  equipWorkOrder:toEquipWorkOrder(ahu)
  maintanancePoint:ahu.tomaintenancePoint()

  saPreHWTemp:read(dischargeDuct and preHeatWheel and air and temp and sensor and equipRef == ahu->id,false)
  saPosHWTemp:read(dischargeDuct and postHeatWheel and air and temp and sensor and equipRef == ahu->id,false)
  eaPreHWTemp:read(returnDuct and preHeatWheel and air and temp and sensor and equipRef == ahu->id,false)
  eaPosHWTemp:read(returnDuct and postHeatWheel and air and temp and sensor and equipRef  == ahu->id,false)

  if(saPreHWTemp==null) throw "No Preheat Temp"
  if(saPosHWTemp==null) throw "No Postheat Temp"

  saPreHWTempHis:saPreHWTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>saPreHWTempMin and v<saPreHWTempMax))
  saPosHWTempHis:saPosHWTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>saPosHWTempMin and v<saPosHWTempMax))

  status:read(heatWheel and run and sensor and equipRef==ahu->id,false)
  cmd:read(heatWheel and run and cmd and equipRef==ahu->id,false)
  ahuOn:ahu.ahuRunPeriods(dates)

  if(cmd!=null and status!=null) systemOn:hisRead([status,cmd],dates).hisMap(v=> if(v!=na() and v==true) 1 else if(v!=na() and v==false) 0 else na()).hisFoldCols(sum).hisFindPeriods(r=> r!=na() and r>1)
  else if(cmd==null and status!=null) systemOn:status.hisRead(dates).hisFindPeriods(r=> r!=na() and r==true)
  else if(cmd!=null and status==null) systemOn:cmd.hisRead(dates).hisFindPeriods(r=> r!=na() and r==true)
  else throw "No CMD or Sts point for HeatWheel"

  spark=hisPeriodIntersection([systemOn,ahuOn])

  if(spark.size==0) return spark.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

  if (eaPreHWTemp!=null and eaPosHWTemp!=null) do
  eaPreHWTempHis:eaPreHWTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>eaPreHWTempMin and v<eaPreHWTempMax))
  eaPosHWTempHis:eaPosHWTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>eaPosHWTempMin and v<eaPosHWTempMax))

  delta:hisRead([saPreHWTemp,saPosHWTemp,eaPreHWTemp,eaPosHWTemp],dates).hisInterpolate().map row=>do
  if(row["v0"]!=na() and row["v1"]!=na() and row["v2"]!=na() and row["v3"]!=na()) {ts:row->ts,v0:(abs(row["v0"]-row["v1"])< threshold) or (abs(row["v3"]-row["v4"])< threshold)}
  else {ts:row->ts,v0:na()}
  end
  issue:delta.hisFindPeriods(v=>(v!=na() and v==true))

  spark=hisPeriodIntersection([spark,issue,saPreHWTempHis,saPosHWTempHis,eaPreHWTempHis,eaPosHWTempHis])
  end

  if(eaPreHWTemp==null or eaPosHWTemp==null) do

  delta:hisRead([saPreHWTemp,saPosHWTemp],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:(abs(row["v0"]-row["v1"])< threshold)}
  else {ts:row->ts,v0:na()}
  end
  issue:delta.hisFindPeriods(v=>(v!=na() and v==true))

  spark=hisPeriodIntersection([spark,issue,saPreHWTempHis,saPosHWTempHis])
  end

  if(maintanancePoint!=null)
  do
  maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
  spark= hisPeriodIntersection([spark,maintanceHis])
  end

  spark1=spark.findAll(ts => ts->v0 > minTime)
  spark2=spark.findAll(ts => ts->v0 > maxTime)

  if(spark1.size>=freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

  end
