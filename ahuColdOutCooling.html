(ahu,dates,coolingBaseTemp:18,valveThreshold:5,minTime:15min,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

  coldOutsidePeriods:ahu.toAhuOutsideTemp().hisRead(dates).hisFindPeriods(v=>(v<coolingBaseTemp and v!=na()))
  ahuCoolingPeriods:ahu.ahuCoolPeriods(dates,valveThreshold)
  ahuOnTime:ahu.ahuRunPeriods(dates)
  equipWorkOrder:toEquipWorkOrder(ahu)
  chillerPlant:ahu["chillerPlantRef"]

if(chillerPlant==null)
 return hisPeriodIntersection([coldOutsidePeriods,ahuCoolingPeriods,ahuOnTime]).findAll(ts=>ts->v0>minTime)
  .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

  if(chillerPlant!=null ) do

  chilleronperiod: chillerPlant.anyChilleron(dates)

  if(chilleronperiod!=null)
 return hisPeriodIntersection([coldOutsidePeriods,ahuCoolingPeriods,ahuOnTime,chilleronperiod]).findAll(ts => ts->v0 > minTime)
  .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

if(chilleronperiod==null)
 return hisPeriodIntersection([coldOutsidePeriods,ahuCoolingPeriods,ahuOnTime]).findAll(ts => ts->v0 > minTime)
  .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

  end
end
