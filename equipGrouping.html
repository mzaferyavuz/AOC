src

() => do

     sites:readAll(site)

     sites.each site =>do

      addToCHWS:(ste,point) => do
        if(read(eType and dis=="Chilled Water System" and siteRef==ste->id,false)==null) commit(diff(null,{eType,dis:"Chilled Water System",siteRef:ste->id},{add}))
        eTypeCHW:read(eType and dis=="Chilled Water System" and siteRef==ste->id)
        commit(diff(point,{eTypeRef:eTypeCHW->id}))
      end

      addToHWS:(ste,point) => do
      if(read(eType and dis=="Hot Water System" and siteRef==ste->id,false)==null) commit(diff(null,{eType,dis:"Hot Water System",siteRef:ste->id},{add}))
        eTypeHWS:read(eType and dis=="Hot Water System" and siteRef==ste->id)
        commit(diff(point,{eTypeRef:eTypeHWS->id}))
      end

      addToMiscSys:(ste,point) => do
        if(read(eType and dis=="Misc Systems" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Misc Systems",siteRef:site->id},{add}))
        eTypeMSys:read(eType and dis=="Misc Systems" and siteRef==site->id)
        commit(diff(point,{eTypeRef:eTypeMSys->id}))
      end

      addToMiscPmp:(ste,point) => do
        if(read(eType and dis=="Misc Pump" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Misc Pump",siteRef:site->id},{add}))
        eTypeMPmp:read(eType and dis=="Misc Pump" and siteRef==site->id)
        commit(diff(point,{eTypeRef:eTypeMPmp->id}))
      end

        change:readAll(equip and (hotWaterPlant or chilledWaterPlant)).each row=>do

        if(row["hotWaterPlant"]!=null) commit(diff(row,{boilerPlant}))
        else if(row["chilledWaterPlant"]!=null) commit(diff(row,{chillerPlant}))
        end

        change2:readAll(equip and hotWaterPlantRef).each row=>do
        if(row["hotWaterPlantRef"]!=null and not (row["boiler"]!=null or row["pump"]!=null) ) commit(diff(row,{boilerPlantRef:row->hotWaterPlantRef->id}))
        else if(row["hotWaterPlantRef"]!=null and (row["boiler"]!=null or row["pump"]!=null) ) commit(diff(row,{equipRef:row->hotWaterPlantRef->id}))
        end

         change3:readAll(equip and chilledWaterPlantRef).each row=>do
          if (row["chilledWaterPlantRef"]!=null and not (row["chiller"]!=null or row["pump"]!=null or row["coolingTower"]!=null)) commit(diff(row,{chillerPlantRef:row->chilledWaterPlantRef->id}))
          else if (row["chilledWaterPlantRef"]!=null and (row["chiller"]!=null or row["pump"]!=null or row["coolingTower"]!=null)) commit(diff(row,{equipRef:row->chilledWaterPlantRef->id}))
          end

       readAll(equip and siteRef==site->id).each row=>do

      // group Roof Top Units
      if(row["rtu"]!=null and row["eTypeRef"]==null) do
          if(read(eType and dis=="Roof Top Unit" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Roof Top Unit",siteRef:site->id},{add}))
          eTypeRTU:read(eType and dis=="Roof Top Unit" and siteRef==site->id)
          commit(diff(row,{eTypeRef:eTypeRTU->id}))
          end

      //group AHUs
      else if(row["ahu"]!=null and row["fcu"]==null and row["eTypeRef"]==null) do
          if(read(eType and dis=="Air Handling Unit" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Air Handling Unit",siteRef:site->id},{add}))
          eTypeAHU:read(eType and dis=="Air Handling Unit" and siteRef==site->id)
          commit(diff(row,{eTypeRef:eTypeAHU->id}))
          end

      //group FCUs
      else if(row["fcu"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Fan Coil Unit" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Fan Coil Unit",siteRef:site->id},{add}))
        eTypeFCU:read(eType and dis=="Fan Coil Unit" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeFCU->id}))
      end

      //group VAVs
      else if(row["vav"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Variable Air Volume" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Variable Air Volume",siteRef:site->id},{add}))
        eTypeVAV:read(eType and dis=="Variable Air Volume" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeVAV->id}))
      end

      //group chiller plant and chillers as Chilled Water System
      else if(row["eTypeRef"]==null and (row["chiller"]!=null or row["chillerPlant"]!=null or row["coolingTower"]!=null) ) do
        addToCHWS(site,row)
      end

      //group chiller plant and chillers as Chilled Water System
      else if(row["eTypeRef"]==null and (row["heatExchanger"]!=null and row["cooling"]!=null)) do
        addToCHWS(site,row)
      end

      //group boiler plant and boilers as Hot Water System
      else if(row["eTypeRef"]==null and (row["boiler"]!=null or row["boilerPlant"]!=null )) do
        addToHWS(site,row)
      end

      //group heat exchanger as Hot Water System
      else if(row["eTypeRef"]==null and (row["heatExchanger"]!=null and row["heating"]!=null)) do
        addToHWS(site,row)
      end

      //pumps associated with the chillerPlant or chiller as Chilled Water System and boilerPlant and boilers as Hot Water System
      else if (row["pump"]!=null) do
       e:row["equipRef"]
         if(e!=null) do
           if(readById(e).has("chiller") or readById(e).has("chillerPlant") or readById(e).has("coolingTower")) addToCHWS(site,row)
           else if(readById(e).has("boiler") or readById(e).has("boilerPlant")) addToHWS(site,row)
           else addToMiscPmp(site,row)
         end
         else addToMiscPmp(site,row)
      end

        //group Chilled Beam
      else if(row["chilledBeam"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Chilled Beam" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Chilled Beam",siteRef:site->id},{add}))
        eTypeACB:read(eType and dis=="Chilled Beam" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeACB->id}))
      end

      //group fan as Ventilating Fans
      else if(row["fan"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Ventilating Fans" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Ventilating Fans",siteRef:site->id},{add}))
        eTypeMFans:read(eType and dis=="Ventilating Fans" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeMFans->id}))
      end

       //group all lightsGroup as Lights
      else if(row["lighting"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Lights" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Lights",siteRef:site->id},{add}))
        eTypeLights:read(eType and dis=="Lights" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeLights->id}))
      end

      //group all Elec Meters as Meter
      else if(row["meter"]!=null and row["elec"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Elec Meter" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Elec Meter",siteRef:site->id},{add}))
        eTypeMeter:read(eType and dis=="Elec Meter" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeMeter->id}))
      end

      //Group radiant heaters
      else if(row["heating"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Radiant Heating" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Radiant Heating",siteRef:site->id},{add}))
        eTypeHeating:read(eType and dis=="Radiant Heating" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeHeating->id}))
      end

      //Group unitHeater
      else if(row["unitHeater"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Unit Heater" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Unit Heater",siteRef:site->id},{add}))
        eTypeHeating:read(eType and dis=="Unit Heater" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeHeating->id}))
      end

      //group exhaust
      else if(row["exhaust"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Exhaust Plant" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Exhaust Plant",siteRef:site->id},{add}))
        eTypeExhaust:read(eType and dis=="Exhaust Plant" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeExhaust->id}))
      end

      //group Air Conditioner
      else if(row["airConditioner"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Air Conditioner" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Air Conditioner",siteRef:site->id},{add}))
        eTypeAirCon:read(eType and dis=="Air Conditioner" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeAirCon->id}))
      end

      //group Air Curtain
      else if(row["airCurtain"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Air Curtain" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Air Curtain",siteRef:site->id},{add}))
        eTypeAirCur:read(eType and dis=="Air Curtain" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeAirCur->id}))
      end

      //Group Condenser
      else if(row["condenser"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Condenser" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Condenser",siteRef:site->id},{add}))
        eTypeCon:read(eType and dis=="Condenser" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeCon->id}))
      end

     //Group CRAC Unit
     else if(row["crac"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Computer Room Air Conditioner" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Computer Room Air Conditioner",siteRef:site->id},{add}))
        eTypeCU:read(eType and dis=="Computer Room Air Conditioner" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeCU->id}))
      end

      //Group CRAH Unit
     else if(row["crah"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Computer Room Air Handler" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Computer Room Air Handler",siteRef:site->id},{add}))
        eTypeCU:read(eType and dis=="Computer Room Air Handler" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeCU->id}))

      //Group Heat Pump
      else if(row["heatPump"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Heat Pump" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Heat Pump",siteRef:site->id},{add}))
        eTypeHP:read(eType and dis=="Heat Pump" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeHP->id}))
      end

      //Group Gas Meter
      else if(row["gas"]!=null and row["meter"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Gas Meter" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Gas Meter",siteRef:site->id},{add}))
        eTypeGasMt:read(eType and dis=="Gas Meter" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeGasMt->id}))
      end

      //Group Water Meter
      else if(row["water"]!=null and row["meter"]!=null and row["thermal"]==null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Water Meter" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Water Meter",siteRef:site->id},{add}))
        eTypeWaterMt:read(eType and dis=="Water Meter" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeWaterMt->id}))
      end

      //Group Thermal Energy Meter
      else if(row["thermal"]!=null and row["energy"]!=null and row["meter"]!=null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Thermal Energy Meter" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Thermal Energy Meter",siteRef:site->id},{add}))
        eTypeThermalEnergyMt:read(eType and dis=="Thermal Energy Meter" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeThermalEnergyMt->id}))
      end

      //Group Radiant Cooling System
      else if(row["cooling"]!=null and row["heatExchanger"]==null and row["eTypeRef"]==null) do
        if(read(eType and dis=="Radiant Cooling System" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Radiant Cooling System",siteRef:site->id},{add}))
        eTypeGasRadCool:read(eType and dis=="Radiant Cooling System" and siteRef==site->id)
        commit(diff(row,{eTypeRef:eTypeGasRadCool->id}))
      end

      //Group Variable Refrigerant Flow
      else if(row["vrf"]!=null and row["eTypeRef"]==null) do
         if(read(eType and dis=="Variable Refrigerant Flow" and siteRef==site->id,false)==null) commit(diff(null,{eType,dis:"Variable Refrigerant Flow",siteRef:site->id},{add}))
         eTypeVRF:read(eType and dis=="Variable Refrigerant Flow" and siteRef==site->id)
         commit(diff(row,{eTypeRef:eTypeVRF->id}))
      end

      //group everything else as Misc Systems
      else do
        if(row["eTypeRef"]==null) do
          addToMiscSys(site,row)
        end
      end

     end
      //commit(diff(read(equipPath),{equipPath:"/site/eType/equip/point"}))
      eTypeRefAdd(site)

      end// end of site each
      addHisAppendNA()
    end
