src

(dates,tempThreshold:3,dmpThreshold:90,issueLimit:50)=> do

equip:readAll((ahu or fcu) and not vavZone and not skipEo)

output:[].toGrid()
spark1:[].toGrid()
spark2:[].toGrid()
spark3:[].toGrid()
spark4:[].toGrid()
znDeltaLow:[].toGrid()
znDeltaHigh:[].toGrid()
znDelta1:[].toGrid()
zoneDemand:[].toGrid()
zoneSensorAvg:[].toGrid()
zoneSpAvg:[].toGrid()
action1:null
siteSuggestion1:null
equip.map row=>do
//ahu1:row["ahuRef"]

znSensor:read(zone and temp and sensor and equipRef==row->id,false)
znSp:read(zone and temp and sp and equipRef==row->id,false)
znDmd1:read(cool and valve and cmd and kind=="Number" and equipRef==row->id,false)
znDmd2:read(cool and valve and cmd and kind=="Bool" and equipRef==row->id,false)

//if (row.has("ahu"))
occupancy2:read(discharge and fan and run and sensor and equipRef==row->id).hisRead(dates)
occupancy:occupancy2.hisFindPeriods(v=>v==true)
//occupancy = togetOccupancy(ahu1,dates)

occupancy1:occupancy.foldCol("v0",sum)
if (znSensor!=null and znSp!=null) do
join:hisRead([znSensor,znSp],dates)
join1:hisJoin([join,occupancy2]).findAll(ts=>(ts["v2"]==true) or (ts["v2"]=="Occupied"))
zoneSensorAvg=join1.findAll(v=>v["v0"]!=na()).foldCol("v0",avg)
zoneSpAvg=join1.findAll(v=>v["v1"]!=na()).foldCol("v1",avg)
znDeltaLow=hisRead([znSensor],dates).hisFindPeriods(v=>(v<row["eominVal"]))//.hisRollup(sum,1day).toRec["v0"]
znDeltaHigh=hisRead([znSensor],dates).hisFindPeriods(v=>v!=na() and (v>row["eomaxVal"]))//.hisRollup(sum,1day).toRec["v0"]
znDelta1=hisRead([znSensor,znSp],dates).hisFoldCols(spread).hisFindPeriods(v=>v!=na() and v>tempThreshold) end

if (znDeltaLow!=null) do
spark1=hisPeriodIntersection([znDeltaLow,occupancy]).foldCol("v0",sum)
spark4=hisPeriodIntersection([znDeltaHigh,occupancy]).foldCol("v0",sum)
spark3=hisPeriodIntersection([znDelta1,occupancy]).foldCol("v0",sum)
end

if (znDmd1!=null)
zoneDemand=znDmd1.hisRead(dates).hisFindPeriods(v=>v!=na() and v>dmpThreshold)

if (znDmd2!=null)
zoneDemand=znDmd2.hisRead(dates).hisFindPeriods(v=>v!=na() and v==true)

if(zoneDemand!=null)
spark2=hisPeriodIntersection([zoneDemand,occupancy]).foldCol("v0",sum)

zoneDeltaEOlow:((spark1/occupancy1)*100).as("%")
if(zoneDeltaEOlow==null) zoneDeltaEOlow=0%
zoneDeltaEOHigh:((spark4/occupancy1)*100).as("%")
if(zoneDeltaEOHigh==null) zoneDeltaEOHigh=0%
zoneDeltaEOlimitPct1:zoneDeltaEOlow+zoneDeltaEOHigh
zoneDeltaPct1:((spark3/occupancy1)*100).as("%")
znDamperPct1:((spark2/occupancy1)*100).as("%")
if (zoneDeltaEOlimitPct1<issueLimit and znDamperPct1>issueLimit)// and zoneDeltaPct1!=null and  znDamperPct1!=null)
action1="switch to temp strategy"

else if (zoneDeltaEOlimitPct1>issueLimit and znDamperPct1<issueLimit)// and zoneDeltaPct1!=null and  znDamperPct1!=null)
action1="switch to demand strategy"

else if (zoneDeltaEOlimitPct1>issueLimit and znDamperPct1>issueLimit)// and zoneDeltaPct1!=null and  znDamperPct1!=null)
action1="Rogue"

else
action1="No issue"

if(zoneDeltaPct1!=null and zoneDeltaPct1>issueLimit)
siteSuggestion1="Setpoints not meeting"

else if(zoneSpAvg!=null and zoneSpAvg<row["eominVal"])
siteSuggestion1="Setpoint too low"

else
siteSuggestion1=null

{eType:row["eTypeRef"], equipName:row->dis, znTmpEOLowlimit:spark1, znTmpEOHighlimit:spark4, zoneDeltaEOlimitPct:zoneDeltaEOlimitPct1, zoneDelta:spark3, zoneDeltaPct:zoneDeltaPct1, valveIssue:spark2, valveIssuePct:znDamperPct1,zoneSenAvg:zoneSensorAvg,zoneSPAvg:zoneSpAvg,siteSuggestion:siteSuggestion1,action:action1}
end


end
