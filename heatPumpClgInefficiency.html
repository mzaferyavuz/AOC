(equipment,dates,tempThreshold:2,minTime:30min,maxTime:60min,freq:2,coolCondition:"off",zoneTempMinVal:10,zoneTempMaxVal:35,zoneTempSPMinVal:18,zoneTempSPMaxVal:28,check:false,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

        equipWorkOrder:toEquipWorkOrder(equipment)

        znTemp:read(zone and temp and sensor and equipRef == equipment->id)
        znTempSp:read(zone and temp and sp and equipRef == equipment->id)

        znTempHis:znTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>zoneTempMinVal and v<zoneTempMaxVal))
        znTempSpHis:znTempSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>zoneTempSPMinVal and v<zoneTempSPMaxVal))

        reverseVal:read(reversing and valve and cmd and equipRef == equipment->id,false)
        if (reverseVal==null) throw "No Reversing Valve"

        issue:comparepoints(znTemp,znTempSp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))

        onPeriods:equipment.toEquipRunSensor().hisRead(dates).hisFindPeriods(v=>v!=na() and v==true)

        maintanancePoint:equipment.tomaintenancePoint()
        occupancyPoint:read(occupied and equipRef == equipment->id,false)

        if(coolCondition=="on") heatPeriods:reverseVal.hisRead(dates).hisFindPeriods(v=>v!=na() and v==true)
          else if(coolCondition=="off") heatPeriods:reverseVal.hisRead(dates).hisFindPeriods(v=>v!=na() and v==false)
          else throw "Invalid condition defined for Reversing Valve"

        coolcmd:readAll(cool and cmd and kind=="Bool" and equipRef == equipment->id)
        if (coolcmd.size>0) coolPeriods:ahuCoolPeriods_inter(equipment,dates) //Compressor check
        else throw "Compressor Point is not present"

      spark:hisPeriodIntersection([issue,onPeriods,coolPeriods,heatPeriods,znTempHis,znTempSpHis])


        if(maintanancePoint!=null)
          do
            maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
            spark= hisPeriodIntersection([spark,maintanceHis])
          end


          if(occupancyPoint!=null and check== true and spark.size>0 )
        do
        occupiedperiods:togetOccupancy(equipment,dates)
        spark= hisPeriodIntersection([spark,occupiedperiods])
        end

       spark1:spark.findAll(ts => ts->v0 > minTime)
        spark2:spark.findAll(ts => ts->v0 > maxTime)

        if(spark1.size>=freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
      else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})


      end
