(dates,emailId,minroll:150,dlDelay:24hr) => do

  out:[].toGrid

  energypoint:readAll(energy and point and ebiref and (equipRef->submeterOf or equipRef->siteMeter)).each row=> do

  prv:null
  cur:null

  ////spike down not roll over
  a:row.hisRead(dates,{-hisTotalized}).map ro=>do

    cur=ro
    if(prv==null) prv=ro
    if(ro["v0"]!=na() and prv["v0"]!=na())  diff:(cur["v0"]-prv["v0"])

    if(ro["v0"]!=na() and diff<0 and ro["v0"]!=null and ro["v0"]>minroll) do
      val:prv
      prv=cur
      {dis:row->dis,current:cur["v0"],previous:val["v0"],difference:diff,cause:"Spike caused due to data value dropping"}
      end

    else do
      prv=cur
      {dis:row->dis,current:null,previous:null,difference:null,cause:null}
      end

    end.findAll(v=> v["current"]!=null)

  if(a.size>0 and out.size==0) out=a
  else if(a.size>0 and out.size>0) out=addRows(out,a)


  ////spike up
  cur1:null
  prv1:null
  weeklyavg:row.hisRead(lastMonth).hisFindAll(v=> v!=na()).hisRollup(sum,1day).hisRollup(avg,1mo).toRec
  if(weeklyavg!=null) weeklyavg=weeklyavg["v0"] else weeklyavg=1000000

  b:row.hisRead(dates,{-hisTotalized}).map ro=>do
    cur1=ro
    if(prv1==null) prv1=ro
    if(ro["v0"]!=na() and prv1["v0"]!=na()) diff:(ro["v0"]-prv1["v0"]) else diff:0

    if(ro["v0"]!=na() and diff>weeklyavg and ro["v0"]!=null and ro["v0"]>minroll) do
      val:prv1
      prv1=cur1
      {dis:row->dis,current:ro["v0"],previous:val["v0"],difference:diff,cause:"Spike caused due to data Spike"}
      end

    else do
      prv1=cur1
      {dis:row->dis,current:null,previous:null,difference:null,cause:null}
      end

    end.findAll(v=> v["current"]!=null)

  if(b.size>0 and out.size==0) out=b
  else if(b.size>0 and out.size>0) out=addRows(out,b)


  ////long data loss
  equip:readById(row->equipRef)
  dl:equip["dlTime"]

  if(row["hisEnd"]!=null) lag:(now()-row->hisEnd).to(1hr) else lag:1min

  if(lag>6hr and row["hisEnd"]!=null) diff(equip,{dlTime:row["hisEnd"]}).commit

  else if(lag<6hr and row["hisEnd"]!=null and dl!=null) diff(equip,{-dlTime}).commit

  if(dl!=null) do

    dltime:(row["hisEnd"]-dl).to(1hr)

    dlMissing: row.hisRead(row["hisEnd"]..dl).hisFindAll(v=> v!=na())

    differences:(row.hisRead(row["hisEnd"],{-hisTotalized}).toRec["v0"]-row.hisRead(dl,{-hisTotalized}).toRec["v0"])

    if(dltime>dlDelay and dlMissing.size==2) c:[{dis:row->dis,current:null,previous:null,difference:differences,cause:"Spike caused due to long data loss - the data stopped at "+ dl +" before starting today"}].toGrid else c:[].toGrid

    if(c.size>0 and out.size==0) out=c
    else if(c.size>0 and out.size>0) out=addRows(out,c)
    end

  ////end of long data loss

  end//// end of each
  if(out.size>0) out=out.reorderCols(out.colNames.moveTo("dis", 0))

  if(out.size>0) do

  out.ioWriteExcel(`io/energySpike.xls`)

  allcount:out.size

  a1:out.findAll(v=> v->cause.contains("Spike caused due to data value dropping"))

  if(a1.size>0) list1: "The following are having issue with data value drop "+ a1.colToList("dis") else list1:""

  a2:out.findAll(v=> v->cause.contains("Spike caused due to data Spike"))

  if(a2.size>0) list2: "The following are having data spike "+ a2.colToList("dis") else list2:""

  a3:out.findAll(v=> v->cause.contains("Spike caused due to long data loss"))

  if(a3.size>0) list3: "The following are having data loss issue In-between "+ a3.colToList("dis") else list3:""


    project:read(projMeta)->dis
    subject:project+" Energy spike Report "+now().date

    body:
     "<html><body>\n"+
       "<b1>EO Summary</b1>\n"+
       "<p> Total number of meters having issue are "+allcount+" </p>"+
       "<p> Please find the attached list of poitns and the details </p>"+
       "<p>  "+list1+" </p>"+
       "<p>  "+list2+" </p>"+
       "<p>  "+list3+" </p>"+
       "<p><small><b> Please do not reply to this email. This is an unmonitored mailbox </b></small></p>"+
     "</body></html>\n"

       "</body></html>\n"

    emailSend(emailId,subject,body,`io/energySpike.xls`)
  end

  end
