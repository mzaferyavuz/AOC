src

(delay:6hr,email) => do

   sz:readCount(point and his and ebiref)
   list:dataLossList()
   projpointsize:readAll(point and his and ebiref).size

   logInfo("dataloss",""+sz)
   sitetz:read(site)->tz

   //Checking history write failure

   hisFailSize:jobStatusAll().findAll(a=>(a->progressMsg.contains("Func failed: hisWrite"))and (a->started>dateTime(today(),00:00))).size

   if (hisFailSize>0) return list.dataLossSend("historyFailed","connectedbuildingsupport@honeywell.com")

    noOfPts:list.size //size of current data loss

   dayNow:now().toTimeZone(sitetz).format("WWWW")

   if(dayNow=="Monday")do            //Check day if Monday--send weekly report

   chekWeek:ioDir(`io/`).findAll(a=>a->name=="dataLossWeeklySheet.xls")

   if(chekWeek.size==0)do

   list.ioWriteExcel(`io/dataLossWeeklySheet.xls`)

      if(projpointsize==noOfPts) list.dataLossSend("allpoints",email)
   else list.dataLossSend("partial",email)end

   else if(chekWeek.size>0)do

   checkWeeklyTime:ioDir(`io/`).findAll(a=>a->name=="dataLossWeeklySheet.xls").toRec["mod"].toTimeZone(sitetz).format("DDMMYYYY")
   nowTime:now().toTimeZone(sitetz).format("DDMMYYYY")

   if(checkWeeklyTime!=nowTime)do

   list.ioWriteExcel(`io/dataLossWeeklySheet.xls`)

   if(projpointsize==noOfPts) list.dataLossSend("allpoints",email)
   else list.dataLossSend("partial",email)end

   end

   end


   ioDLReport:ioDir(`io/`).findAll(a=>a->name=="dataLossSheet.csv") //Checking if old report present

   if (ioDLReport.size>0) do
           sizeOfioReport:ioReadCsv(`io/dataLossSheet.csv`).size
             if(noOfPts>sizeOfioReport) do
                 if(projpointsize==noOfPts) list.dataLossSend("allpoints",email)
                 else list.dataLossSend("partial",email)
                                         end

            else if (noOfPts<sizeOfioReport)do
                                    if(noOfPts==0)list.dataLossSend("recovered",email)
                                     else list.dataLossSend("resuming",email)
                                            end
                          end

    else if (ioDLReport.size==0) do
                if(projpointsize==noOfPts) list.dataLossSend("allpoints",email)
                 else list.dataLossSend("partial",email)
                                end
   end
