(equip,dates,preheatThreshold:50,mixedTemp:10,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Operations") => do

spark1:[].toGrid
spark2:[].toGrid
spark:[].toGrid

equipWorkOrder:toEquipWorkOrder(equip)
maintanancePoint:equip.tomaintenancePoint()

preheatValve:read(preheat and valve and cmd and equipRef==equip->id).hisRead(dates).hisFindPeriods(v=>(v!=na() and v<preheatThreshold))

mat:read(mixed and air and temp and sensor and equipRef==equip->id,false)
preheatOff:read(preheat  and temp and sensor and equipRef==equip->id,false)
ahuStatus:read(discharge and fan and run and sensor and kind=="Bool" and equipRef==equip->id)
preheatsp:read(preheat and temp and sp and equipRef==equip->id,false)
matsp:read(mixed and air and temp and sp and equipRef==equip->id,false)

if(mat!=null and preheatOff!=null) mixed:hisRead([mat,preheatOff],dates).hisFoldCols(max).hisFindPeriods(v=>(v!=na() and v<mixedTemp))
else if(mat!=null and preheatOff==null) mixed:mat.hisRead(dates).hisFindPeriods(v=>(v!=na() and v<mixedTemp))
else if(mat==null and preheatOff!=null) mixed:preheatOff.hisRead(dates).hisFindPeriods(v=>(v!=na() and v<mixedTemp))
else if(mat==null and preheatOff==null) throw "No Mixed OR PreHeat Temps"

equipOn:ahuStatus.hisRead(dates).hisFindPeriods(v=> (v!=na() and v==true))

spark=hisPeriodIntersection([equipOn,mixed,preheatValve])
spark_inter:spark

if(maintanancePoint!=null)
do

maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
spark_inter= hisPeriodIntersection([spark,maintanceHis])
end

spark1=spark_inter.findAll(ts => ts->v0 > minTime)
spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
end
