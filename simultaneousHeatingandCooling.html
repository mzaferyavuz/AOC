(ahu,dates,coolthreshold:10,heatthreshold:10,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

  equipWorkOrder:toEquipWorkOrder(ahu)
  //Get periods when cooling and heating
  boilerPlant:ahu["boilerPlantRef"]
  chillerPlant:ahu["chillerPlantRef"]
  ahuOnTime:ahu.ahuRunPeriods(dates)
  spark:[].toGrid
  occupancyPoint:read(occupied and cmd and equipRef == ahu->id,false)
  maintanancePoint:ahu.tomaintenancePoint()


  cool: ahuCoolPeriods(ahu, dates,coolthreshold)
  heat: ahuHeatPeriods(ahu, dates,heatthreshold)

  if(chillerPlant==null and boilerPlant==null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime])

 if(chillerPlant!=null and boilerPlant==null)
 do

 chilleronperiod: chillerPlant.anyChilleron(dates)

if(chilleronperiod==null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime])

 else
 spark=hisPeriodIntersection([cool,heat,ahuOnTime,chilleronperiod])

end

 if(chillerPlant==null and boilerPlant!=null)
 do
 boileronperiod: boilerPlant.anyboileron(dates)
if(boileronperiod==null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime])

else
 spark=hisPeriodIntersection([cool,heat,ahuOnTime,boileronperiod])

end

 if(chillerPlant!=null and boilerPlant!=null)
 do
 boileronperiod: boilerPlant.anyboileron(dates)
 chilleronperiod: chillerPlant.anyChilleron(dates)

if(boileronperiod==null and chilleronperiod!=null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime,chilleronperiod])

if(boileronperiod!=null and chilleronperiod==null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime,boileronperiod])

if(boileronperiod==null and chilleronperiod==null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime])

if(boileronperiod!=null and chilleronperiod!=null)
 spark=hisPeriodIntersection([cool,heat,ahuOnTime,boileronperiod,chilleronperiod])
end

spark_inter:spark
  if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter= hisPeriodIntersection([spark_inter,maintanceHis])
    end
 if(occupancyPoint!=null )
    do
    occupiedperiods:togetOccupancy(ahu,dates)
    spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
    end

  spark1:spark_inter.findAll(ts => ts->v0 > minTime)
  spark2:spark_inter.findAll(ts => ts->v0 > maxTime)

  if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})



  end
