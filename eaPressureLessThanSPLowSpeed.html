src

(ahu,dates,pressDb:10,fanThreshold:85%,cmdFBMismatch:10,disPresMin:1,disPresMax:600,disPreSpMin:100,disPresSpMax:600,maxTime:60min,minTime:30min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Comfort") => do

    equipWorkOrder:toEquipWorkOrder(ahu)
    spark:[].toGrid()

    disPres:read(exhaust and air and pressure and sensor and equipRef==ahu->id)
    disPresSp:read(exhaust and air and pressure and sp and equipRef==ahu->id)

    disPresWithinLim:disPres.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>disPresMin and v<disPresMax) )
    disPresSpWithinLim:disPresSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>disPreSpMin and v<disPresSpMax) )

    lowTime:hisRead([disPres,disPresSp],dates).hisInterpolate().map row=>do

   if(row["v0"]!=na() and row["v1"]!=na())
                   {ts:row->ts,v0:(row["v1"] - row["v0"])>(row["v1"]*(pressDb/100))}
              else {ts:row->ts,v0:na()}

                   end.hisFindPeriods(v=>v!=na() and v==true)


    fanStatusOn:read(exhaust and fan and run and sensor and equipRef==ahu->id).hisRead(dates).hisFindPeriods(v=>v!=na() and v==true)

    fanCmdOn:read(exhaust and fan and run and cmd and equipRef==ahu->id,false)
    if(fanCmdOn!=null) do
    fanCmdOnPeriods:fanCmdOn.hisRead(dates).hisFindPeriods(v=>v!=na() and v==true)
    fanStatus:hisPeriodIntersection([fanStatusOn,fanCmdOnPeriods])
    end
    else fanStatus:fanStatusOn

    fanSpeedCmd:read(exhaust and fan and speed and cmd and equipRef==ahu->id)
    fanSpeedCmdHigh:fanSpeedCmd.hisRead(dates).hisFindPeriods(v=>v!=na() and v<fanThreshold)

    fanSpeedSensor:read(exhaust and fan and speed and sensor and equipRef==ahu->id,false)
    if(fanSpeedCmd!=null and fanSpeedSensor!=null)do
    vsdCmdFBMisMatch:vsdCmdFeedbackMismatch_inter(fanSpeedCmd,fanSpeedSensor,dates,cmdFBMismatch)
    spark=hisPeriodIntersection([vsdCmdFBMisMatch,fanSpeedCmdHigh,fanStatus,lowTime,disPresWithinLim,disPresSpWithinLim])
    end
    else spark=hisPeriodIntersection([fanSpeedCmdHigh,fanStatus,lowTime,disPresWithinLim,disPresSpWithinLim])

      maintanancePoint:ahu.tomaintenancePoint()
       occupancyPoint:read(occupied and equipRef == ahu->id,false)

   if(maintanancePoint!=null)
     do
       maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
       spark=hisPeriodIntersection([spark,maintanceHis])
     end
   if(occupancyPoint!=null)
     do
        occupiedperiods:togetOccupancy(ahu,dates)
          if(occupiedperiods.size>0)
            do
              spark=hisPeriodIntersection([spark,occupiedperiods])
            end
     end

        spark1:spark.findAll(ts => ts->v0 > minTime)
        spark2:spark.findAll(ts => ts->v0 > maxTime)

     if(spark1.size>freq)
       do
       return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
       end
       else if(spark2.size>0)
       do
      return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
     end
  end
