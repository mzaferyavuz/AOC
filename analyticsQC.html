() => do
ioDelete(`io/qcchecklog.txt`)
file:ioAppend(`io/qcchecklog.txt`)
ioWriteStr("-----QC Analytics Checks----- \n",file)

totalPoint:readAll(point and not weather).size
totalEquip:readAll(equip).size

ioWriteStr(""+totalPoint+"->Total Point Configured"+"\n",file)
ioWriteStr(""+totalEquip+"->Total Equipments Configured"+"\n",file)
controlsBasic:readAll(rule and controlsBasic).size
controlsStandard:readAll(rule and controlsStandard).size
energy:readAll(rule and energy).size
mechanical:readAll(rule and mechanical).size

ioWriteStr(""+controlsBasic+"==='controlsBasic' Rules Added"+"\n",file)
ioWriteStr(""+controlsStandard+"==='controlsStandard' Rules Added"+"\n",file)
ioWriteStr(""+energy+"==='Energy' Rules Added"+"\n",file)
ioWriteStr(""+mechanical+"==='Mechanical' Rules Added"+"\n",file)
ioWriteStr("******ERROR CHECKS******* \n",file)

errorCount:0

sites:readAll(site)
sites.each site =>do

if(site["area"]==null) do
ioWriteStr(""+site->dis+"->The site does not have 'area' tag"+"\n",file)
errorCount=errorCount+1
end

if(site["area"]!=null) do
ioWriteStr("Site Name:"+" "+site->dis+" "+"Area: "+site["area"]+""+"\n",file)
end

if(site["weatherStationRef"]==null) do
ioWriteStr(""+site->dis+"->The site does not have 'weatherStationRef' tag"+"\n",file)
errorCount=errorCount+1
end
end

numbers:readAll(point and not weather and kind=="Number")
numbers.each num=>do

if(num["unit"]==null) do
ioWriteStr(""+num->dis+"->Unit Missing for Number Point"+"\n",file)
errorCount=errorCount+1
end
end

bools:readAll(point and not weather and unit and kind=="Bool")
if(bools.size>0) bools.each bool=> do

ioWriteStr(""+bool->dis+"->This Boolean point has a 'Unit' tag"+"\n",file)
errorCount=errorCount+1
end

hisAppendNA:readAll(point and not hisAppendNA and not weather)
if(hisAppendNA.size>0) hisAppendNA.each hisAppend=>do

ioWriteStr(""+hisAppend->dis+"->The point does not have 'hisAppendNA' tag"+"\n",file)
errorCount=errorCount+1
end

pointErr:readAll(point and not weather and hisFunc and ebiref)
if(pointErr.size>0) pointErr.each eType=>do

ioWriteStr(""+eType->dis+"->The points has both ebiref and hisFunc"+"\n",file)
errorCount=errorCount+1
end

temp:readAll(temp and point and not weather).findAll(a=>a["unit"]!="°F" and a["unit"]!="°C")
if(temp.size>0) temp.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Temperature"+"\n",file)
errorCount=errorCount+1
end

pressure:readAll(pressure and point).findAll(a=>a["unit"]!="Pa" and a["unit"]!="kPa" and a["unit"]!="psi" and a["unit"]!="inH₂O")
if(pressure.size>0) pressure.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Pressure Point"+"\n",file)
errorCount=errorCount+1
end

humid:readAll(humidity and point and not weather).findAll(a=>a["unit"]!="%")
if(humid.size>0) humid.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Humidity"+"\n",file)
errorCount=errorCount+1
end

ppm:readAll(point and(co or co2)).findAll(a=>a["unit"]!="ppm")
if(ppm.size>0) ppm.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for CO/CO2 Sensor"+"\n",file)
errorCount=errorCount+1
end

if(read(chillerPlant,false)!=null)
do
readAll(equip and not chillerPlantRef).findAll(a=> readAll(cool and cmd and kind=="Number" and equipRef==a->id).size>0).each eq=>do
eqp:eq->dis
ioWriteStr(""+eqp+"->This equip missing chillerPlantRef tag "+"\n",file)
errorCount=errorCount+1
end
readAll(equip and chillerPlantRef).findAll(a=> readAll(cool and cmd and kind=="Bool" and equipRef==a->id).size>0).each eq=>do
eqp:eq->dis
ioWriteStr(""+eqp+"->This equip has chillerPlantRef tag with DX cooling"+"\n",file)
errorCount=errorCount+1
end
end

if(read(boilerPlant,false)!=null)
do
readAll(equip and not boilerPlantRef).findAll(a=> readAll((heat or reheat) and cmd and kind=="Number" and equipRef==a->id).size>0).each eq=>do
eqp:eq->dis
ioWriteStr(""+eqp+"->This equip missing boilerPlantRef tag "+"\n",file)
errorCount=errorCount+1
end
readAll(equip and boilerPlantRef).findAll(a=> readAll((heat or reheat) and cmd and kind=="Bool" and equipRef==a->id).size>0).each eq=>do
eqp:eq->dis
ioWriteStr(""+eqp+"->This equip has boilerPlantRef tag with binary heating"+"\n",file)
errorCount=errorCount+1
end
end

speed:readAll(point and speed).findAll(a=>a["unit"]!="%"and a["unit"]!="Hz" and a["unit"]!="rpm")
if(speed.size>0) speed.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Speed Point"+"\n",file)
errorCount=errorCount+1
end

ener:readAll(point and energy).findAll(a=>a["unit"]!="kWh")
if(ener.size>0) ener.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Energy Point"+"\n",file)
errorCount=errorCount+1
end

flow:readAll(point and flow).findAll(a=>a["unit"]!="L/s"and a["unit"]!="cfm" and a["unit"]!="cfm" and a["unit"]!="gal/min" and a["unit"]!="m³/s")
if(flow.size>0) flow.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Flow"+"\n",file)
errorCount=errorCount+1
end

gas:readAll(point and gas).findAll(a=>a["unit"]!="GJ"and a["unit"]!="m³")
if(gas.size>0) gas.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Gas meters"+"\n",file)
errorCount=errorCount+1
end

eTypeR:readAll(point and not eTypeRef and not weather)
if(eTypeR.size>0) eTypeR.each a=>do

ioWriteStr(""+a->dis+"->There is no eTypeRef for this point"+"\n",file)
errorCount=errorCount+1
end

power:readAll(point and power).findAll(a=>a["unit"]!="W"and a["unit"]!="kW")
if(power.size>0) power.each a=>do

ioWriteStr(""+a->dis+"->INCORRECT UNIT for Power"+"\n",file)
errorCount=errorCount+1

end
stringpointErr:readAll(point and not weather and kind=="Str" and not hisOnWrite)
if(stringpointErr.size>0) stringpointErr.each eType=>do

ioWriteStr(""+eType->dis+"->The points has kind as 'Str' and no hisOnWrite"+"\n",file)
errorCount=errorCount+1
end

if(duplicateAssetPoint("point")!=null) duplicatePoint:duplicateAssetPoint("point").each dupli=>do
point:dupli->name
ioWriteStr(""+point+"->This is duplicate Point"+"\n",file)
errorCount=errorCount+1
end

if(duplicateAssetPoint("equip")!=null) duplicateAsset:duplicateAssetPoint("equip").each dupli=>do
asset:dupli->name
ioWriteStr(""+asset+"->This is duplicate Asset"+"\n",file)
errorCount=errorCount+1
end

sparkDebug:ruleDebug("rules").findAll(a=>a->ruleStatus=="fault")
if(sparkDebug.size>0) sparkDebug.each debug=>do
name:debug->id.refDis
ioWriteStr(""+name+"->This Rule is in FAULT Condition"+"\n",file)
errorCount=errorCount+1
end

readAll(rule and not disabled).findAll(a=> ruleDebug("rule",a->id).size==0).sort("dis").each rul=>do
rule:rul->dis
ioWriteStr(""+rule+"->This has zero target"+"\n",file)
errorCount=errorCount+1
end

readAll(rule and not disabled).findAll(a=> ruleDebug("rule",a->id).findAll(a=> a->ruleStatus!="err").size==0).sort("dis").each rul=>do
rule:rul->dis
ioWriteStr(""+rule+"->This has all the target in Error"+"\n",file)
errorCount=errorCount+1
end

if(comfortAoc().size==0)
ioWriteStr("Comfort check is empty"+"\n",file)
errorCount=errorCount+1

if(co2Aoc().size==0 and read(zone and co2,false)!=null)
ioWriteStr("CO2 check is empty"+"\n",file)
errorCount=errorCount+1

if(humidityAoc().size==0 and read(zone and humidity and sp,false)!=null)
ioWriteStr("Humidity check is empty"+"\n",file)
errorCount=errorCount+1

lintErrors:lint(null)
if(lintErrors.size>0) lintErrors.each lint=>do

ioWriteStr("*LINT ERROR:*"+lint->id.refDis+"->"+lint->msg+""+"\n",file)
errorCount=errorCount+1
end

ioWriteStr(""+errorCount+" Errors Found"+"\n",file)
ioWriteStr("******JOB Status******* \n",file)

readAll(job and not disabled).each row=> do
ioWriteStr("*jobExpr:*"+" "+row->dis+" "+"Status: "+row["jobLastStatus"]+""+"\n",file)
end

ioWriteStr("---End of Checks---\n",file)
ioReadLines(file)
end
