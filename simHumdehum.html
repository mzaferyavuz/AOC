src

(ahu,dates,tempThreshold:2,threshold:10,humThreshold:5,coolingValveThreshold:10,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

equipWorkOrder:toEquipWorkOrder(ahu)
spark:[].toGrid()

humidity:read(zone and humidity and sensor and equipRef == ahu->id,false)
if(humidity==null) humidity=read("return" and humidity and sensor and equipRef == ahu->id,false)
if(humidity==null) throw("No humidity sensor point in:")+dis(ahu)
humidity=humidity.hisRead(dates)

setpoint:read(zone and humidity and sp and equipRef == ahu->id,false)
if(setpoint==null) setpoint=read("return" and humidity and sp and equipRef == ahu->id,false)
if(setpoint==null) throw("No humidity sp point in:")+dis(ahu)
setpoint=setpoint.hisRead(dates)

command:read(humidifier and cmd and equipRef == ahu->id,false)
if(command ==null) throw("No Humidifier cmd point defined in:")+dis(ahu)
history:command.hisRead(dates)

if(command->kind == "Bool") humidTime:history.hisFindPeriods(v=>(v!=na() and v==true))
else humidTime:history.hisFindPeriods(v =>(v!=na() and v > threshold))


coolPeriods:ahuCoolPeriods_inter(ahu,dates,coolingValveThreshold)

notMeetingTime:hisJoin([setpoint,humidity]).hisInterpolate().map row=>do
if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:((abs(row["v1"]-row["v0"])/100*row["v1"])>(humThreshold*(row["v0"].as(1))))}
else {ts:row->ts,v0:na()}
end.hisFindPeriods(v=>(v!=na() and v==true))

onTime:ahu.ahuRunPeriods(dates)

datsp:read(discharge and temp and sp and equipRef==ahu->id,false)
zoneTempSp:read(zone and temp and sp and equipRef==ahu->id,false)

if(datsp!=null) do
dat:read(discharge and temp and sensor and equipRef==ahu->id)
issue:comparepoints(datsp,dat,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
end
else if(zoneTempSp!=null) do
zoneTemp:read(zone and temp and sensor and equipRef==ahu->id)
issue:comparepoints(zoneTempSp,zoneTemp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
end

spark=hisPeriodIntersection([issue,notMeetingTime,coolPeriods,humidTime,onTime])

maintanancePoint:ahu.tomaintenancePoint()
occupancyPoint:read(occupied and equipRef == ahu->id,false)

if(maintanancePoint!=null)
do
maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
spark=hisPeriodIntersection([spark,maintanceHis])
end
if(occupancyPoint!=null)
do
occupiedperiods:togetOccupancy(ahu,dates)
if(occupiedperiods.size>0)
do
spark=hisPeriodIntersection([spark,occupiedperiods])
end
end

spark1:spark.findAll(ts => ts->v0 > minTime)
spark2:spark.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
