src

() => do

readAll(rule and not disabled).map r=>do

name:read(projMeta)->dis
ids:read(projMeta)["name"]
priority:"None"
typ:"None"
fnc:r->ruleFunc.split("(")[0]
function:read(func and name==fnc,false)

if(r["nameType"]=="Standard Name") nType:"Standard Name" else nType:"Non Standard Name"
if(r["ruleLibrary"]=="Extension Rule") rType:"Extension Rule" else rType:"Custom Rule"

if(r->ruleFunc.contains("\"High\"")==true) priority="High" else if(r->ruleFunc.contains("\"Medium\"")==true) priority="Medium" else if (r->ruleFunc.contains("\"Low\"")==true) priority="Low"
else if(r->ruleFunc.contains("\"high\"")==true) priority="High" else if(r->ruleFunc.contains("\"medium\"")==true) priority="Medium" else if (r->ruleFunc.contains("\"low\"")==true) priority="Low"

if(priority=="None" and function!=null)
do
if(function->src.contains("priority:\"High\"")==true) priority="High" else if (function->src.contains("priority:\"Medium\"")==true) priority="Medium" else if (function->src.contains("priority:\"Low\"")==true) priority="Low"
else if(function->src.contains("priority:\"high\"")==true) priority="High" else if (function->src.contains("priority:\"medium\"")==true) priority="Medium" else if (function->src.contains("priority:\"low\"")==true) priority="Low"
end

if(r->ruleFunc.contains("\"Energy\"")==true) typ="Energy" else if(r->ruleFunc.contains("\"Operations\"")==true) typ="Operations" else if (r->ruleFunc.contains("\"Comfort\"")==true) typ="Comfort"
else if(r->ruleFunc.contains("\"energy\"")==true) typ="Energy" else if(r->ruleFunc.contains("\"operations\"")==true) typ="Operations" else if (r->ruleFunc.contains("\"comfort\"")==true) typ="Comfort"

if(typ=="None" and function!=null)
do
if(function->src.contains("\"Energy\"")==true) typ="Energy" else if(function->src.contains("\"Operations\"")==true) typ="Operations" else if (function->src.contains("\"Comfort\"")==true) typ="Comfort"
else if(function->src.contains("\"energy\"")==true) typ="Energy" else if(function->src.contains("\"operations\"")==true) typ="Operations" else if (function->src.contains("\"comfort\"")==true) typ="Comfort"
end

nm:""
if(nmm:funcs(name==r->ruleFunc).size>0) nm=r->ruleFunc
else nm=r->ruleFunc.split("(")[0]

{projectName:name,projectId:ids,rule:r->dis,nameType:nType,ruleLibrary:rType,priority:priority,type:typ,funcName:nm}

end.sort("rule")
end
