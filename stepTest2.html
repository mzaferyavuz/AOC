() => do

  readAll(applied and eo and sp).map row=>do

  appVal:read(id==row->id).hisRead(now()).table[0]->v0
  appts:read(id==row->id).hisRead(now()).table[0]->ts

  eoSP:read(id==row->id)["eoSpRef"]
  eospval:eoSP.hisRead(now()).table[0]->v0
  eots:eoSP.hisRead(now()).table[0]->ts

  eoSwitch:read(id==row->id)["switchRef"]
  eostat:eoSwitch.hisRead(now()).table[0]->v0
  swiTs:eoSwitch.hisRead(now()).table[0]->ts


  eqp:read(equip and dis==row->equipRef->dis)

  if(eqp.has("chillerPlant")==true)do

  eqRunSen:read(enable and cmd and equipRef==row->equipRef,false)

  if(eqRunSen==null)eqRunSenVal:"No enable for the plant"
  else if(eqRunSen!=null)eqRunSenVal:eqRunSen.hisRead(now()).table[0]->v0

  end


  else do
  eqRunSen:read(discharge and fan and run and sensor and equipRef==row->equipRef)
  eqRunSenVal:eqRunSen.hisRead(now()).table[0]->v0
  end

  if(eqRunSenVal==false)comm:"Equip is OFF"

  else if(eqRunSenVal==true or eqRunSenVal=="No enable for the plant")do

            if(appts==eots)do
                              if(eots==swiTs)do
                                           if(eostat!=true)comm:"EO SWITCH IS 'NOT' ON"
                                            else if(eostat==true)do
                                                      if(eospval==appVal)comm:"ALL CRITERIAS MET"
                                                      else if(eospval!=appVal)comm:"EO SP != APPLIED SP"
                                                      else if(eospval==na() or appVal==na())comm:"Points reading NA Value"
                                                              end
                                              else if(eostat==na())comm:"Points reading NA Value"
                                             end
                                  else if(eots!=swiTs)comm:"TimeStamp not in Sync"
                         end
              else if(appts!=eots)comm:"TimeStamp not in Sync"
  end

  {eo:row->ebiref,equipment:row->equipRef->dis,remark:comm,equipStatus:eqRunSenVal,appliedlastVal:appVal,appliedTs:appts,eoSPlastVal:eospval,eoSPTs:eots,eostatus:eostat,eoStatTs:swiTs}

  end

  end
