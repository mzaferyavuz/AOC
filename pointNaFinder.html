(point,dates,rec,dataLossTime:60min,naTime:360min,naRate:2) => do

  pnts:readAll(point and ebiref and hisEnd and dis==point->dis).map row => do
  nat:[].toGrid
  nat=row.hisRead(dates).hisClip.hisFindPeriods((v, ts) =>(v==na() and (ts<row->hisEnd)))

    if(nat!=null) do
      nat=nat.findAll(ts => ts->v0 <naTime)

      if(nat.size>0) do

        st:row["siteRef"]
        natsz:nat.size
        nat1:nat[-1]

        spanend:(nat1.toRec["ts"]+nat1.toRec["v0"])==row->hisEnd

        if(row["hisEnd"]!=null) lag:(now().toTimeZone(row->tz)-row->hisEnd).to(1hr)
        else lag:25hr

        tot:row.hisRead(dates).size

        if(spanend==true) natsz=(natsz-1)
        if(spanend==true) nas:row.hisRead(dates).hisFindAll((v, ts)=> (v==na() and ts< nat1.toRec["ts"])).size
        else nas:row.hisRead(dates).hisFindAll(v=> v==na()).size

        pect:(nas/tot)*100.as(1%)

        if(natsz>naRate and lag<dataLossTime) {dis:row->dis,ebiref:row->ebiref,issue:"yes",ratio:pect,instance_count:nas,site:st}
      end
    end

  end.sort("dis").findAll(v=> v->issue=="yes")


  ////reading na

  grid:[].toGrid

  list:readAll(point and his and ebiref and hisEnd and dis==point->dis).each row=>do
  tz:row->tz

  grid1:row.hisRead(dates).hisFindPeriods(v=>(v==na()))

  if(grid1!=null) do

    if(grid1.size>0) do
      grid1=grid1.findAll(ts=>ts->v0>naTime)

      if(grid1.size>0) do
        point:{dis:row->dis,site:row->siteRef,equipName:row->equipRef,ebiRef:row->ebiref,projName:read(projMeta)->dis}
        if(row["hisEnd"]!=null) lag:(now().toTimeZone(tz)-row->hisEnd).to(1hr) else lag:25hr
        if(lag<dataLossTime) grid=grid.addRows(point.toGrid)
        end
      end

    end

  end


  if(rec=="inter" and pnts.size>0) return out:true else return out:false

  if(rec="na" and grid.size>0) return out:true else return out:false

  end
