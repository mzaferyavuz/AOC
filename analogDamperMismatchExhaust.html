(equipment,dates,threshold:10,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Operations") => do

  equipWorkOrder:toEquipWorkOrder(equipment)

  outAnalog:read(exhaust and damper and cmd and kind=="Number" and equipRef == equipment->id,false)
  fbAnalog:read(exhaust and damper and sensor and kind=="Number" and equipRef == equipment->id,false)

  outBool:read(exhaust and damper and cmd and kind=="Bool" and equipRef == equipment->id,false)
  fbBool:read(exhaust and damper and sensor and kind=="Bool" and equipRef == equipment->id,false)

  if(fbAnalog!=null and outAnalog==null) throw "No Analog Damper Command"
  if(fbAnalog==null and outAnalog!=null) throw "No Analog Damper Status"

  if(fbBool!=null and outBool==null) throw "No Bool Damper Command"
  if(fbBool==null and outBool!=null) throw "No Bool Damper Status"

  maintanancePoint:equipment.tomaintenancePoint()
  occupancyPoint:read(occupied and cmd and equipRef == equipment->id,false)

  if(fbAnalog!=null and outAnalog!=null) do

  issue:hisRead([outAnalog,fbAnalog],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:(abs((row["v1"])-(row["v0"]))>=threshold)}
  else {ts:row->ts,v0:na()}
  end.hisFindPeriods(v=>v!=na() and v==true)

  else if(fbBool!=null and outBool!=null) do

  issue:hisRead([outBool,fbBool],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na())
  {ts:row->ts,v0:((row["v0"])!=(row["v1"]))}
  else {ts:row->ts,v0:na()}
  end.hisFindPeriods(v=>v!=na() and v==true)
  end

  spark:issue

  if(maintanancePoint!=null)
  do
  maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
  spark= hisPeriodIntersection([spark,maintanceHis])
  end

  spark1:spark.findAll(ts => ts->v0 > minTime)
  spark2:spark.findAll(ts => ts->v0 > maxTime)

  if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  end
