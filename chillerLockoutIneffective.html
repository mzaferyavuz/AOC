(chillerPlant,dates,oatLo:14,minTime:15min,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

 equipWorkOrder:toEquipWorkOrder(chillerPlant)
 chillerRunPeriods:chillerPlant.anyChillerOnInSystem(dates)
 coldOutsidePeriods:chillerPlant.toAhuOutsideTemp().hisRead(dates).hisFindPeriods(v=>(v<oatLo and v!=na()))

 hisPeriodIntersection([chillerRunPeriods,coldOutsidePeriods]).findAll(ts => ts->v0 >= minTime)
 .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
end
