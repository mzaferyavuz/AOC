(dates,factor:0.85,val:95,emailrec:false,emailId:userCur()->email) => do

grid:[].toGrid

zones:readAll(equip and eo and eoSp)//.findAll(v=> readAll(zone and temp and sensor and equipRef->dis==v->dis).size>0 and readAll(zone and temp and sp and equipRef->dis==v->dis).size>0).sort("dis")

aa:dates.eachDay d => do

rougezonesTemp:[].toGrid
rougezonesDemand:[].toGrid

  rz:zones.map row=> do

    temp:read(zone and temp and sensor and equipRef==row->id,false)

    if(row["ahu"]!=null) analog:read(cool and valve and cmd and equipRef==row->id)
    else if(row["vav"]!=null) analog:read(discharge and damper and cmd and equipRef==row->id)
    else analog:null
    eosp:row["eoSp"]

    if(row["ahu"]!=null or row["fcu"]!=null) rs:read(discharge and fan and run and sensor and equipRef==row->id)
    else if (row["vav"]!=null) rs:read(discharge and fan and run and sensor and equipRef==row->ahuRef->id)
    runtime:rs.hisRead(d).hisFindAll(v=> v!=na()).hisFindPeriods(v=> v!=na() and v==true)

    if(runtime.size>0) do

      runtime=runtime.hisRollup(sum,1day).toRec["v0"].to(1min)

    if(temp!=null) do

      rougezonesTemp=hisRead([temp,rs],d).hisInterpolate.map ro=>do
      if(ro["v0"]!=na() and ro["v1"]!=na()) {ts:ro->ts,v0:(ro["v0"]>eosp and ro["v1"]==true)}
      else {ts:ro->ts,v0:na()}
      end.hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1day).findAll(ts => ts->v0 > (runtime*factor))
      end

    if(analog!=null) do

      val1:row["eoVal"]

      if(val1!=null) val=val1
      if(val>99) val=99

      rougezonesDemand=hisRead([analog,rs],d).hisInterpolate.map ro=>do
      if(ro["v0"]!=na() and ro["v1"]!=na()) {ts:ro->ts,v0:(ro["v0"]>val and ro["v1"]==true)}
      else {ts:ro->ts,v0:na()}
      end.hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1day).findAll(ts => ts->v0 > (runtime*factor))
      end

    if(analog==null and temp!=null) do

      if(rougezonesTemp.size>0) {ts:d,equip:row->dis,rogueTemp:"Rogue",rogueDemand:"NA",totalRun:runtime}
      else if(rougezonesTemp.size==0 and rougezonesDemand.size>0) {ts:d,equip:row->dis,rogueTemp:"No Issue",rogueDemand:"NA",totalRun:runtime}
      end

    else if(analog!=null and temp==null) do

      if(rougezonesDemand.size>0) {ts:d,equip:row->dis,rogueTemp:"NA",rogueDemand:"Rogue",totalRun:runtime}
      else if(rougezonesDemand.size==0) {ts:d,equip:row->dis,rogueTemp:"NA",rogueDemand:"No Issue",totalRun:runtime}
      end

    else if(analog!=null and temp!=null) do

      if(rougezonesTemp.size>0 and rougezonesDemand.size>0) {ts:d,equip:row->dis,rogueTemp:"Rogue",rogueDemand:"Rogue",totalRun:runtime}
      else if(rougezonesTemp.size>0 and rougezonesDemand.size==0) {ts:d,equip:row->dis,rogueTemp:"Rogue",rogueDemand:"No Issue",totalRun:runtime}
      else if(rougezonesTemp.size==0 and rougezonesDemand.size>0) {ts:d,equip:row->dis,rogueTemp:"No Issue",rogueDemand:"Rogue",totalRun:runtime}
      else {ts:d,equip:row->dis,rogueTemp:"No Issue",rogueDemand:"No Issue",totalRun:runtime}
      end

  end

  end

  if(grid.size==0 and rz.size>0) grid=rz
  else if(grid.size>0 and rz.size>0) grid=addRows(grid,rz)
end

grid1:grid.findAll(v=> v->totalRun!=na() and v->totalRun>0min).findAll(v=> v->rogueDemand=="Rogue" or v->rogueTemp=="Rogue")
grid2:grid1

ab:grid1.map row=> do

if(grid2.findAll(v=> v->rogueDemand=="No Issue" and v->equip==row->equip).size>0) {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:"No Issue",totalRun:row->totalRun}
else if(grid2.findAll(v=> v->rogueTemp=="No Issue" and v->equip==row->equip).size>0) {ts:row->ts,equip:row->equip,rogueTemp:"No Issue",rogueDemand:row->rogueDemand,totalRun:row->totalRun}
else {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun}
end

ab.sort("equip").findAll(v=> v->totalRun>0min and (v->rogueDemand=="No Issue" or v->rogueDemand=="NA") and (v->rogueTemp=="No Issue" or v->rogueTemp=="NA")).unique("equip").keepCols(["equip"]).each eq=>do
ab=ab.findAll(v=> v->equip!=eq->equip)
end

ab

/////////////////////////////////////////////////////Rules part - to add the sparks generated////////////////////////////////////////////////////

sprk:navFilter("sparkTarget", null).sparks(dates)

abc:ab.map row=> do
a:sprk.findAll(v=> v->equipRef->dis==row->equip)

  if(a.size>0) do
  b:a.unique("ruleRef").map z=> {rul:readById(z->ruleRef)->dis}
  {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:b.colToList("rul") }
  end
    else {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:null}
end.unique("equip")

abcd:abc.map row=> do

a:readAll(equip and vav and dis==row->equip)
b:readAll(equip and not vav and dis==row->equip and (ahu or fcu))
c:read(equip and dis==row->equip)["chillerPlantRef"]
d:read(equip and dis==row->equip)["boilerPlantRef"]

////////////////////////////////////////////////////-----------------if equip is VAV
  if(a.size>0) do

  ahu:null
  ah:[].toGrid
  ahucheck:read(equip and dis==row->equip)["ahuRef"]

  if(ahucheck!=null) do

  ahu=readById(read(equip and dis==row->equip)["ahuRef"])->dis
  ah=sprk.findAll(v=> v->equipRef->dis==ahu).unique("ruleRef")
  if(ah.size>0) ah=ah.map z=>do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else ah=null

  chil:(readById(ahucheck)["chillerPlantRef"])->dis
  ch:sprk.findAll(v=> v->equipRef->dis==chil).unique("ruleRef")
  if(ch.size>0) ch=ch.map z=> do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else ch=null
  end

    if(d==null) {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:row["targetsSpark"],ahuName:ahu,ahuSpark:ah,chillerplant:chil,chillerplantSpark:ch,boilerplantSpark:null,boilerplantSpark:null}

    else if(d!=null) do
      boil:readById(read(equip and dis==row->equip)["boilerPlantRef"])->dis
      bh:sprk.findAll(v=> v->equipRef->dis==boil).unique("ruleRef")
      if(bh.size>0) bh=bh.map z=> do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else bh=null

      {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:row["targetsSpark"],ahuName:ahu,ahuSpark:ah,chillerplant:chil,chillerplantSpark:ch,boilerplant:boil,boilerplantSpark:bh}
      end
  end

////////////////////////////////////////////////////-----------------if equip is AHU / FCU
  else if(b.size>0) do

    if(c!=null and d==null) do
      chil:readById(read(equip and dis==row->equip)["chillerPlantRef"])->dis
      ch:sprk.findAll(v=> v->equipRef->dis==chil).unique("ruleRef")
      if(ch.size>0) ch=ch.map z=> do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else ch=null
      {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:row["targetsSpark"],ahuName:null,ahuSpark:null,chillerplant:chil,chillerplantSpark:ch,boilerplant:null,boilerplantSpark:null}
      end

    else if(c==null and d!=null) do
      boil:readById(read(equip and dis==row->equip)["boilerPlantRef"])->dis
      bh:sprk.findAll(v=> v->equipRef->dis==boil).unique("ruleRef")
      if(bh.size>0) bh=bh.map z=> do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else bh=null
      {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:row["targetsSpark"],ahuName:null,ahuSpark:null,chillerplant:null,chillerplantSpark:null,boilerplant:boil,boilerplantSpark:bh}
      end

    else if(c!=null and d!=null) do
      boil:readById(read(equip and dis==row->equip)["boilerPlantRef"])->dis
      bh:sprk.findAll(v=> v->equipRef->dis==boil).unique("ruleRef")
      if(bh.size>0) bh=bh.map z=> do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else bh=null

      chil:readById(read(equip and dis==row->equip)["chillerPlantRef"])->dis
      ch:sprk.findAll(v=> v->equipRef->dis==chil).unique("ruleRef")
      if(ch.size>0) ch=ch.map z=> do {rul:readById(z->ruleRef)->dis} end.sort("rul").colToList("rul") else ch=null
      {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:row["targetsSpark"],ahuName:null,ahuSpark:null,chillerplant:chil,chillerplantSpark:ch,boilerplant:boil,boilerplantSpark:bh}
      end

    else if(c==null and b==null and d==null) do
      {ts:row->ts,equip:row->equip,rogueTemp:row->rogueTemp,rogueDemand:row->rogueDemand,totalRun:row->totalRun,targetsSpark:row["targetsSpark"],ahuName:null,ahuSpark:null,chillerplant:null,chillerplantSpark:null,boilerplant:null,boilerplantSpark:null}
      end
      end

end

if(abcd.size>0) abcd=abcd.unique("equip").sort("equip").removeCols(["totalRun","ts"])

if(emailrec==true) do
  abcd.ioWriteExcel(`io/EO/eo_Rogue.xls`)
  emailSend(emailId,"EO Rogue data "+read(projMeta)->dis,"Please find the attched EO Rogue data for "+read(projMeta)->dis,[`io/EO/eo_Rogue.xls`])
  end

abcd
end
