  (equip,dates,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Energy",check:true,execute:true) => do

  equipWorkOrder:toEquipWorkOrder(equip)

    maintanancePoint:equip.tomaintenancePoint()

   equipOn:read("return" and fan and run and sensor and equipRef==equip->id)
   equipOnPeriods:equipOn.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))
   equipCmdOff:read("return" and fan and run and cmd and equipRef==equip->id)
   equipCmdOffPeriods:equipCmdOff.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
   spark:hisPeriodIntersection([equipOnPeriods,equipCmdOffPeriods])
   occupancyPoint:read(occupied and equipRef == equip->id,false)

    spark_inter:spark

if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter= hisPeriodIntersection([spark,maintanceHis])
    end

if(occupancyPoint!=null and check==true)
  do
      occupiedperiods:togetOccupancy(equip,dates)
      spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
  end

  spark1:spark_inter.findAll(ts => ts->v0 > minTime)
  spark2:spark_inter.findAll(ts => ts->v0 > maxTime)

ruleCost:costCalc(equip,spark1,"return",pf:1,execute)

    if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type),spark_opexCost:ruleCost})
    else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type),spark_opexCost:ruleCost})

    end
