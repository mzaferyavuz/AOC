(ahu,dates,diff:2,oadMin:25,oatLo:12,oatHi:21,oaEnthMin:45,co2sp:800,minTime:15min,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

     ahuRunTime:ahu.ahuRunPeriods(dates)
     econUsePeriods:toAhuOutsideDamper(ahu).hisRead(dates).hisFindPeriods(val=>(val>oadMin and val!=na()))
     oat:toAhuOutsideTemp(ahu)
     oatUnsuitabilityPeriods:oat.hisRead(dates).hisFindPeriods(val=>(val<oatLo or val>oatHi and val!=na()))
     co2pt:read(co2 and sensor and equipRef==ahu->id,false)
     rat:ahu.toAhuReturnTemp(false)
      equipWorkOrder:toEquipWorkOrder(ahu)

    econused:toAhuOutsideDamper(ahu).hisRead(dates).hisFindPeriods(val=>(val>=oadMin and val!=na()))

   ecomode:read(economizer and enable and cmd and equipRef == ahu->id,false)
    if(ecomode!=null) do
    ecomodeperiod:ecomode.hisRead(dates).hisFindPeriods(v=>(v==true and v!=na()))
    econused=hisPeriodIntersection([econUsePeriods,ecomodeperiod])
    end

     raen:ahu.toAhuReturnEnthalpy(false)
     //compare enthalpies for economy usage suitability if return air enthalpy point is available
     if(raen!=null) do
      oaen:toAhuOutsideEnthalpy(ahu)
      enthUnsuitablePeriods:comparepoints(oaen,raen,dates,diff).hisFindPeriods(matchPointVal(_,"greater"))
      enthLowPeriods:oaen.hisRead(dates).hisFindPeriods(val=>(val<oaEnthMin and val!=na()))

      issueTime:hisPeriodUnion([oatUnsuitabilityPeriods,enthUnsuitablePeriods,enthLowPeriods])

       if(co2pt!=null) do
       co2LowPeriods:co2pt.hisRead(dates).hisFindPeriods(val=>(val<co2sp and val!=na()))
       return hisPeriodIntersection([issueTime,co2LowPeriods,econUsePeriods,ahuRunTime]).findAll(ts=>ts->v0>minTime)
       end
       else return hisPeriodIntersection([issueTime,econused,ahuRunTime]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
     end

    //if enthalpy not available and return air temp is available, compare return air temp with outside air temp to get economy suitability periods

    else if(rat!=null) do
     tempUnsuitabilityPeriods:comparepoints(oat,rat,dates,diff).hisFindPeriods(matchPointVal(_,"greater"))
      issueTime:hisPeriodUnion([oatUnsuitabilityPeriods,tempUnsuitabilityPeriods])

       if(co2pt!=null) do
       co2LowPeriods:co2pt.hisRead(dates).hisFindPeriods(val=>val<co2sp)
       return hisPeriodIntersection([issueTime,co2LowPeriods,econused,ahuRunTime]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
       end
       else return hisPeriodIntersection([issueTime,econused,ahuRunTime]).findAll(ts=>ts->v0>minTime)
    end

    else return hisPeriodIntersection([oatUnsuitabilityPeriods,econused,ahuRunTime]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

  end
