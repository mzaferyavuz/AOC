(point,dates,achDefaultLimit:4,minTime:30min,maxTime:60min,statuscheck:true,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Operations") => do

eqp:readById(point->equipRef)
equipWorkOrder:toEquipWorkOrder(eqp)
site:readById(eqp->siteRef)

limitequip:eqp["achLimit"]
limitsite:site["achLimit"]

if(limitequip!=null) achDefaultLimit=eqp["achLimit"] else if(limitequip==null and limitsite!=null) achDefaultLimit=site["achLimit"]

if(eqp["area"]==null) throw "Area not present for the zone"
if(eqp["roomHeight"]==null) throw "Height not present for the zone"

issue:point.hisRead(dates).hisFindPeriods((v, ts) =>(v!=na() and v<achDefaultLimit))

if(read(discharge and air and flow and sp and equipRef->id==eqp->id,false)!=null) do
  achSPTable:hbAchCalc(eqp,dates,"Setpoint").hisFindPeriods((v, ts) =>(v!=na() and v>achDefaultLimit))
  issue=hisPeriodIntersection([issue,achSPTable])
  end

spark_inter:stsCheck(point,dates,issue,statuscheck)

spark_inter=occupancyandMaintenance(eqp,dates,spark_inter)

spark1:spark_inter.findAll(ts => ts->v0 > minTime)
spark2:spark_inter.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
end
