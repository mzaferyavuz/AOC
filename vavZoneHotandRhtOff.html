src

(vav, dates,tempdeviation:3,dischargeTempThreshold:2,minTime:30min,reheatValveThreshold:5,flowThreshold:10,maxTime:60min,freq:2,znTmpLowLimit:10,znTmpHighLimit:35,znTmpspLowLimit:18,znTmpspHighLimit:28,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

vavFlow: toVavFlow(vav)
vavFlowSP: toVavFlowSp(vav)
ahu:vav.toAHUofVAV()
temp:vav.toAhuZoneTemp()
sp:vav.toAhuZoneSetpoint()
eqp:readById(vav->ahuRef)
equipWorkOrder:toEquipWorkOrder(vav)
reheatPeriods:vav.togetreheatPeriods(dates,reheatValveThreshold,false)

ahuStatus:read(discharge and fan and run and sensor and equipRef == ahu->id,false)
if(ahuStatus==null) throw "AHU fan status point missing"

ps:hisRead([temp,sp,ahuStatus,vavFlow,vavFlowSP],dates).hisInterpolate().map row => do
  if(row["v0"]!=na() and row["v1"]!=na() and row["v2"]!=na() and row["v3"]!=na() and row["v4"]!=na()) {ts: row->ts,v0:((row["v0"] - row["v1"])>tempdeviation)and(row["v2"]==true) and (abs(row["v3"]-row["v4"])<((flowThreshold/100)*row["v4"])) and row["v0"]>=znTmpLowLimit and row["v0"]<=znTmpHighLimit and row["v1"]>=znTmpspLowLimit and row["v1"]<=znTmpspHighLimit}
  else {ts: row->ts,v0:na()}
  end.hisFindPeriods(v=>(v!=na() and v==true))

coolingInefficiencyPeriod:ahudischargecheck(eqp,dates,dischargeTempThreshold)

ps=hisPeriodIntersection([ps,reheatPeriods,coolingInefficiencyPeriod])

ps=occupancyandMaintenance(vav,dates,ps)

spark1:ps.findAll(ts => ts->v0 > minTime)
spark2:ps.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
