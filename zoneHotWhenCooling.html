src

(ahu,dates,coolvalveMinThreshold:2%,coolvalveMaxThreshold:98%,tempThreshold:2,minTime:30min,chilledWatertempLim:8,chilledWatertempThreshold:2,maxTime:60min,freq:2,zoneTempMinVal:10,zoneTempMaxVal:35,zoneTempSPMinVal:18,zoneTempSPMaxVal:28,check:false,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

  equipWorkOrder:toEquipWorkOrder(ahu)

  occupiedperiods:[].toGrid
  spark:[].toGrid

  spark1:[].toGrid
  spark2:[].toGrid

  heatElement:read(heat and cmd and equipRef == ahu->id,false)
  chilledWaterTempSensor:[].toGrid
  occupancyPoint:read(occupied and equipRef == ahu->id,false)
  znTemp:ahu.toAhuZoneTemp(false)
  chillerPlant:ahu["chillerPlantRef"]
  if(znTemp==null) znTemp=ahu.toAhuReturnTemp()
  if(readAll(zone and temp and sp and equipRef==ahu->id).size>1) throw "More than one SP"
  else znTempSp:read(zone and temp and sp and equipRef==ahu->id)
  maintanancePoint:ahu.tomaintenancePoint()


  ahuOn:ahu.ahuRunPeriods(dates)
  coolTime:ahuCoolPeriodslessthan100(ahu,dates,coolvalveMinThreshold,coolvalveMaxThreshold)
  znTempHis:znTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>zoneTempMinVal and v<zoneTempMaxVal))
  znTempSpHis:znTempSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>zoneTempSPMinVal and v<zoneTempSPMaxVal))
  issueTime:comparepoints(znTemp,znTempSp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))

  if(chillerPlant==null)
   spark=hisPeriodIntersection([issueTime,coolTime,znTempHis,znTempSpHis,ahuOn])

  if(chillerPlant!=null ) do

    chilleronperiod: chillerPlant.anyChilleron(dates)
   chilledWaterTemp:read(chilled and leaving and water and temp and sensor and equipRef==chillerPlant->id,false)
   chilledWaterTempsp:read(chilled and leaving and water and temp and sp and equipRef==chillerPlant->id,false)

  if(chilledWaterTemp!=null and chilledWaterTempsp==null) chilledWaterTempSensor=chilledWaterTemp.hisRead(dates).hisFindPeriods(v=>v!=na() and (v<chilledWatertempLim))
   else if (chilledWaterTemp!=null and chilledWaterTempsp!=null) do
   chilledWaterTempSensor=hisRead([chilledWaterTemp,chilledWaterTempsp],dates).hisInterpolate().map row => do
          if(row["v0"]!=na() and row["v1"]!=na()) {ts: row->ts,
                                            v0:
                                              if (abs(row["v0"] - row["v1"])<chilledWatertempThreshold) "okay"
                                              else "nok"
                                              }
              else {ts: row->ts,v0:na()}
        end.hisFindPeriods(matchPointVal(_,"okay"))
         end

  if(chilleronperiod!=null and chilledWaterTemp!=null) spark= hisPeriodIntersection([issueTime,coolTime,ahuOn,chilleronperiod,chilledWaterTempSensor,znTempHis,znTempSpHis])

  else if(chilleronperiod!=null and chilledWaterTemp==null) spark= hisPeriodIntersection([issueTime,coolTime,ahuOn,chilleronperiod,znTempHis,znTempSpHis])

  else if(chilleronperiod==null and chilledWaterTemp!=null) spark= hisPeriodIntersection([issueTime,coolTime,ahuOn,chilledWaterTempSensor,znTempHis,znTempSpHis])

  end

spark_inter:spark

  if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter= hisPeriodIntersection([spark,maintanceHis])
    end

  if(heatElement!=null and spark_inter.size>0)
    do
      noHtgPeriods:ahu.ahuNoHeatPeriods_inter(dates,1)
      spark_inter= hisPeriodIntersection([spark_inter,noHtgPeriods])
    end

  if(occupancyPoint!=null and check== true and spark_inter.size>0)
    do
    occupiedperiods=togetOccupancy(ahu,dates)
    spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
    end

  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

  if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
