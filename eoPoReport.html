(dates,emailIdTicket,emailId,factor:0.85,aVal:95,hardlimitdeviation:0.75,hLimitTripTime:120min) => do

table:eoRogue(dates,factor,aVal,false,"").map row=>do

equip:read(equip and dis==row->equip)
status:equip["eoStatus"]
eocr:equip["eoCritical"]
strategy:equip["eoStrategy"]
disabled:equip["eoDisabled"]

////rogue condition
if(row["rogueTemp"]=="Rogue" and row["rogueDemand"]=="Rogue") rogueCondition:"Both"
else if(row["rogueTemp"]!="Rogue" and row["rogueDemand"]=="Rogue") rogueCondition:"Demand"
else if(row["rogueTemp"]=="Rogue" and row["rogueDemand"]!="Rogue") rogueCondition:"Temp"
else if(row["rogueTemp"]=="NA" and row["rogueDemand"]=="NA") rogueCondition:"None"
else rogueCondition:"None"

////Remark
if(rogueCondition=="Both") do
  iterated:equip.eo_SP_iteration(dates,aVal,factor,hardlimitdeviation).table.toRec
  if(iterated!=null) rmk:"EO Setpoint can be updated to "+(((iterated["setpoint"]*100).round)/100) else rmk:null
  end

else rmk:null

////Action

if(disabled==null) do

  if(rogueCondition=="Both") do

    if (eocr!=null and status==null and rmk==null) action:"Don't remove Critical Asset"
    else if(eocr!=null and status=="Field Action Pending" and rmk==null) action:"Critical Asset with Pending field action"
    else if(eocr==null and status=="Field Action Pending" and rmk==null) action:"Pending field action"
    else if(eocr==null and status==null and rmk==null) action:"To be removed rogue zone"
    else if(eocr==null and status==null and rmk!=null) action:"Rogue zone "+rmk
    else action:"None"
    end

  else if(rogueCondition=="Demand" and strategy=="Temp") action:"None"
  else if(rogueCondition=="Temp" and strategy=="Demand") action:"None"
  else if(rogueCondition=="Demand" and strategy=="Demand" and row["rogueTemp"]!="NA") action:"Change the EO Strategy to Temp based"
  else if(rogueCondition=="Temp" and strategy=="Temp" and row["rogueDemand"]!="NA") action:"Change the EO Strategy to Demand based"
  else if(rogueCondition=="Demand" and strategy=="Demand" and row["rogueTemp"]=="NA") action:"To be removed rogue zone"
  else if(rogueCondition=="Temp" and strategy=="Temp" and row["rogueDemand"]=="NA") action:"To be removed rogue zone"
  else action:"None"
  end

  else if(disabled!=null) do

    if(rogueCondition=="Demand" or rogueCondition=="Temp") action:"Enable the EO for the zone"
    else action:"None"
    end

else action:"None"

////Rules
ahu:row["ahuName"]
ahuSprk:row["ahuSpark"]
boilerPlant:row["boilerplant"]
chillerPlant:row["chillerplant"]
equipmentSprk:row["targetsSpark"]
boilerPlantSprk:row["boilerplantSpark"]
chillerPlantSprk:row["chillerplantSpark"]

{equipment:row->equip,rogue:rogueCondition,currentStrategy:strategy,action:action,remark:rmk,ruleonequip:equipmentSprk,ahuConnected:ahu,ahurule:ahuSprk,chPlant:chillerPlant,chPlantRule:chillerPlantSprk,boilerPlant:boilerPlant,boilerPlantRule:boilerPlantSprk}
end////end of map

////Count of things
crticalcount:readAll(equip and eoCritical).size
eonotruning:readAll(equip and eoSp and eoDisabled).size
allcount:readAll(equip and eo and not eoDisabled).size
withissue:table.findAll(v=> v->action.contains("removed")).size
pendingsiteactioncount:table.findAll(v=> v->action.contains("field")).size
crticalroguecount:table.findAll(v=> v->action.contains("Critical") and v->rogue!="None").size
aoctaskpending:table.findAll(v=> v->action.contains("Change") or v->action.contains("update") or v->action.contains("Enable")).size
noissuecount:(allcount-withissue-aoctaskpending-crticalroguecount)

////Action report mail segment

if(table.findAll(v=> v->action!="None").size>0) do

  project:read(projMeta)->dis
  subject:"AOC "+project+" EO Daily Action / Report "+now().date
  table.findAll(v=> v->action!="None").ioWriteExcel(`io/eo/eo_report.xls`)

  body:
   "<html><body>\n"+
     "<b1>EO Summary</b1>\n"+
     "<p> Total number of zones under EO "+allcount+" </p>"+
     "<p> Number of Non Rogue Zones "+noissuecount+" </p>"+
     "<p> Number of critical zones "+crticalcount+" </p>"+
     "<p> Number of critical zones which are rogue "+crticalroguecount+" </p>"+
     "<p> Number of zones which are rogue "+withissue+" </p>"+
     "<p> Number of zones where site action is pending "+pendingsiteactioncount+" </p>"+
     "<p> Number of zones where AOC action is pending "+aoctaskpending+" </p>"+
     "<p> Number of zones which are Skipped from EO  "+eonotruning+" </p>"+
     "<p><small><b> Please do not reply to this email. This is an unmonitored mailbox </b></small></p>"+
   "</body></html>\n"

     "</body></html>\n"

  emailSend(emailIdTicket,subject,body,`io/eo/eo_report.xls`)
  end

else do

  project:read(projMeta)->dis
  subject:project+" EO Daily Report "+now().date

  body:
   "<html><body>\n"+
     "<b1>EO Summary</b1>\n"+
     "<p> Total number of zones under EO "+allcount+" </p>"+
     "<p> Number of Non Rogue Zones "+noissuecount+" </p>"+
     "<p> Number of critical zones "+crticalcount+" </p>"+
     "<p> Number of critical zones which are rogue "+crticalroguecount+" </p>"+
     "<p> Number of zones which are rogue "+withissue+" </p>"+
     "<p> Number of zones where site action is pending "+pendingsiteactioncount+" </p>"+
     "<p> Number of zones which are Skipped from EO  "+eonotruning+" </p>"+
     "<p><small><b> Please do not reply to this email. This is an unmonitored mailbox </b></small></p>"+
   "</body></html>\n"

     "</body></html>\n"

  emailSend(emailId,subject,body)
  end

table
end
