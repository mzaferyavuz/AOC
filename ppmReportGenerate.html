(dates:pastWeek,email) => do

  points:readAll(point and kind=="Number")
  denom:points.size
  projName:read(projMeta)->dis
  sDate:null
  eDate:null

  data:points.map ((row,i)=>do
    jobProgress((i/denom)*100,row->dis)
  his:row.hisRead(dates,{limit:null}).findAll(ts=>(ts["v0"]!=na() and ts["v0"]!=null))
   tCount:his.size

    if(isFunc(dates)==true) do
      sDate=dates().start
      eDate=dates().end
    end
    else  do
      sDate=dates.start
      eDate=dates.end
    end

    hCount:0
    lCount:0
    rCount:0

    pntEqp:row->equipRef->dis

  mean:his.foldCol("v0",mean)
  stddev:his.foldCol("v0",standardDeviation)

  max:his.hisRollup(max,1day).foldCol("v0",avg)
  min:his.hisRollup(min,1day).foldCol("v0",avg)

  if(row["minVal"]!=null) expMin:row->minVal
  else expMin:(mean-(3*stddev)).as(row["unit"])
  if(row["maxVal"]!=null) expMax:row->maxVal
  else expMax:(mean+(3*stddev)).as(row["unit"])

  outliers:his.each row=>do
  if(row["v0"]>expMax) hCount=hCount+1
  if(row["v0"]<expMin) lCount=lCount+1
  else rCount=rCount+1
  end

  beyondLim:((hCount+lCount)/tCount*100).as(1%)

  {equipment:pntEqp,sensorName:row->dis,startDate:sDate,endDate:eDate,meanValue:mean.format("0.00"),
  standardDeviation:stddev.format("0.00"),expectedMinimum:expMin.format("0.00"),avgDailyMinimum:min.format("0.00"),
  expectedMaximum:expMax.format("0.00"),avgDailyMaximum:max.format("0.00"),lessThanExpMinVals:lCount.format("0.00"),
  higherThanExpMaxVals:hCount.format("0.00"),percentageBeyondLimit:beyondLim.as(1%).format("0.00"),
  comment:if(beyondLim>20 and beyondLim<40) "*" else if(beyondLim>=40 and beyondLim<80) "**" else if(beyondLim>80) "***"  else "OK"}
  end).addColMeta("meanValue",{name:"Average Point Value"}).sort("equipment")

  file:parseUri("io/ppmReport_"+projName+"_"+sDate+"to"+eDate+".csv")
  data.ioWriteCsv(file)
  emailSend(email,"PPM task report for "+projName,"PFA ppm Report",file)
end
