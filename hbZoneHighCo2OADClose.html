(ahu,dates,oadMin:25,spdeviation:10,defsetpoint:700ppm,minTime:30min,maxTime:60min,freq:2,co2MinVal:200,co2MaxVal:1400,co2spMaxVal:700,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

site:readById(ahu->siteRef)
ahuOn:ahu.ahuRunPeriods(dates)
oad:ahu.toAhuOutsideDamper(false)
equipWorkOrder:toEquipWorkOrder(ahu)

limitequip:ahu["co2SPLimit"]
limitsite:site["co2SPLimit"]

if(limitequip!=null) defsetpoint=ahu["co2SPLimit"] else if(limitequip==null and limitsite!=null) defsetpoint=site["co2SPLimit"]

if(oad!=null) do
  oadMinSp:read(outside and damper and position and sp,false)
  if(oadMinSp!=null) oadClose:comparepoints(oadMinSp,oad,dates,0).hisFindPeriods(matchPointVal(_,"greater")) else oadClose:oad.hisRead(dates).hisFindPeriods(v=>(v!=na() and v<=oadMin))
  end

else throw "No OAD Damper"

co2:read(co2 and sensor and zone and equipRef==ahu->id,false)
co2sp:read(co2 and sp and zone and not min and equipRef==ahu->id,false)
if(co2==null) throw "No Zone CO2 sensor"

if(co2["minVal"]!=null) co2MinVal=(co2->minVal)
if(co2["maxVal"]!=null) co2MaxVal=(co2->maxVal)

co2His:co2.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>co2MinVal and v<co2MaxVal))

if(co2sp==null) do
  issue:hisRead(co2,dates).hisFindPeriods(v=>(v!=na() and v>((defsetpoint)+((spdeviation/100)*(defsetpoint)))))
  spark:hisPeriodIntersection([ahuOn,issue,oadClose,co2His])
  end

else if(co2sp!=null) do

if(co2sp["maxVal"]!=null) co2spMaxVal=(co2sp->maxVal)

  issue:hisRead([co2,co2sp],dates).map row=>do
    if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:(row["v0"] - row["v1"])>(row["v1"]*(spdeviation/100))} else {ts:row->ts,v0:na()}
    end.hisFindPeriods(v=>(v!=na() and v==true))

  co2spHis:co2sp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v<co2spMaxVal))

  spark:hisPeriodIntersection([ahuOn,issue,co2His,co2spHis,oadClose])
  end

spark=occupancyandMaintenance(ahu,dates,spark,false)

spark1:spark.findAll(ts => ts->v0 > minTime)
spark2:spark.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
