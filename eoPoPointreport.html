(dates) => do

  zones:readAll(equip and eoPlant).map row=>do

  tempsp:read(temp and sp and eo and applied and equipRef==row->id,false)
  pressuresp:read(pressure and sp and eo and applied and equipRef==row->id,false)
  switchT:read(temp and switch and eo and sensor and equipRef==row->id,false)
  switchP:read(pressure and switch and eo and sensor and equipRef==row->id,false)

  if(row["ahu"]!=null or row["fcu"]!=null) sensor:read(discharge and fan and run and sensor and equipRef==row->id)
  else if (row["vav"]!=null) sensor:read(discharge and fan and run and sensor and equipRef==row->ahuRef->id)
  else sensor:read(run and sensor and equipRef==row->id,false)

  if(sensor!=null) do
  rs:sensor.hisRead(dates).hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1mo)
  if(rs.size>0) rs=rs.toRec["v0"] else rs=null
  end
  else rs:null

  if(switchT!=null) switch:switchT.hisRead(dates).hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1mo).toRec["v0"] else switch:null
  if(switchP!=null) switch1:switchP.hisRead(dates).hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1mo).toRec["v0"] else switch1:null
  sch:read(eo and schedule).hisRead(dates).hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1mo).toRec["v0"]

  ////////////////////////////////////////////Data Loss

  dL:null
  tempspname:null
  pressurespname:null

  if(tempsp!=null) do
  tempspname=tempsp["dis"]
  if(tempsp["hisEnd"]!=null) do lag:(now()-tempsp->hisEnd).to(1hr) end else lag:25hr
  if(lag>24hr and dL==null) dL=tempsp->dis
  end

  if(pressuresp!=null) do
  pressurespname=pressuresp["dis"]
  if(pressuresp["hisEnd"]!=null) lag1:(now()-pressuresp->hisEnd).to(1hr) else lag1:25hr
  string1:pressuresp->dis
  if(lag1>24hr and dL==null) dL=string1 else if (lag1>24hr and dL!=null) dL= dL +" * " + string1
  end

  if(sensor!=null) do
  if(sensor["hisEnd"]!=null) lag2:(now()-sensor->hisEnd).to(1hr) else lag2:25hr
  string2:sensor->dis
  if(lag2>24hr and dL==null) dL=string2 else if (lag2>24hr and dL!=null) dL= dL +" * " + string2
  end

  ////////////////////////////////////////////spark

  aPlant:row["ahuRef"]
  hPlant:row["boilerPlantRef"]
  cPlant:row["chillerPlantRef"]

  sprk:navFilter("sparkTarget", null).sparks(dates).findAll(v=> v->equipRef->dis==row->dis)
  if(cPlant!=null) sprk1:navFilter("sparkTarget", null).sparks(dates).findAll(v=> v->equipRef->dis==cPlant->dis) else sprk1:[].toGrid
  if(hPlant!=null) sprk2:navFilter("sparkTarget", null).sparks(dates).findAll(v=> v->equipRef->dis==hPlant->dis) else sprk2:[].toGrid
  if(aPlant!=null) sprk3:navFilter("sparkTarget", null).sparks(dates).findAll(v=> v->equipRef->dis==aPlant->dis) else sprk3:[].toGrid

  if(sprk.size>0) rl:sprk.unique("ruleRef").map a=> do {rul:readById(a->ruleRef)->dis} end.colToList("rul") else rl:null
  if(sprk1.size>0) cpr:sprk1.unique("ruleRef").map a=> do {rul:readById(a->ruleRef)->dis} end.colToList("rul") else cpr:null
  if(sprk2.size>0) bpr:sprk2.unique("ruleRef").map a=> do {rul:readById(a->ruleRef)->dis} end.colToList("rul") else bpr:null
  if(sprk3.size>0) apr:sprk3.unique("ruleRef").map a=> do {rul:readById(a->ruleRef)->dis} end.colToList("rul") else apr:null

  {equipment:row->dis,eo_schedule:sch,equip_RunTime:rs,eo_Temp_RunTime:switch,tempSP:tempspname,pressureSP:pressurespname,eo_Pressure_RunTime:switch1,equip_Rule:rl,cillerPlant_Rule:cpr,boilerPlant_Rule:bpr,ahu_Rule:apr,points_data_Loss:dL}
  end

  if(zones.size>0) zones=zones.sort("equipment")
  end
