src

(dates,tempThreshold:3,dmpThreshold:90,issueLimit:50)=> do

equip:readAll(vav and not skipEo)

output:[].toGrid()
spark1:[].toGrid()
spark2:[].toGrid()
spark3:[].toGrid()
spark4:[].toGrid()
znDeltaLow:[].toGrid()
znDeltaHigh:[].toGrid()
znDelta1:[].toGrid()
zoneDemand:[].toGrid()
heatdemand1:[].toGrid()
zoneSensorAvg:[].toGrid()
zoneSpAvg:[].toGrid()
action1:null
equip.map row=>do
ahu1:row["ahuRef"]
ahu2:readById(ahu1)
znSensor:read(zone and temp and sensor and equipRef==row->id,false)
znSp:read(zone and temp and sp and equipRef==row->id,false)
znDmd:read(damper and cmd and equipRef==row->id,false)
heatDemand:read(reheat and valve and cmd and equipRef==row->id,false)

occupancy:togetOccupancy(row,dates)
if (occupancy==null and ahu1!=null) do
occupancy = read(discharge and fan and run and sensor and equipRef==ahu2->id).hisRead(dates).hisFindPeriods(v=>v==true) end

occupancy1:occupancy.foldCol("v0",sum)
if (znSensor!=null and znSp!=null) do
join:hisRead([znSensor,znSp],dates)
occupancy2:read(discharge and fan and run and sensor and equipRef==ahu2->id).hisRead(dates)
if (occupancy2.size!=0 or occupancy2!=null) do
join1:hisJoin([join,occupancy2]).findAll(ts=>(ts["v2"]==true) or (ts["v2"]=="Occupied"))
zoneSensorAvg=join1.foldCol("v0",avg)
zoneSpAvg=join1.foldCol("v1",avg) end
znDeltaLow=hisRead([znSensor],dates).hisFindPeriods(v=>v!=na() and (v<row["eominVal"]))//.hisRollup(sum,1day).toRec["v0"]
znDeltaHigh=hisRead([znSensor],dates).hisFindPeriods(v=>v!=na() and (v>row["eomaxVal"]))//.hisRollup(sum,1day).toRec["v0"]
znDelta1=hisRead([znSensor,znSp],dates).hisFoldCols(spread).hisFindPeriods(v=>v!=na() and v>tempThreshold) end

if (znDeltaLow!=null) do
spark1=hisPeriodIntersection([znDeltaLow,occupancy]).foldCol("v0",sum)
spark4=hisPeriodIntersection([znDeltaHigh,occupancy]).foldCol("v0",sum)
spark3=hisPeriodIntersection([znDelta1,occupancy]).foldCol("v0",sum)
end

if (heatDemand!=null)
 heatdemand1:heatDemand.hisRead(dates).hisFindPeriods(v=>v>dmpThreshold).foldCol("v0",sum)
if (znDmd!=null)
zoneDemand=znDmd.hisRead(dates).hisFindPeriods(v=>v!=na() and v>dmpThreshold)

if(zoneDemand!=null)
spark2=hisPeriodIntersection([zoneDemand,occupancy]).foldCol("v0",sum)
heatdemand2:((heatdemand1/occupancy1)*100).as("%")
zoneDeltaEOlow:((spark1/occupancy1)*100).as("%")
if(zoneDeltaEOlow==null) zoneDeltaEOlow=0%
zoneDeltaEOHigh:((spark4/occupancy1)*100).as("%")
if(zoneDeltaEOHigh==null) zoneDeltaEOHigh=0%
zoneDeltaEOlimitPct1:zoneDeltaEOlow+zoneDeltaEOHigh
zoneDeltaPct1:((spark3/occupancy1)*100).as("%")
znDamperPct1:((spark2/occupancy1)*100).as("%")

if(znDamperPct1==null)
znDamperPct1=0%
if(heatdemand2==null)
heatdemand2=0%
if (zoneDeltaEOlimitPct1<issueLimit and (znDamperPct1>issueLimit or heatdemand2>issueLimit))// and zoneDeltaPct1!=null and  znDamperPct1!=null)
action1="switch to temp strategy"

else if (zoneDeltaEOlimitPct1>issueLimit and (znDamperPct1<issueLimit or heatdemand2<issueLimit))// and zoneDeltaPct1!=null and  znDamperPct1!=null)
action1="switch to demand strategy"

else if (zoneDeltaEOlimitPct1>issueLimit and (znDamperPct1>issueLimit or heatdemand2>issueLimit))// and zoneDeltaPct1!=null and  znDamperPct1!=null)
action1="Rogue"

else
action1="No issue"


{ahuref:row["ahuRef"], equipName:row->dis, znTmpEOLowlimit:spark1, znTmpEOHighlimit:spark4, zoneDeltaEOlimitPct:zoneDeltaEOlimitPct1, zoneDelta:spark3, zoneDeltaPct:zoneDeltaPct1, znDamper:spark2, heatingDemand:heatdemand2,zoneSenAvg:zoneSensorAvg,zoneSPAvg:zoneSpAvg, znDamperPct:znDamperPct1,action:action1}
end


end
