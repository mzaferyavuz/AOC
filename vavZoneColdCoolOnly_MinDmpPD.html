src

(vav, dates,tempdeviation:3,minTime:30min,dischargeTempThreshold:2,damperThreshold:20, maxTime:60min,freq:2,znTmpLowLimit:10,znTmpHighLimit:35,znTmpspLowLimit:18,znTmpspHighLimit:28,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

ahu:vav.toAHUofVAV()
temp:vav.toAhuZoneTemp()
sp:vav.toAhuZoneSetpoint()
eqp:readById(vav->ahuRef)
vavdamp:toVavDamper(vav)
equipWorkOrder:toEquipWorkOrder(vav)

ahuStatus:read(discharge and fan and run and sensor and equipRef == ahu->id,false)
if(ahuStatus==null) throw "AHU fan status point missing"

ps:hisRead([temp,sp,ahuStatus,vavdamp],dates).hisInterpolate().map row => do
if(row["v0"]!=na() and row["v1"]!=na() and row["v2"]!=na()) {ts: row->ts,v0:((row["v1"] - row["v0"])>tempdeviation)and(row["v2"]==true) and row["v0"]>=znTmpLowLimit and row["v0"]<=znTmpHighLimit and row["v1"]>=znTmpspLowLimit and row["v1"]<=znTmpspHighLimit and row["v3"]<damperThreshold}
  else {ts: row->ts,v0:na()}
end.hisFindPeriods(v=>(v!=na() and v==true))

ahuovercoolingPeriod:eqp.ahudischargecheck(dates,dischargeTempThreshold)

ps=hisPeriodIntersection([ps,ahuovercoolingPeriod])

ps=occupancyandMaintenance(vav,dates,ps)

spark1:ps.findAll(ts => ts->v0 > minTime)
spark2:ps.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
