  (equip,dates,threshold:10,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Operations") => do

  spark1:[].toGrid
  spark2:[].toGrid
  spark:[].toGrid
  issue:[].toGrid
    equipWorkOrder:toEquipWorkOrder(equip)
     maintanancePoint:equip.tomaintenancePoint()

  out:read(exhaust and fan and speed and cmd and equipRef==equip->id,false)
  fb:read(exhaust and fan and speed and sensor and equipRef==equip->id,false)
if(out==null and fb==null) throw "No VSD Cmd or Feedback"
  ahuStatus:read(exhaust and fan and run and sensor and kind=="Bool" and equipRef==equip->id)
  ahuCmd:read(exhaust and fan and run and cmd and kind=="Bool" and equipRef==equip->id,false)

equipOn:ahuStatus.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))
equipOff:ahuStatus.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
  if(out!=null and fb!=null)
do
    issue=hisRead([out,fb],dates).hisInterpolate().map row=>do

    if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:row["v0"],v1:row["v1"]}
    else {ts:row->ts,v0:na(),v1:na()}
    end
  vsdmore:issue.hisFoldCols(max).hisFindPeriods(v=>(v!=na() and v>threshold))
  vsdless:issue.hisFoldCols(max).hisFindPeriods(v=>(v!=na() and v<threshold))
end
else if(out!=null and fb==null)
do
  issue=out.hisRead(dates).hisInterpolate().map row=>do

    if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:row["v0"]}
    else {ts:row->ts,v0:na()}
    end
  vsdmore:issue.hisFindPeriods(v=>(v!=na() and v>threshold))
  vsdless:issue.hisFindPeriods(v=>(v!=na() and v<threshold))
end
else if(out==null and fb!=null)
do
  issue=fb.hisRead(dates).hisInterpolate().map row=>do
    if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:row["v0"]}
    else {ts:row->ts,v0:na()}
    end
  vsdmore:issue.hisFindPeriods(v=>(v!=na() and v>threshold))
  vsdless:issue.hisFindPeriods(v=>(v!=na() and v<threshold))
end
if(ahuCmd==null)
do
issue1:hisPeriodIntersection([equipOn,vsdless])
issue2:hisPeriodIntersection([equipOff,vsdmore])
spark=hisPeriodUnion([issue1,issue2])
end
else if(ahuCmd!=null)
do
cmdOn:ahuCmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true))
cmdOff:ahuCmd.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
issue1:hisPeriodIntersection([equipOn,vsdless,cmdOn])
issue2:hisPeriodIntersection([equipOff,vsdmore,cmdOff])
spark=hisPeriodUnion([issue1,issue2])
end
spark_inter:spark
   if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter= hisPeriodIntersection([spark,maintanceHis])
    end
  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)
 if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  end
