src

//Return/zone air humidity is higher than setpoint with Dehumidification , run only on airhandlers with dehum tag

(ahu,dates,ztype:"zone",error:5,threshold:5,ahuMinSpeed:20,dehumid:20,coolValve:80,minTime:15min,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

humidity:read((parseFilter(ztype)) and humidity and sensor and equipRef == ahu->id,false)
setpoint:read((parseFilter(ztype)) and humidity and sp and equipRef == ahu->id,false)

  if((humidity==null) and (setpoint==null))  do

humidity=read("return" and humidity and sensor and equipRef == ahu->id)
setpoint=read("return" and humidity and sp and equipRef == ahu->id)


end

issueTime:comparepoints(humidity,setpoint,dates,threshold).hisFindPeriods(matchPointVal(_,"greater"))

onTime:ahu.ahuRunPeriods(dates,ahuMinSpeed)
equipWorkOrder:toEquipWorkOrder(ahu)
coolPeriods:ahu.ahuCoolPeriods(dates,coolValve)

dehumidifier:read(dehumidifier and (sensor or cmd) and run and equipRef == ahu->id,false)
if(dehumidifier!=null)
do

  dehumidifieronperiods:ahu.toahudehumidfier(dates,dehumid,false)

//Compute fault periods
  return hisPeriodIntersection([issueTime,onTime,dehumidifieronperiods]).findAll(ts=>ts->v0>minTime)
 .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
end

else
 return hisPeriodIntersection([issueTime,onTime,coolPeriods]).findAll(ts=>ts->v0>minTime)
 .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
