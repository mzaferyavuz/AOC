(ahu,dates,diff:2,co2deviation:5,oadMin:25,oatLo:12,oatHi:21,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

  ahuRunTime:ahu.ahuRunPeriods(dates)

  spark:[].toGrid()
  spark1:[].toGrid()
  spark2:[].toGrid()
  spark_inter:[].toGrid()

  maintanancePoint:ahu.tomaintenancePoint()

  equipWorkOrder:toEquipWorkOrder(ahu)

  econUsePeriods:[].toGrid()

  econUsePeriods=toAhuOutsideDamper(ahu).hisRead(dates).hisClip().hisFindPeriods(v=>(v!=na() and v>=oadMin))


  oat:read(outside and temp and (sensor or in) and equipRef == ahu->id, false)

   if(oat==null)
       do
       w: toWeather(ahu, true)
       if (w == null) return null
       oat=read(temp and weatherStationRef == w->id, true)
       end

   oatUnsuitabilityPeriods:oat.hisRead(dates).hisClip().hisFindPeriods(v=>(v!=na() and (v<oatLo or v>oatHi)))

  spark=hisPeriodIntersection([econUsePeriods,ahuRunTime,oatUnsuitabilityPeriods])


   raen:ahu.toAhuReturnEnthalpy(false)
   oaen:ahu.toAhuOutsideEnthalpy()

  if(raen!=null and oaen!=null)

  do

        dev:hisRead([raen,oaen],dates).map row=>do

              if(row["v0"]!=na() and row["v1"]!=na())

                   {ts:row->ts,v0:(row["v0"] - row["v1"])<=(diff)}

              else {ts:row->ts,v0:na()}

                   end


        enthUnsuitablePeriods:dev.hisFindPeriods(v=>(v!=na() and v==true))


  spark=hisPeriodIntersection([econUsePeriods,ahuRunTime,oatUnsuitabilityPeriods,enthUnsuitablePeriods])

  end

  spark_inter=spark

  co2pt:read(co2 and sensor and equipRef==ahu->id,false)
  co2sp:read(co2 and sp and equipRef==ahu->id,false)

   if(co2pt!=null and co2sp!=null)

   do
         notDCVPeriods:hisRead([co2pt,co2sp],dates).hisClip().hisInterpolate().map row => do
            if(row["v0"]!=na() and row["v1"]!=na()) {ts: row->ts,
                                              v0:
                                                if ((row["v1"] - row["v0"])>(row["v1"]*(co2deviation/100)))"okay"
                                                else "nok"
                                                }
                else {ts: row->ts,v0:na()}
          end.hisFindPeriods(matchPointVal(_,"okay"))

  spark_inter= hisPeriodIntersection([spark,notDCVPeriods])

  end

   if(maintanancePoint!=null)
        do
          maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
          spark_inter= hisPeriodIntersection([spark_inter,maintanceHis])
        end


      spark1=spark_inter.findAll(ts => ts->v0 > minTime)
      spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

      if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
    else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

      end
