() => do

//hotWaterHeat
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(heat and valve and cmd and kind=="Number" and equipRef==row->id,false)!=null) commit(diff(row,{hotWaterHeat}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and hotWaterHeat).map row=> if(read(heat and valve and cmd and kind=="Number" and equipRef==row->id,false)==null) commit(diff(row,{-hotWaterHeat}))

//deHum
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(humidity and sp and kind=="Number" and equipRef==row->id,false)!=null) commit(diff(row,{deHum}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and deHum).map row=> if(read(humidity and sp and kind=="Number" and equipRef==row->id,false)==null) commit(diff(row,{-deHum}))

//vavZone
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(vav and equip and ahuRef==row->id,false)!=null) commit(diff(row,{vavZone}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and vavZone).map row=> if(read(vav and equip and ahuRef==row->id,false)==null) commit(diff(row,{-vavZone}))

//directZone
  readAll(ahu and not vavZone).each row=>do
  zoneSP:readAll(zone and temp and sp and equipRef==row->id)
  zoneTemps:readAll(zone and temp and sensor and equipRef==row->id)
  returnTemps:readAll("return" and temp and sensor and equipRef==row->id)
  returnSP:readAll("return" and temp and sp and equipRef==row->id)
  if((zoneTemps.size==1 or returnTemps.size==1) and (zoneSP.size==1 or returnSP.size==1)) commit(diff(row,{directZone,comfortAsset}))
  else if((zoneTemps.size==1 or returnTemps.size==1) and (zoneSP.size==2 or returnSP.size==2)) commit(diff(row,{directZone,comfortAsset}))
  end

  readAll(ahu and directZone).each row=>do
  zoneSP:readAll(zone and temp and sp and equipRef==row->id)
  returnSP:readAll("return" and temp and sp and equipRef==row->id)
  if(zoneSP.size==0 and returnSP.size==0) commit(diff(row,{-directZone,-comfortAsset}))
  end

  //multiZone
  readAll(ahu).map row=>do
  zoneSPs:readAll(zone and temp and sp and equipRef==row->id)
  zoneTemPs:readAll(zone and temp and sensor and equipRef==row->id)
  returnTemPs:readAll("return" and temp and sensor and equipRef==row->id)
  if((zoneTemPs.size>1 or returnTemPs.size>1) and zoneSPs.size==1)  commit(diff(row,{multiZone}))
  else if((zoneTemPs.size>1 or returnTemPs.size>1) and zoneSPs.size==2)  commit(diff(row,{multiZone}))
  end

  readAll(ahu and multiZone).map row=>do
  zoneTemPs:readAll(zone and temp and sensor and equipRef==row->id)
  returnTemPs:readAll("return" and temp and sensor and equipRef==row->id)
  if(zoneTemPs.size<2 and returnTemPs.size<2)  commit(diff(row,{-multiZone}))
  end

//variableVolume
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(discharge and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)!=null) and read(discharge and pressure and sensor and equipRef ==row->id,false)!=null) commit(diff(row,{variableVolume}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and variableVolume).map row=> if((read(discharge and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)==null) and read(discharge and pressure and sensor and equipRef ==row->id,false)==null) commit(diff(row,{-variableVolume}))

//saVariableSpeed
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(discharge and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)!=null)) commit(diff(row,{saVariableSpeed}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and variableSpeed).map row=> if((read(discharge and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)==null)) commit(diff(row,{-saVariableSpeed}))

//raVariableSpeed
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read("return" and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)!=null)) commit(diff(row,{raVariableSpeed}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and raVariableSpeed).map row=> if((read("return" and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)==null)) commit(diff(row,{-raVariableSpeed}))

//eaVariableSpeed
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(exhaust and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)!=null)) commit(diff(row,{eaVariableSpeed}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and eaVariableSpeed).map row=> if((read(exhaust and fan and cmd and speed and kind=="Number" and equipRef==row->id,false)==null)) commit(diff(row,{-eaVariableSpeed}))

//sapControl
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(discharge and pressure and sensor and equipRef ==row->id,false)!=null and read(discharge and pressure and sp and equipRef ==row->id,false)!=null) commit(diff(row,{sapControl}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and sapControl).map row=> if(read(discharge and pressure and sensor and equipRef ==row->id,false)==null or read(discharge and pressure and sp and equipRef ==row->id,false)==null) commit(diff(row,{-sapControl}))

//rapControl
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read("return"  and pressure and sensor and equipRef ==row->id,false)!=null and read("return"  and pressure and sp and equipRef ==row->id,false)!=null) commit(diff(row,{rapControl}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and rapControl).map row=> if(read("return"  and pressure and sensor and equipRef ==row->id,false)==null or read("return"  and pressure and sp and equipRef ==row->id,false)==null) commit(diff(row,{-rapControl}))

//eapControl
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(exhaust and pressure and sensor and equipRef ==row->id,false)!=null and read(exhaust and pressure and sp and equipRef ==row->id,false)!=null) commit(diff(row,{eapControl}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and eapControl).map row=> if(read(exhaust and pressure and sensor and equipRef ==row->id,false)==null) or read(exhaust and pressure and sp and equipRef ==row->id,false)==null) commit(diff(row,{-eapControl}))

//znpControl
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(zone and pressure and sensor and equipRef ==row->id,false)!=null) commit(diff(row,{znpControl}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and znpControl).map row=> if(read(zone and pressure and sensor and equipRef ==row->id,false)==null) commit(diff(row,{-znpControl}))

//coolOnly
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(heat and cmd and equipRef==row->id,false)==null and read(cool and cmd and equipRef==row->id,false)!=null) commit(diff(row,{coolOnly}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and coolOnly).map row=> if(read(heat and cmd and equipRef==row->id,false)!=null) commit(diff(row,{-coolOnly}))

//heatOnly
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(cool and cmd and equipRef==row->id,false)==null and read(heat and cmd and equipRef==row->id,false)!=null) commit(diff(row,{heatOnly}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and heatOnly).map row=> if(read(cool and cmd and equipRef==row->id,false)!=null) commit(diff(row,{-heatOnly}))

//heatRecovery
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read((heatWheel or energyWheel) and (cmd or sensor) and equipRef==row->id,false)!=null ) commit(diff(row,{heatRecovery}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and heatRecovery).map row=> if(read((heatWheel or energyWheel) and (cmd or sensor) and equipRef==row->id,false)==null ) commit(diff(row,{-heatRecovery}))

//statusOnly
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(discharge and fan and run and cmd and equipRef==row->id,false)==null and read(discharge and fan and run and sensor and equipRef==row->id,false)!=null) commit(diff(row,{statusOnly}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and statusOnly).map row=> if(read(discharge and fan and run and cmd and equipRef==row->id,false)!=null) commit(diff(row,{-statusOnly}))

//cmdOnly
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(discharge and fan and run and cmd and equipRef==row->id,false)!=null and read(discharge and fan and run and sensor and equipRef==row->id,false)==null) commit(diff(row,{cmdOnly}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and cmdOnly).map row=> if(read(discharge and fan and run and sensor and equipRef==row->id,false)!=null) commit(diff(row,{-cmdOnly}))

//scheduled
  readAll(ahu or fcu or vrf or fan or rtu or airConditioner).map row=> if(read(occupied and cmd and equipRef==row->id,false)!=null) commit(diff(row,{scheduled}))
  readAll((ahu or fcu or vrf or fan or rtu or airConditioner) and scheduled).map row=> if(read(occupied and cmd and equipRef==row->id,false)==null) commit(diff(row,{-scheduled}))

//freshAir
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(outside and damper and position and (cmd or sensor) and equipRef==row->id,false)!=null) commit(diff(row,{freshAir}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and freshAir).map row=> if(read(outside and damper and position and (cmd or sensor) and equipRef==row->id,false)==null) commit(diff(row,{-freshAir}))

//chilledTemp
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(chilled and water and temp and sensor and equipRef==row->id,false)!=null) commit(diff(row,{chilledTemp}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and chilledTemp).map row=> if(read(chilled and water and temp and sensor and equipRef==row->id,false)==null) commit(diff(row,{-chilledTemp}))

//hotTemp
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(hot and water and temp and sensor and equipRef==row->id,false)!=null) commit(diff(row,{hotTemp}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and hotTemp).map row=> if(read(hot and water and temp and sensor and equipRef==row->id,false)==null) commit(diff(row,{-hotTemp}))

 //raFan
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read("return" and fan and equipRef==row->id,false)!=null)) commit(diff(row,{raFan}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and raFan).map row=> if((read("return" and fan and equipRef==row->id,false)==null)) commit(diff(row,{-raFan}))

  //eaFan
  readAll(ahu or fcu or vrf or fan or rtu or airConditioner).map row=> if((read(exhaust and fan and equipRef==row->id,false)!=null)) commit(diff(row,{eaFan}))
  readAll((ahu or fcu or vrf or fan or rtu or airConditioner) and eaFan).map row=> if((read(exhaust and fan and equipRef==row->id,false)==null)) commit(diff(row,{-eaFan}))

  //returnAir
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read("return" and damper and position and (cmd or sensor) and equipRef==row->id,false)!=null) commit(diff(row,{returnAir}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and freshAir).map row=> if(read("return" and damper and position and (cmd or sensor) and equipRef==row->id,false)==null) commit(diff(row,{-returnAir}))

//exhaustAir
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(exhaust and damper and position and (cmd or sensor) and equipRef==row->id,false)!=null) commit(diff(row,{exhaustAir}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and freshAir).map row=> if(read(exhaust and damper and position and (cmd or sensor) and equipRef==row->id,false)==null) commit(diff(row,{-exhaustAir}))

  //daAir
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(discharge and damper and position and (cmd or sensor) and equipRef==row->id,false)!=null) commit(diff(row,{daAir}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and freshAir).map row=> if(read(discharge and damper and position and (cmd or sensor) and equipRef==row->id,false)==null) commit(diff(row,{-daAir}))

  //coControl
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(co and sensor and equipRef==row->id,false)!=null) and read(co and sp and equipRef==row->id,false)!=null) commit(diff(row,{coControl}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and coControl).map row=> if((read(co and sensor and equipRef==row->id,false)==null) or read(co and sp and equipRef==row->id,false)==null) commit(diff(row,{-coControl}))

  //co2Control
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(co2 and sensor and equipRef==row->id,false)!=null) commit(diff(row,{co2Control}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and co2Control).map row=> if((read(co2 and sensor and equipRef==row->id,false)==null) or read(co2 and sp and equipRef==row->id,false)==null) commit(diff(row,{-co2Control}))

  //oaFlowControl
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(outside and air and flow and sensor and equipRef==row->id,false)!=null) and read(outside and air and flow and sp and equipRef==row->id,false)!=null) commit(diff(row,{oaFlowControl}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and oaFlowControl).map row=> if((read(outside and air and flow and sensor and equipRef==row->id,false)==null) or read(outside and air and flow and sp and equipRef==row->id,false)==null) commit(diff(row,{-oaFlowControl}))

 //preheatvalve
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if(read(preheat and valve and kind=="Number" and equipRef==row->id,false)!=null) commit(diff(row,{preheatvalve}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and preheatvalve).map row=> if(read(preheat and valve and kind=="Number" and equipRef==row->id,false)==null) commit(diff(row,{-preheatvalve}))

  //cvFB
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(cool and cmd and valve and kind=="Number" and equipRef==row->id,false)!=null) and (read(cool and sensor and valve and kind=="Number" and equipRef==row->id,false)!=null)) commit(diff(row,{cvFB}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and cvFB).map row=> if((read(cool and cmd and valve and kind=="Number" and equipRef==row->id,false)==null) or (read(cool and sensor and valve and kind=="Number" and equipRef==row->id,false)==null)) commit(diff(row,{-cvFB}))

  //hvFB
  readAll(ahu or fcu or vrf or rtu or airConditioner).map row=> if((read(heat and cmd and valve and kind=="Number" and equipRef==row->id,false)!=null) and (read(heat and sensor and valve and kind=="Number" and equipRef==row->id,false)!=null)) commit(diff(row,{hvFB}))
  readAll((ahu or fcu or vrf or rtu or airConditioner) and hvFB).map row=> if((read(heat and cmd and valve and kind=="Number" and equipRef==row->id,false)==null) or (read(heat and sensor and valve and kind=="Number" and equipRef==row->id,false)==null)) commit(diff(row,{-hvFB}))

  //dualDuct
  readAll(ahu).map row=> if(read(point and (hotDeck or coldDeck) and equipRef==row->id,false)!=null) commit(diff(row,{dualDuct}))
  readAll(ahu and dualDuct).map row=> if(read(point and (hotDeck or coldDeck) and equipRef==row->id,false)==null) commit(diff(row,{-dualDuct}))

readAll((ahu or fcu or vrf or rtu or airConditioner) and equip).sort("dis")

end
