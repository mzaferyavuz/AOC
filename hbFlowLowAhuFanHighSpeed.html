(eqp, dates,spdeviation:10,achDefaultLimit:4,fanThreshold:85,minTime:30min,maxTime:60min,freq:2,flowLowLimit:22,flowHighLimit:3100,pressureCheck:false,ahuPressureVal:50,pressureSpdeviation:50,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

site:readById(eqp->siteRef)
equipWorkOrder:toEquipWorkOrder(eqp)

sp:read(discharge and flow and sp and equipRef == eqp->id,false)
flow:read(discharge and flow and sensor and equipRef == eqp->id,false)
fanSpeedCmd:read(discharge and fan and speed and cmd and equipRef==eqp->id)
ahuStatus:read(discharge and fan and run and sensor and equipRef == eqp->id,false)
fanSpeedSensor:read(discharge and fan and speed and sensor and equipRef==eqp->id,false)

pressureSPPoint:read(discharge and pressure and sp and equipRef == eqp->id,false)
pressurePoint:read(discharge and pressure and sensor and equipRef == eqp->id,false)

if(flow==null) throw "No Flow sensor present in "+ dis(eqp)
if(sp==null) throw "No Flow set-point present in "+ dis(eqp)
if(ahuStatus==null) throw "AHU fan status point missing"+ dis(eqp)

if(flow["minVal"]!=null) flowLowLimit=(flow->minVal)
if(flow["maxVal"]!=null) flowHighLimit=(flow->maxVal)
if(eqp["achLimit"]!=null) achDefaultLimit=eqp["achLimit"]

spark:hisRead([flow,flowSP,ahuStatus],dates).hisInterpolate().map row => do
  if(row["v0"]!=na() and row["v1"]!=na() and row["v2"]!=na()) {ts: row->ts,v0:((row["v1"] - row["v0"])>(row["v1"]*(spdeviation/100))and(row["v2"]==true) and row["v0"]>=flowLowLimit and row["v0"]<=flowHighLimit)}
  else {ts: row->ts,v0:na()}
  end.hisFindPeriods(v=>(v!=na() and v==true))

fanSpeedCmdHigh:fanSpeedCmd.hisRead(dates).hisFindPeriods(v=>v>fanThreshold)

spark=hisPeriodIntersection([spark,fanSpeedCmdHigh])

if(fanSpeedCmd!=null and fanSpeedSensor!=null)do
  vsdCmdFBMisMatch:vsdCmdFeedbackMismatch_inter(fanSpeedCmd,fanSpeedSensor,dates,cmdFBMismatch)
  spark=hisPeriodIntersection([spark,vsdCmdFBMisMatch])
  end

if(pressureCheck==true) do

if((pressurePoint!=null) and (pressureSPPoint!=null)) do
  pressuremismatch:eqp.dischargeFanStatic_inter(dates,pressureSpdeviation,fanThreshold)
  spark=hisPeriodIntersection([spark,pressuremismatch])
  end

  else if((pressurePoint!=null)and(pressureSPPoint==null)) do
    pressuremismatch:pressurePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>ahuPressureVal))
    spark=hisPeriodIntersection([spark,pressuremismatch])
    end
  end

limitequip:eqp["achLimit"]
limitsite:site["achLimit"]

if(limitequip!=null) achDefaultLimit=eqp["achLimit"] else if(limitequip==null and limitsite!=null) achDefaultLimit=site["achLimit"]

achSPTable:hbAchCalc(eqp,dates,"Setpoint").hisFindPeriods((v, ts) =>(v!=na() and v>achDefaultLimit))
achSensorTable:hbAchCalc(eqp,dates,"Sensor").hisFindPeriods((v, ts) =>(v!=na() and v>achDefaultLimit))

spark=hisPeriodIntersection([spark,achSPTable,achSensorTable])

spark=occupancyandMaintenance(eqp,dates,spark)

spark1:spark.findAll(ts => ts->v0 > minTime)
spark2:spark.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  else if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
end
