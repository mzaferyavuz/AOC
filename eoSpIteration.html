(row,dates,aVal:95,factor:0.85,dev:0.75,filter:true) => do

  strategy:row["eoStrategy"]
  maxsp:row["maxeoSp"]
  dVal:row["eoVal"]
  eosp:row["eoSp"]

  //// Components
  temp:read(zone and temp and sensor and equipRef==row->id)
  if(row["ahu"]!=null) analog:read(cool and valve and cmd and equipRef==row->id)
  else if(row["vav"]!=null) analog:read(discharge and damper and cmd and equipRef==row->id) else analog:null

  //// Runtime
  if(row["ahu"]!=null or row["fcu"]!=null) rs:read(discharge and fan and run and sensor and equipRef==row->id)
  else if (row["vav"]!=null) rs:read(discharge and fan and run and sensor and equipRef==row->ahuRef->id)

  runtime:rs.hisRead(dates).hisFindPeriods(v=> v!=na() and v==true)

  rougezonesTemp:[].toGrid

  if(runtime.size>0) do
    runtime=runtime.hisRollup(sum,1day).renameCol("v0","runTime")

    ///////Loop Start
    i:0
    eoSpMax:row["maxeoSp"]
    if(eoSpMax!=null) maxdeviation:eoSpMax-eosp else maxdeviation:dev

    readAll(point)[0..99].each loop=> do

      offset:(maxdeviation/99)*i
      value:eosp+offset
      i=i+1
      grid:hisRead([temp,rs],dates).hisInterpolate.map ro=>do
      if(ro["v0"]!=na() and ro["v1"]!=na() and ro["v1"]==true) {ts:ro->ts,v0:ro["v0"]>(value),v1:value} else {ts:ro->ts,v0:na(),v1:value}
      end.hisFindPeriods(v=> v!=na() and v==true).hisRollup(sum,1day).hisFindAll(v=> v!=null)

      if(grid.size>0) grid=hisJoin([grid.map rows=> do {ts:rows["ts"],rogueTime:rows["v0"],sp:value,zones:row->dis} end,runtime])//.findAll(ts => ts->v0 > (runtime*factor))

      if(grid.size>0 and rougezonesTemp.size==0) rougezonesTemp=grid
      else if(grid.size>0 and rougezonesTemp.size>0) rougezonesTemp=addRows(rougezonesTemp,grid)

      end

    ///////Loop End
    end

  if(rougezonesTemp.size>0) rougezonesTemp=rougezonesTemp.reorderCols(rougezonesTemp.colNames.moveTo("ts", 0)).map loops=>do
  if(loops["sp"]!=null) {ts:loops->ts,issueTime:loops["rogueTime"],totalRunTime:loops["runTime"],ratio:(loops["rogueTime"]/loops["runTime"]).as(1),setPoint:loops["sp"],zon:loops["zones"]}
  end.sort("setPoint").sort("ts")

  diff:null
  prv:null
  pts:null

  rougezonesTem:rougezonesTemp.map l=> do

  cur:l["ratio"]
  cts:l["ts"]

  if(prv==null) do
    prv=cur
    pts=cts
    {ts:l->ts,rTime:l["issueTime"],eRunTime:l["totalRunTime"],ratio:cur,setpoint:l["setPoint"],devi:na(),zone:l["zon"]}
    end

  else do

    if(pts==cts) diff=prv-cur
    prv=cur
    pts=cts

    if(diff>0.1) {ts:l->ts,rTime:l["issueTime"],eRunTime:l["totalRunTime"],ratio:cur,setpoint:l["setPoint"],devi:diff,zone:l["zon"]}
    end

  end

  rougezonesTem.findAll(v=> v->devi!=na()).sortr("devi")// rougezonesTemp
  end
