(point,dates,tolerance:1,calday:8day,minTime:10000min,matchLowVal:2%,matchHighVal:98%,statuscheck:false,ruleWorkOrder:"None",priority:"Medium",type:"Operations") => do

      eqp:readById(point->equipRef)
      equipWorkOrder:toEquipWorkOrder(eqp)
      maintanancePoint:point->equipRef.tomaintenancePoint()

     issue:point.hisRead(toDateSpan((dates-calday)..dates)).hisFindAll(v=>(v!=na() and v>matchLowVal and v<matchHighVal) ).hisRollup(spread,10min).hisFindPeriods(a=>a<tolerance and a!=null)

      if(statuscheck==true)
                                  do
         pointequip:point->equipRef

      if(readById(pointequip)["ahu"]!=null) onPeriods:pointequip.ahuRunPeriods(dates) // if equip is ahu
         else if (readById(pointequip)["vav"]!=null) do  // if equip is vav check run period of associated air handler
           ahu:readById(pointequip)["ahuRef"]
           onPeriods:ahu.ahuRunPeriods(dates)
      end

      else do
         sts:read(run and sensor and equipRef==pointequip->id,false) //read run sensor point of the equipment
         if(sts!=null) onPeriods:sts.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==true)) //read history of status point if point is found
         else throw "No status sensor for equip"
      end

      issueTime:hisPeriodIntersection([issue,onPeriods]) //show issue only when equipment is ON
      end  //end of statuscheck


   else issueTime:issue

      spark_inter:issueTime

      if(maintanancePoint!=null) do
        maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
        spark_inter= hisPeriodIntersection([issueTime,maintanceHis])
    end


     spark_inter.findAll(ts=>ts->v0>minTime)
  .addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

    end
