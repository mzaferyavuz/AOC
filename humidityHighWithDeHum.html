(ahu,dates,error:5,coolingValveThreshold:10,threshold:5,humidityMin:5,humidityMax:95,humiditySpMin:20,humiditySpMax:80,chilledWatertempSp:7,chilledWatertempThreshold:1,check:true,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

equipWorkOrder:toEquipWorkOrder(ahu)
chilledWaterTempSensor:[].toGrid

humidity:read(zone and humidity and sensor and equipRef == ahu->id,false)
if(humidity==null) humidity=read("return" and humidity and sensor and equipRef == ahu->id,false)
if(humidity==null) throw("No humidity sensor point in:")+dis(ahu)

setpoint:read(zone and humidity and sp and equipRef == ahu->id,false)
if(setpoint==null) setpoint=read("return" and humidity and sp and equipRef == ahu->id,false)
if(setpoint==null) throw("No humidity sp point in:")+dis(ahu)


highTime:hisRead([setpoint,humidity],dates).hisInterpolate().map row=>do
if(row["v0"]!=na() and row["v1"]!=na())
                 {ts:row->ts,v0:(row["v1"] - row["v0"])>(row["v0"]*(error/100))}
            else {ts:row->ts,v0:na()}

                 end.hisFindPeriods(v=>v==true)

humLim:humidity.hisRead(dates).hisFindAll(a=> a!=na()).hisFindPeriods(v =>((v!=na() and (v > humidityMin) and (v<humidityMax))))
humSpLim:setpoint.hisRead(dates).hisFindAll(a=> a!=na()).hisFindPeriods(v =>((v!=na() and (v > humiditySpMin) and (v<humiditySpMax))))

coolPeriods:ahuCoolPeriods_inter(ahu,dates,coolingValveThreshold)

onTime:ahu.ahuRunPeriods(dates)


spark:hisPeriodIntersection([highTime,humLim,humSpLim,onTime,coolPeriods])

chillerPlant:ahu["chillerPlantRef"]

  if(chillerPlant!=null ) do

    chilleronperiod: chillerPlant.anyChilleron(dates)
   chilledWaterTemp:read(chilled and leaving and water and temp and sensor and equipRef==chillerPlant->id,false)
   chilledWaterTempsp:read(chilled and leaving and water and temp and sp and equipRef==chillerPlant->id,false)

  if(chilledWaterTemp!=null and chilledWaterTempsp==null) chilledWaterTempSensor=chilledWaterTemp.hisRead(dates).hisFindPeriods(v=>v!=na() and (v<chilledWatertempSp))
   else if (chilledWaterTemp!=null and chilledWaterTempsp!=null) do
   chilledWaterTempSensor=hisRead([chilledWaterTemp,chilledWaterTempsp],dates).hisFindAll(a=> a!=na()).hisInterpolate().map row => do
          if(row["v0"]!=na() and row["v1"]!=na()) {ts: row->ts,
                                            v0:
                                              if (abs(row["v0"] - row["v1"])<chilledWatertempThreshold) "okay"
                                              else "nok"
                                              }
              else {ts: row->ts,v0:na()}
        end.hisFindPeriods(matchPointVal(_,"okay"))
         end

  if(chilleronperiod!=null and chilledWaterTemp!=null) spark= hisPeriodIntersection([spark,chilleronperiod,chilledWaterTempSensor])

  else if(chilleronperiod!=null and chilledWaterTemp==null) spark= hisPeriodIntersection([spark,chilleronperiod])

  else if(chilleronperiod==null and chilledWaterTemp!=null) spark= hisPeriodIntersection([spark,chilledWaterTempSensor])

  end



  command:read(humidifier and cmd and equipRef == ahu->id,false)
   if(command!=null and check==true)
    do
    history:command.hisRead(dates)
     if(command->kind == "Bool") humidTime:history.hisFindPeriods(v=>(v!=na() and v==false))
  else humidTime:history.hisFindPeriods(v =>(v!=na() and v < threshold))
    spark=hisPeriodIntersection([spark,humidTime])
    end



 maintanancePoint:ahu.tomaintenancePoint()
 occupancyPoint:read(occupied and equipRef == ahu->id,false)

 if(maintanancePoint!=null)
   do
     maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
     spark=hisPeriodIntersection([spark,maintanceHis])
   end
 if(occupancyPoint!=null)
   do
      occupiedperiods:togetOccupancy(ahu,dates)
        if(occupiedperiods.size>0)
          do
            spark=hisPeriodIntersection([spark,occupiedperiods])
          end
   end

      spark1:spark.findAll(ts => ts->v0 > minTime)
      spark2:spark.findAll(ts => ts->v0 > maxTime)

   if(spark1.size>freq)
     do
     return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
     end
     else if(spark2.size>0)
     do
    return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
   end
end
