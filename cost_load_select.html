(equip,type:"supply",req:"kwT") => do

  kwT:null
  kwP:null
  kwhP:null

  if(type=="supply") do
    kwT=equip["saFanLoad"]
    if(kwT==null) kwP=read(discharge and elec and power and sensor and equipRef==equip->id,false)
    if(kwP==null) kwhP=read(discharge and elec and energy and sensor and equipRef==equip->id,false)
  end

    else if(type=="return") do
      kwT=equip["raFanLoad"]
      if(kwT==null) kwP=read("return" and elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read("return" and elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="exhaust") do
      kwT=equip["eaFanLoad"]
      if(kwT==null) kwP=read(exhaust and elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(exhaust and elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="humidifier") do
      kwT=equip["humLoad"]
      if(kwT==null) kwP=read(humidify and elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(humidify and elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="coolingTower") do
      kwT=equip["ctFanLoad"]
      if(kwT==null) kwP=read(elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="fan") do
      kwT=equip["fanCoolLoad"]
      if(kwT==null) kwP=read(elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="pump") do
      kwT=equip["pumpLoad"]
      if(kwT==null) kwP=read(elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="light") do
      kwT=equip["lightLoad"]
      if(kwT==null) kwP=read(elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="elecHeater") do
      kwT=equip["elecHeaterLoad"]
      if(kwT==null) kwP=read(heating and elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(heating and elec and energy and sensor and equipRef==equip->id,false)
    end

    else if(type=="dxCool") do
      kwT=equip["dxCoolLoad"]
      if(kwT==null) kwP=read(cooling and elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(cooling and elec and energy and sensor and equipRef==equip->id,false)
    end

    else do
      kwT=equip["elecLoad"]
      if(kwT==null) kwP=read(elec and power and sensor and equipRef==equip->id,false)
      if(kwP==null) kwhP=read(elec and energy and sensor and equipRef==equip->id,false)
    end


  if(req=="kwT") return kwT
  if(req=="kwP") return kwP
  if(req=="kwhP") return kwhP
  end
