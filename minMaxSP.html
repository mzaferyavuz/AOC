(sample,dates) => do

  all: ()=>do

  list:readAll(applied and eo and sp)

  list.map row=>do

  point:read(id==row->id)
  eq:point["equipRef"]

  if(point.has("pressure")==true)type:"Pressure"
  else if(point.has("temp")==true)type:"Temperature"

  status:read(run and sensor and equipRef==eq->id,false)
  if(status!=null)pointndstat:hisRead([point,status],dates).findAll(a=>a["v1"]==true)
  else if (status==null)pointndstat:point.hisRead(dates)

  pointMax:pointndstat.hisFindAll(a=>a!=na()).foldCol("v0",max)
  pointMin:pointndstat.hisFindAll(a=>a!=na()).foldCol("v0",min)

  pmax:pointMax.as(1)
  pmin:pointMin.as(1)

  dmax:pointndstat.hisFindPeriods(v=>v==pmax).foldCol("v0",sum)
  dmin:pointndstat.hisFindPeriods(v=>v==pmin).foldCol("v0",sum)

  {pointName:row->ebiref,parameter:type,equipName:row->equipRef->dis,parameter:type,max:pointMax,durMax:dmax,min:pointMin,durMin:dmin}
  end

  end

  temp: ()=>do

  list:readAll(applied and eo and sp and temp)

  list.map row=>do

  point:read(id==row->id)
  eq:point["equipRef"]

  if(point.has("temp")==true)type:"Temperature"

  status:read(run and sensor and equipRef==eq->id,false)
  if(status!=null)pointndstat:hisRead([point,status],dates).findAll(a=>a->v1==true)
  else if (status==null)pointndstat:point.hisRead(dates)

  pointMax:pointndstat.hisFindAll(a=>a!=na()).foldCol("v0",max)
  pointMin:pointndstat.hisFindAll(a=>a!=na()).foldCol("v0",min)

  pmax:pointMax.as(1)
  pmin:pointMin.as(1)

  dmax:pointndstat.hisFindPeriods(v=>v==pmax).foldCol("v0",sum)
  dmin:pointndstat.hisFindPeriods(v=>v==pmin).foldCol("v0",sum)

  {pointName:row->ebiref,parameter:type,equipName:row->equipRef->dis,parameter:type,max:pointMax,durMax:dmax,min:pointMin,durMin:dmin}
  end

  end

  press: ()=>do

  list:readAll(applied and eo and sp and pressure)

  list.map row=>do

  point:read(id==row->id)
  eq:point["equipRef"]

  if(point.has("pressure")==true)type:"Pressure"

  status:read(run and sensor and equipRef==eq->id,false)
  if(status!=null)pointndstat:hisRead([point,status],dates).findAll(a=>a->v1==true)
  else if (status==null)pointndstat:point.hisRead(dates)

  pointMax:pointndstat.hisFindAll(a=>a!=na()).foldCol("v0",max)
  pointMin:pointndstat.hisFindAll(a=>a!=na()).foldCol("v0",min)

  pmax:pointMax.as(1)
  pmin:pointMin.as(1)

  dmax:pointndstat.hisFindPeriods(v=>v==pmax).foldCol("v0",sum)
  dmin:pointndstat.hisFindPeriods(v=>v==pmin).foldCol("v0",sum)

  {pointName:row->ebiref,parameter:type,equipName:row->equipRef->dis,parameter:type,max:pointMax,durMax:dmax,min:pointMin,durMin:dmin}
  end

  end

      sampleMap:
        {
          "All EO Applied Setpoints": all,
          "Temperature Setpoints":temp,
          "Pressure Setpoints":press
        }

        fn: sampleMap[sample]
        if (fn == null) fn = all
        fn()

  end
