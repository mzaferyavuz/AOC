(dates) => do

    grid2:[].toGrid
    grid:null
    ec:null
    scnt:null

    all:ruleSparks(null,dates,null)
    if(all.size>0) faultList:all.unique(["ruleRef"]).each row=>do

    ecount:all.findAll(a=>(a->ruleRef==row->ruleRef)).unique(["ruleRef"]).map x=>do
    rules:readAll(rule and auto and dis==x->ruleRef->dis)
    sts:"Manual"
    if(rules.size==1) sts="Auto"

    ec:all.findAll(a=>(a->ruleRef==row->ruleRef)).unique(["targetRef"]).foldCol("dur",count)
    scnt:all.findAll(a=>(a->ruleRef==row->ruleRef)).foldCol("dur",count)
    sdur:all.findAll(a=>(a->ruleRef==row->ruleRef)).foldCol("dur",sum)
    grid={ruleName:row->ruleRef->dis,sam_count:ec,spark_count:scnt,spark_dur:sdur,rulestatus:sts}
    end

    grid2=addRows(grid2,ecount)
    end

    grid2
    allrule:readAll(rule and not disabled).map row=>do
    rules:readAll(rule and auto and not disabled and dis==row->dis)
    status:"Manual"
    if(rules.size==1) status="Auto"
    if(grid2.findAll(v=> v->ruleName==row->dis).size==0) {ruleName:row->dis,rulestatus:status,sam_count:0,spark_count:0,spark_dur:0}
    end

    if(allrule.size>1) grid2=addRows(grid2,allrule)
    else if(allrule.size==1) grid2=addRow(grid2,allrule)

    help:readAll(rule and not disabled).keepCols(["dis","help"]).renameCol("dis","ruleName")//.addColMeta("ruleName", {dis:ruleName})
    grid2=join(grid2,help,"ruleName").renameCol("help","rule_Logic")

    grid2=addRow(grid2,{ruleName:"total",sam_count:grid2.colToList("sam_count").fold(sum),spark_count:grid2.colToList("spark_count").fold(sum),spark_dur:grid2.colToList("spark_dur").fold(sum)}).sort("ruleName")
    end
