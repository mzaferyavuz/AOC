(ahu,dates,coolvalveMinThreshold:0%,coolvalveMaxThreshold:2%,tempThreshold:2, minTime:30min,maxTime:60min,freq:2,disTempMinVal:6,disTempMaxVal:60,disTempSPMinVal:6,disTempSPMaxVal:60, damperMatch:2,check:true,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

  equipWorkOrder:toEquipWorkOrder(ahu)
  occupiedperiods:[].toGrid
  spark:[].toGrid
  spark1:[].toGrid
  spark2:[].toGrid

  clgPointNumPeriods:[].toGrid
  heatElement:read(heat and cmd and equipRef == ahu->id,false)
  coolElement:read(cool and cmd and equipRef == ahu->id,false)

  if(heatElement==null and coolElement==null)
  throw "No Heat or Cool Points"

  occupancyPoint:read(occupied and equipRef == ahu->id,false)

  coldDeckTemp: read(coldDeck and temp and sensor and equipRef == ahu->id, true)
  if(readAll(coldDeck and temp and sp and equipRef==ahu->id).size>1)
  throw "More than one SP"
  else
  coldDeckTempSp: read(coldDeck and temp and sp and equipRef == ahu->id,true)
  cdDamper:read(coldDeck and damper and cmd and equipRef== ahu->id,false)

  maintanancePoint:ahu.tomaintenancePoint()

  ahuOn:ahu.ahuRunPeriods_inter(dates)

  coolTime:ahuNoCoolPeriods_inter(ahu,dates,coolvalveMaxThreshold)
  coldDeckTempHis:coldDeckTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>disTempMinVal and v<disTempMaxVal))
  coldDeckTempSpHis:coldDeckTempSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>disTempSPMinVal and v<disTempSPMaxVal))
  issueTime:comparepoints(coldDeckTempSp,coldDeckTemp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))

  spark=hisPeriodIntersection([issueTime,coolTime,coldDeckTempHis,coldDeckTempSpHis,ahuOn])

  spark_inter:spark

  if(cdDamper!=null)
    do
      if(cdDamper->kind=="Bool")  cdDamperOpenPeriods:cdDamper.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false)) //check if Bool point is true
      else  cdDamperOpenPeriods:cdDamper.hisRead(dates).hisFindPeriods(v => (v!=na() and v < damperMatch))// else its analog and check if it is greater than match

    spark_inter= hisPeriodIntersection([cdDamperOpenPeriods,spark_inter])
    end

  if(heatElement!=null and spark_inter.size>0)
    do
    noHtgPeriods:ahu.ahuNoHeatPeriods_inter(dates,1)
    spark_inter= hisPeriodIntersection([spark_inter,noHtgPeriods])
    end

  if(occupancyPoint!=null and check== true and spark_inter.size>0)
    do
    occupiedperiods=togetOccupancy(ahu,dates)
    spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
    end

  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

  if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
