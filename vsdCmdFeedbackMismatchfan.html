src

(equipment,dates,ztype:"discharge",threshold:1,minTime:15min,ruleWorkOrder:"None",priority:"Medium",type:"Operations") => do

  equipWorkOrder:toEquipWorkOrder(equipment)

  out:read((parseFilter(ztype))  and fan and speed and cmd and equipRef == equipment->id)
  fb:read((parseFilter(ztype))  and fan and speed and sensor and equipRef == equipment->id)
  status:read((parseFilter(ztype))  and fan and run and sensor and equipRef == equipment->id)
  statusOn:status.hisRead(dates).hisFindPeriods(v=>(v==true and v!=na()))

  if(fb["unit"]=="Hz" and out["unit"]=="Hz")
  issue:hisRead([out,fb],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:abs((row["v1"])-(row["v0"]))>=threshold}
  else {ts:row->ts,v0:na()}
  end
  else if(fb["unit"]=="%"and out["unit"]=="%")
  issue:hisRead([out,fb],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:abs(((row["v1"]).as(1%))-(row["v0"]))>=threshold}
  else {ts:row->ts,v0:na()}
  end
  else if(fb["unit"]=="Hz"and out["unit"]=="%")
  issue:hisRead([out,fb],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:abs(((row["v1"]*2).as(1%))-(row["v0"]))>=threshold}
  else {ts:row->ts,v0:na()}
  end
  else if(fb["unit"]=="%"and out["unit"]=="Hz")
  issue:hisRead([out,fb],dates).map row=>do
  if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:abs(((row["v1"]))-(row["v0"]*2).as(1%))>=threshold}
  else {ts:row->ts,v0:na()}
  end
  issue1:issue.hisFindPeriods(v=>(v==true and v!=na()))
  hisPeriodIntersection([issue1,statusOn]).findAll(ts=> ts-> v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
