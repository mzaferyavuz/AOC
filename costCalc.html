(equip,spark1,type:"chiller",pf:1,execute:true) => do

  ruleCost:""
  site:readById(equip->siteRef)
  cost:site["elecCost"]

  if(spark1.size>0 and cost!=null and execute==true) do

    grid:[].toGrid
    pff:equip["pf"]
    if(pff!=null) pf=pff
    meter:equip["elecMeterLoad"]

  ////equipment type block to pick the load tag, kw, kwh point in the equipment (if available else returns null)
    kwT:cost_load_select(equip,type,"kwT")
    kwP:cost_load_select(equip,type,"kwP")
    kwhP:cost_load_select(equip,type,"kwhP")

  ////Load tag on equipment
  if(kwT!=null) return ruleCost=(((kwT.as(1))*cost*pf*100).round)/100

  ////kW Point inside equipment
  else if (kwT==null and kwP!=null) do

    tss:spark1.map row => do
      span:(row["ts"]..row["ts"]+row["v0"]).toSpan()
      a:kwP.hisRead(span,{limit:null}).hisClip.hisFindAll(v=> v!=na()).hisRollup(avg,1mo)
      if(a.size>0) energyd:a.toRec()["v0"] else energyd:null

      {ts:row["ts"],v0:energyd}
      end.hisFindAll(v=> v!=null)

      if(tss.size>0) tss=tss.hisRollup(avg,1mo).toRec()["v0"].as(1) else tss=null
    if(tss!=null) return ruleCost=((tss*cost*pf*100).round)/100
    else return ruleCost=""
    end

  ////kWh Point inside equipment
  else if (kwT==null and kwP==null and kwhP!=null) do

    tss:spark1.map row => do
      times:row["v0"].to(1hr).as(1)
      span:(row["ts"]..row["ts"]+row["v0"]).toSpan()
      b:kwhP.hisRead(span,{limit:null}).hisClip.hisFindAll(v=> v!=na()).hisRollup(sum,1mo)

      if(b.size>0) energycon:b.toRec()["v0"] else energycon:null
      if(energycon!=null) energycon=(energycon/times)

      {ts:row["ts"],v0:energycon}
      end.hisFindAll(v=> v!=null)

      if(tss.size>0) tss=tss.hisRollup(avg,1mo).toRec()["v0"].as(1) else tss=null
    if(tss!=null) return ruleCost=((tss*cost*pf*100).round)/100
    else return ruleCost=""
  end

  ////Only Meter is present for the equipment
  else if(kwT==null and kwP==null and kwhP==null and meter!=null) do

    pfact:read(pf and point and equipRef==meter->id,false)
    kWattH:read(elec and energy and point and equipRef==meter->id,false)
    kWatt:read(elec and demand and point and equipRef==meter->id,false)

    if(kWattH==null and kWatt==null) return ruleCost=""

      tss:spark1.map row => do
        cst:null
        energyd:null
        energycon:null
        times:row["v0"].to(1hr).as(1)
        span:(row["ts"]..row["ts"]+row["v0"]).toSpan()

        if(pfact!=null) do
          b:pfact.hisRead(span,{limit:null}).hisClip.hisFindAll(v=> v!=na()).hisRollup(avg,1mo)
          if (b.size>0) pf=b.toRec()["v0"]
          end

        if(kWatt!=null) do
          a:kWatt.hisRead(span,{limit:null}).hisClip.hisFindAll(v=> v!=na()).hisRollup(avg,1mo)
          if(a.size>0) do
          energyd=a.toRec()["v0"]
          if(energyd!=null) cst=((energyd.as(1)*cost*pf*100).round)/100
          end
          end

        else if(kWatt==null and kWattH!=null) do
          c:kWattH.hisRead(span,{limit:null}).hisClip.hisFindAll(v=> v!=na()).hisRollup(sum,1wk)
          if(c.size>0) energycon=c.toRec()["v0"]
          if(energycon!=null) cst=(((energycon/times).as(1)*cost*pf*100).round)/100
          end

        {ts:row["ts"],v0:cst}
        end.hisRollup(avg,1mo)

      if(tss.size>0) return ruleCost=((tss.toRec["v0"]*100).round)/100
        else return ruleCost=""
    end

        else return ruleCost=""

  ////Inital if condition closure
    else ruleCost=""
  end
