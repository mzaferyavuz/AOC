() => do

  rulesList:readAll(rule and not disabled).map r=>do

  typ:"None"
  ruleOn:r->ruleOn
  fnc:r->ruleFunc.split("(")[0]
  function:read(func and name==fnc,false)
  if(r["auto"]!=null)ams:"Auto" else ams:"Manual"

  equipmentlist:sparkDebugRule(r->id).findAll(v=> v["status"]=="ok")
  equipmentlist1:sparkDebugRule(r->id).findAll(v=> v->status=="err" and not v["errTrace"].contains("hisFindPeriods"))

  if(equipmentlist1.size>0) equipmentlist=addRows(equipmentlist,equipmentlist1)

  equipmentlist=equipmentlist.map row=>do
  r:readById(row->targetRef)
  if(r["equip"]!=null) equipment:r->dis else if(r["equip"]==null) equipment:r->equipRef->dis
  {target:equipment}
  end

  if(equipmentlist.size>0) equipmentlist=equipmentlist.colToList("target") else equipmentlist=[]

  if(funcs(lib=="aoc" and name==fnc).size>0) rType:"Extension Rule"
  else if(funcs(lib=="aoc" and name==fnc).size==0) rType:"Custom Rule"

  if(r->ruleFunc.contains("\"Energy\"")==true) typ="Energy" else if(r->ruleFunc.contains("\"Operations\"")==true) typ="Operations" else if (r->ruleFunc.contains("\"Comfort\"")==true) typ="Comfort"
  else if(r->ruleFunc.contains("\"energy\"")==true) typ="Energy" else if(r->ruleFunc.contains("\"operations\"")==true) typ="Operations" else if (r->ruleFunc.contains("\"comfort\"")==true) typ="Comfort"

  if(typ=="None" and rType=="Custom Rule" and function!=null) do
    if(function->src.contains("\"Energy\"")==true) typ="Energy" else if(function->src.contains("\"Operations\"")==true) typ="Operations" else if (function->src.contains("\"Comfort\"")==true) typ="Comfort"
    else if(function->src.contains("\"energy\"")==true) typ="Energy" else if(function->src.contains("\"operations\"")==true) typ="Operations" else if (function->src.contains("\"comfort\"")==true) typ="Comfort"
    end

  {ruleName:r->dis,ruleLibrary:rType,type:typ,funcName:fnc,eqFilter:ruleOn,listofequip:equipmentlist,autoManStatus:ams}

  end.sort("ruleName")

  final:comfortAoc().keepCols(["equip","zone"]).map r=> do

  znName:r->zone->dis

  autoComfort:rulesList.findAll(v=> v->listofequip.contains(znName) and v->autoManStatus=="Auto" and v->type=="Comfort").colToList("ruleName")
  manualComfort:rulesList.findAll(v=> v->listofequip.contains(znName) and v->autoManStatus=="Manual" and v->type=="Comfort").colToList("ruleName")

  autoOperation:rulesList.findAll(v=> v->listofequip.contains(znName) and v->autoManStatus=="Auto" and v->type=="Operations").colToList("ruleName")
  manualOperation:rulesList.findAll(v=> v->listofequip.contains(znName) and v->autoManStatus=="Manual" and v->type=="Operations").colToList("ruleName")

  autoEnergy:rulesList.findAll(v=> v->listofequip.contains(znName) and v->autoManStatus=="Auto" and v->type=="Energy").colToList("ruleName")
  manualEnergy:rulesList.findAll(v=> v->listofequip.contains(znName) and v->autoManStatus=="Manual" and v->type=="Energy").colToList("ruleName")

  {zoneName:znName,comfortRuleAuto:autoComfort,comfortRuleManual:manualComfort,operationsRuleAuto:autoOperation,operationsRuleManual:manualOperation,energyRuleAuto:autoEnergy,energyRuleManual:manualEnergy}
  end

  final.sort("zoneName")
  end
