src

(system,dates,spdeviation:10,minTime:30min,maxTime:60min,defsetpoint:200,flowMinVal:1,flowMaxVal:1500,flowSPMinVal:1,flowSPMaxVal:1500,freq:2,ruleWorkOrder:"None",priority:"High",type:"Comfort") => do

spark:[].toGrid
spark1:[].toGrid
spark2:[].toGrid
systemOn:[].toGrid

equipWorkOrder:toEquipWorkOrder(system)
flow:read((chilled or hot) and water and flow and sensor and kind == "Number" and equipRef->id==system->id)
flowSp:read((chilled or hot) and water and flow and sp and kind == "Number" and equipRef->id==system->id,false)
flowlim:flow.hisRead(dates).hisFindPeriods(v=> v!=na() and v<flowMaxVal and v>flowMinVal)

if(system["chillerPlant"]!=null) systemOn=system.anyChillerOnInSystem(dates)
else if(system["boilerPlant"]!=null) systemOn=system.anyBoilerOnInSystem(dates)

if(flowSp==null)
do
issue:flow.hisRead(dates).hisFindPeriods(v=>(v!=na() and v<((defsetpoint)-((spdeviation/100)*(defsetpoint)))))
spark=hisPeriodIntersection([systemOn,flowlim,issue])
end

else
do
flowSPLim:flowSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>flowSPMinVal and v<flowSPMaxVal))

dev:hisRead([flow,flowSp],dates).map row=>do
if(row["v0"]!=na() and row["v1"]!=na()) {ts:row->ts,v0:(row["v0"]<(row["v1"]-row["v1"]*(spdeviation/100)))}
else {ts:row->ts,v0:na()}
end

issue:dev.hisFindPeriods(v=>(v!=na() and v==true))
spark=hisPeriodIntersection([systemOn,issue,flowlim,flowSPLim])
end

spark1=spark.findAll(ts => ts->v0 > minTime)
spark2=spark.findAll(ts => ts->v0 > maxTime)

if(spark1.size>freq)return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
if(spark2.size>0)return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
end
