(dates) => do

  j:0
  grid:[].toGrid

  pnts:readAll(point and ebiref and hisEnd and equipRef->meter).each row => do

  timediff:null
  maxTimediff:null
  prevTimestamp:null
  history:row.hisRead(dates).hisClip

    if(history.size>1) hist:history[0..-2].map r => do

      diff:null
      curTimestamp:r["ts"]

      if(prevTimestamp==null) prevTimestamp=curTimestamp
      else if (prevTimestamp!=null) diff=(curTimestamp-prevTimestamp).to(1min)
      prevTimestamp=curTimestamp

      {pnt:row->dis,difference:diff}
    end.findAll(v=> v["difference"]!=null)

      if(history.size>1) do
        timediff=hist.colToList(hist.colNames[1]).fold(min)
        maxTimediff=hist.colToList(hist.colNames[1]).fold(max)//+1min
      end

  ab:{point:row->dis,minInterval:timediff,maxInterval:maxTimediff}

  if(j==0) grid=ab.toGrid
  else if(j>0) grid=addRow(grid,ab)
  j=j+1
  end

  grid//.findAll(v=> v["minInterval"]!=null)
  end
