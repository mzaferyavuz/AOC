func:ruleLanguageChange

ruleLanguageChange(language)

Run this function to change the rule languages dis and help content. csv name should be languageChange.csv. Please input desired language to translate to: German Dutch Italian French Spanish

src

(language) => do
  csv:ioReadCsv(`io/languageChange.csv`)
  libraryFuncs:funcs(ver==2)
  readAll(rule).each row=> do
    //German
    if(language=="German") do
    record:csv.find(a=>a->name==row->ruleFunc.split("(")[0])
    if (record["disGerman"]!=null) diff(row,{dis:record->disGerman,help:record->helpGerman, disEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->dis,helpEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->help, translated}).commit()
    else diff(row,{notTranslated, disabled}).commit()
    end
    //Dutch
    else if(language=="Dutch") do
    record:csv.find(a=>a->name==row->ruleFunc.split("(")[0])
    if (record["disDutch"]!=null) diff(row,{dis:record->disDutch,help:record->helpDutch, disEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->dis,helpEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->help, translated}).commit()
    else diff(row,{notTranslated, disabled}).commit()
    end
    //Italian
    else if(language=="Italian") do
    record:csv.find(a=>a->name==row->ruleFunc.split("(")[0])
    if (record["disItalian"]!=null) diff(row,{dis:record->disItalian,help:record->helpItalian, disEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->dis,helpEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->help, translated}).commit()
    else diff(row,{notTranslated, disabled}).commit()
    end
    //French
    else if(language=="French") do
    record:csv.find(a=>a->name==row->ruleFunc.split("(")[0])
    if (record["disFrench"]!=null) diff(row,{dis:record->disFrench,help:record->helpFrench, disEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->dis,helpEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->help, translated}).commit()
    else diff(row,{notTranslated, disabled}).commit()
    end
   //Spanish
    else if(language=="Spanish") do
    record:csv.find(a=>a->name==row->ruleFunc.split("(")[0])
    if (record["disSpanish"]!=null) diff(row,{dis:record->disSpanish,help:record->helpSpanish, disEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->dis,helpEn:funcs(name==row->ruleFunc.split("(")[0]).toRec->help, translated}).commit()
    else diff(row,{notTranslated, disabled}).commit()
    end

    else throw "Please input desired language to translate to: \"German\", \"Dutch\", \"Italian\", \"French\", \"Spanish\" "

  end


  //adding SI params to all rules
  readAll(rule).each r=>do
    if((r->ruleFunc).contains("(")==false)
      do
      record2:libraryFuncs.find(a=>a->name==r->ruleFunc.split("(")[0])
      diff(r,{ruleFunc:[r->ruleFunc,record2->param].concat}).commit
      end

     end
end
