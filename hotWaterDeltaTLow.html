(ahu,dates,diff:2,tempSP:71,hotWaterTempThreshold:2,ahuHotWaterdiffTemp:5,heatValveThreshold:98,tempThreshold:2,minTime:30min,maxTime:60min,freq:2,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

     ahuRunTime:ahu.ahuRunPeriods(dates)
     spark:[].toGrid()


     equipWorkOrder:toEquipWorkOrder(ahu)

      boilerPlant:ahu["boilerPlantRef"]

      datsp:read(discharge and temp and sp and equipRef==ahu->id,false)
      zoneTempSp:read(zone and temp and sp and equipRef==ahu->id,false)

      if(boilerPlant!=null ) do

        boileronperiod: boilerPlant.anyboileron(dates)
        hotWaterTemp:read(hot and leaving and water and temp and sensor and equipRef==boilerPlant->id)
        hotWaterTempSP:read(hot and leaving and water and temp and sp and equipRef==boilerPlant->id,false)
           if(hotWaterTempSP!=null) do
             issue:hisRead([hotWaterTemp,hotWaterTempSP],dates).map row=>do

        if(row["v0"]!=na() and row["v1"]!=na())

             {ts:row->ts,v0:abs(row["v0"]-row["v1"])<hotWaterTempThreshold}

             else {ts:row->ts,v0:na()}

           end.hisFindPeriods(v=>(v!=na() and v==true))
          end
        else do
         issue:hotWaterTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>tempSP))
       end
    else throw "No Boiler Plant Ref"


     heatPointNum:readAll(cmd and heat and position and kind=="Number" and equipRef==ahu->id).hisRead(dates).hisMap(v=> if(v!=na() and v > heatValveThreshold) 1 else 0).hisFoldCols(sum).map row=>do
        if(row["v0"]>0) do
            {ts:row->ts,v0:true}
          end
        else
            {ts:row->ts,v0:false}
          end.hisFindPeriods(v=>(v!=na() and v==true))


     hotEnteringTempSensor:read(entering and hot and water and temp and sensor and kind=="Number" and equipRef==ahu->id)
     hotLeavingTempSensor:read(hot and water and temp and sensor and leaving and kind=="Number" and equipRef==ahu->id)
     issue1:comparepoints(hotLeavingTempSensor,hotEnteringTempSensor,dates,ahuHotWaterdiffTemp).hisFindPeriods(matchPointVal(_,"greater"))


      if(datsp!=null) do
    dat:read(discharge and temp and sensor and equipRef==ahu->id)
    issue2:comparepoints(datsp,dat,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
    end
    else if(zoneTempSp!=null) do
    zoneTemp:read(zone and temp and sensor and equipRef==ahu->id)
    issue2:comparepoints(zoneTempSp,zoneTemp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
    end



     spark=hisPeriodIntersection([issue,heatPointNum,boileronperiod,ahuRunTime,issue1,issue2])

      maintanancePoint:ahu.tomaintenancePoint()
     occupancyPoint:read(occupied and equipRef == ahu->id,false)

 if(maintanancePoint!=null)
   do
     maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
     spark=hisPeriodIntersection([spark,maintanceHis])
   end
 if(occupancyPoint!=null)
   do
      occupiedperiods:togetOccupancy(ahu,dates)
        if(occupiedperiods.size>0)
          do
            spark=hisPeriodIntersection([spark,occupiedperiods])
          end
   end

      spark1:spark.findAll(ts => ts->v0 > minTime)
      spark2:spark.findAll(ts => ts->v0 > maxTime)

   if(spark1.size>freq)
     do
     return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
     end
     else if(spark2.size>0)
     do
    return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
   end

  end
