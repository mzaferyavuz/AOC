(energydashboard:true) => do
  ioDelete(`io/dashboardchecklog.txt`)
  file:ioAppend(`io/dashboardchecklog.txt`)
  ioWriteStr("---Starting Dashboard Readiness Checks--- \n",file)
  errorCount:0
  warnCount:0
  comfortExtn:read(ext =="comfortdashboard",false)
  if(comfortExtn==null) do
  ioWriteStr("comfortdashboard extension not enabled \n",file)
  errorCount=errorCount+1
  end
  vavs:readAll(vav)
  vavs.each vav =>do
      if(vav["dis"]==null) do
      ioWriteStr(""+vav->dis+" does not have dis tag"+"\n",file)
      errorCount=errorCount+1
    end

    if(read(zone and temp and sensor and equipRef==vav->id,false)==null) do
      ioWriteStr("No Zone Temperature Sensor in "+vav->dis+"\n",file)
      errorCount=errorCount+1
    end

    if(read(zone and temp and sp and equipRef==vav->id,false)==null) do
      ioWriteStr("No Zone Temperature Setpoint in "+vav->dis+"\n",file)
      errorCount=errorCount+1
    end

    ahu:vav["ahuRef"]
        if(ahu==null) do
      ioWriteStr("No ahuRef assigned for "+dis(vav)+"\n",file)
      errorCount=errorCount+1
    end

    else if(read(discharge and fan and run and sensor and equipRef==ahu->id,false)==null) do
      ioWriteStr("No Occupancy point for "+ahu->dis+" serving "+dis(vav)+"\n",file)
      errorCount=errorCount+1
    end
  end// end of vav each loop
  fcus:readAll(fcu or (ahu and directZone))
  fcus.each fcu =>do
    if(read(zone and temp and sensor and equipRef==fcu->id,false)==null) do
      ioWriteStr("No Zone Temperature Sensor in "+dis(fcu)+"\n",file)
      errorCount=errorCount+1
    end

    if(read(zone and temp and sp and equipRef==fcu->id,false)==null) do
      ioWriteStr("No Zone Temperature Setpoint in "+dis(fcu)+"\n",file)
      errorCount=errorCount+1
    end

    if(read(discharge and fan and run and sensor and equipRef==fcu->id,false)==null) do
      ioWriteStr("No Fan Status "+dis(fcu)+"\n",file)
      errorCount=errorCount+1
    end
  end// end of fcu/ahu each loop

  zneqps:readAll(equip).findAll(e=> read(zone and temp and (sensor or sp) and equipRef==e->id,false)!=null)
  zneqps.each zneqp=>do
     if(zneqp["fcu"]==null and zneqp["vav"]==null and zneqp["directZone"]==null and zneqp["multiZone"]==null) do
      ioWriteStr("Equipment contains zone points, but not Zone equipment "+dis(zneqp)+"\n",file)
      errorCount=errorCount+1
    end
  end

  ioWriteStr(""+errorCount+" Errors Found"+"\n",file)
  ioWriteStr("---End of Dashboard Readiness Checks---\n",file)
  ioReadLines(file)

end
