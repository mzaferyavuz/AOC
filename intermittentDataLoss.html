(emailId) => do

  grid:[].toGrid
  pnts:intermitDL(yesterday()..today()).each row => do

  hist:[].toGrid
  name:row["point"]
  defInterval:row["minInterval"]
  if(defInterval!=null) defInterval=defInterval.round
  maxInterval:row["maxInterval"]
  pnt:read(point and dis==name,false)

  if(pnt!=null and defInterval!=null and (defInterval!=maxInterval)) do
  diffPrev:0min
  prevTimestamp:null

    hist=pnt.hisRead(yesterday()..today())[0..-2].hisFindAll(v=> v!=null).map r => do
    diff:null
    curTimestamp:r["ts"]

      if(prevTimestamp!=null) do
        diff=(curTimestamp-prevTimestamp).to(1min)
        span:(prevTimestamp..curTimestamp).toSpan
      end

    prevTimestamp=curTimestamp

    if(prevTimestamp!=null and diff!=defInterval and diff!=null) {point:pnt->dis,timeSpan:span,difference:diff,default:defInterval}
      else {point:pnt->dis,timeSpan:null,difference:null,default:null}
    end.findAll(v=> v["difference"]!=null)
  end

  if(grid.size==0 and hist.size>0) grid=hist
  else if (hist.size>0) grid=addRows(grid,hist)
  end

  cols:grid.colNames.moveTo("default", -1).moveTo("difference", -1).moveTo("point", 0).moveTo("timeSpan", -1)
  grid.reorderCols(cols)

  if(grid.size>0) do
  project:read(projMeta)->dis
  subject:project+" intermittent Data Loss Notification"
  grid.ioWriteExcel(`io/intermittent_Data_Loss_points.xls`)
  emailSend(emailId,subject,"Please find the attached list of points which has data loss intermittently",`io/intermittent_Data_Loss_points.xls`)
  end
  end
