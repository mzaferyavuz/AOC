(ahu,dates,diff:2,oadMin:15,oatLo:12,oatHi:21,minTime:15min,ruleWorkOrder:"None",priority:"High",type:"Energy") => do

   ahuRunTime:ahu.ahuRunPeriods(dates)
   oadClosePeriods:toAhuOutsideDamper(ahu).hisRead(dates).hisFindPeriods(val=>(val<=oadMin and val!=na()))
   oat:toAhuOutsideTemp(ahu)
   oatSuitabilityPeriods:oat.hisRead(dates).hisFindPeriods(val=>(val>oatLo and val<oatHi and val!=na()))
   coolPeriods:ahu.ahuCoolPeriods(dates)
    equipWorkOrder:toEquipWorkOrder(ahu)

   econotused:toAhuOutsideDamper(ahu).hisRead(dates).hisFindPeriods(val=>(val<=oadMin and val!=na()))

   ecomode:read(economizer and enable and cmd  and equipRef == ahu->id,false)
    if(ecomode!=null) do
    ecomodeperiod:ecomode.hisRead(dates).hisFindPeriods(v=>(v==false and v!=na()))
    econotused=hisPeriodIntersection([oadClosePeriods,ecomodeperiod])
    end

   raen:ahu.toAhuReturnEnthalpy(false)
    if(raen!=null) do
    oaen:toAhuOutsideEnthalpy(ahu)
    enthSuitablePeriods:comparepoints(raen,oaen,dates,diff).hisFindPeriods(matchPointVal(_,"greater"))
    return result:hisPeriodIntersection([enthSuitablePeriods,econotused,oatSuitabilityPeriods,coolPeriods,ahuRunTime]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
   end

  rat:ahu.toAhuReturnTemp(false)
  if(rat!=null) do
   tempSuitabilityPeriods:comparepoints(rat,oat,dates,diff).hisFindPeriods(matchPointVal(_,"greater"))
  return result:hisPeriodIntersection([tempSuitabilityPeriods,econotused,oatSuitabilityPeriods,coolPeriods,ahuRunTime]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  end

  return result:hisPeriodIntersection([oatSuitabilityPeriods,econotused,coolPeriods,ahuRunTime]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
