(point, dates, tolerance: 0.1,ruleWorkOrder:"None",priority:"Medium",type:"Operations",minTime:23.8,check:false) => do

  eqp:readById(point->equipRef)
  equipWorkOrder:toEquipWorkOrder(eqp)

  date:dates-(1day)

  grid1:hisRead(point, date..dates)

  if(check==false)
  do return  grid1.hisRollup(spread,23hr).hisFindPeriods((v, ts) => v != null and v != na() and v.abs < tolerance).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  end

  if(check==true)
  do

   pointequip:point->equipRef

    if(readById(pointequip)["ahu"]!=null) onPeriods:pointequip.ahuRunPeriods(date..dates) // if equip is ahu
    else if (readById(pointequip)["vav"]!=null) do  // if equip is vav check run period of associated air handler
         ahu:readById(pointequip)["ahuRef"]
         onPeriods:ahu.ahuRunPeriods(date..dates)
         end
    else do
    sts:read(run and sensor and equipRef==pointequip->id,false) //read run sensor point of the equipment
      if(sts!=null) onPeriods:sts.hisRead(date..dates).hisFindPeriods(v=>(v!=na() and v==true)) //read history of status point if point is found
      else throw "No status sensor for equip"
      end


    grid2:grid1.hisRollup(spread,23hr)
    .hisFindPeriods((v, ts) => v != null and v != na() and v.abs < tolerance)

     return hisPeriodIntersection([grid2, onPeriods]).findAll(ts=>ts->v0>minTime).addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
  end






end
