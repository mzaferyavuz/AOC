(equip,spark1,type:"pump",pf:1,defTime:true,time:960,execute:true) => do

  ruleCost:""
  site:readById(equip->siteRef)
  cost:site["elecCost"]

  if(spark1.size>0 and cost!=null and execute==true) do
  kw:null
  kwm:null
  kwhm:null
  kWatts:null
  grid:[].toGrid
  pff:equip["pf"]
  meter:equip["elecMeterLoad"]

    if(type=="pump") do
    kw=equip["pumpLoad"]
    if(kw==null) kwm=read(pump and power and sensor and equipRef==equip->id,false)
    if(kwm==null or defTime==false) kwhm=read(pump and energy and sensor and equipRef==equip->id,false)
    end

      else if(type=="light") do
      kw=equip["lightLoad"]
      if(kw==null) kwm=read(lighting and power and sensor and equipRef==equip->id,false)
      if(kwm==null or defTime==false) kwhm=read(lighting and energy and sensor and equipRef==equip->id,false)
      end

        else if(type=="humidifier") do
        kw=equip["humLoad"]
        if(kw==null) kwm=read(humidifier and power and sensor and equipRef==equip->id,false)
        if(kwm==null or defTime==false) kwhm=read(humidifier and energy and sensor and equipRef==equip->id,false)
        end

    if(meter!=null and kwm==null and kw==null and kwhm==null) do
    pfact:read(pf and point and equipRef==meter->id,false)
    kWattH:read(elec and energy and point and equipRef==meter->id,false)
    kWatt:read(elec and demand and point and equipRef==meter->id,false)
    if(kWattH==null and kWatt==null) return ruleCost=""
    end

      else if(kwm!=null or kw!=null or kwhm!=null) do
      pfact:null
      kWatt:null
      kWattH:null
      if(kw!=null) kWatts=kw
      else if(kwm!=null) kWatt=kwm
      if(kwhm!=null) kWattH=kwhm
      end

    if(meter!=null or kwm!=null or kw!=null or kwhm!=null) do

    tss:spark1.map row => do
    energycon:""
    energyd:""
    span:(row["ts"]..row["ts"]+row["v0"]).toSpan()

    if(kWatts!=null) energyd=kWatts
    else if(kWatts==null and kWatt!=null) do
    a:kWatt.hisRead(span).hisFindAll(v=> v!=na()).hisRollup(avg,1mo)
    if(a.size>0) energyd=a.addColMeta("v0",{name:"energyd"}).toRec()["v0"]
    end

    if(pff!=null) pf=pff
    else if(pff==null and pfact!=null) do
    b:pfact.hisRead(span).hisFindAll(v=> v!=na()).hisRollup(avg,1mo)
    if (b.size>0) pf=b.addColMeta("v0",{name:"pf"}).toRec()["v0"]
    end

    if(kWattH!=null) do
    c:kWattH.hisRead(span).hisFindAll(v=> v!=na()).hisRollup(sum,1mo)
    if(c.size>0) energycon=c.addColMeta("v0",{name:"energy"}).toRec()["v0"]
    end

    {ts:row["ts"],sprk:row["v0"],pfacter:pf,energy:energycon,demand:energyd}
    end

    grid=tss.map row=> do

    if(pfact!=null or pff!=null) pfm:row["pf"]
    if(kWatt!=null or kWatts!=null) kwma:row["demand"].as(1)
    if(kWattH!=null) kwh:row["energy"].as(1)

    times:row["sprk"].to(1hr).as(1)

    if(defTime==false) do
    time=times
    if(kWattH!=null) ruleCost=cost*kwh
    else if(kWattH==null and (kWatt!=null or kWatts!=null) and pff!=null) ruleCost=cost*kwma*pff*time
    else if(kWattH==null and (kWatt!=null or kWatts!=null) and pfact!=null) ruleCost=cost*kwma*pfm*time
    else if(kWattH==null and (kWatt!=null or kWatts!=null) and pfact==null and pff==null) ruleCost=cost*kwma*pf*time
    end

      else if(defTime==true) do
      if((kWatt!=null or kWatts!=null) and pff!=null) ruleCost=cost*kwma*pff*time
      else if((kWatt!=null or kWatts!=null) and pfact!=null) ruleCost=cost*kwma*pfm*time
      else if((kWatt!=null or kWatts!=null) and pfact==null and pff==null) ruleCost=cost*kwma*pf*time
      else if(kWatt==null and kWatts==null and kWattH!=null) ruleCost=cost*(kwh/times)*time
      end

    {ts:row["ts"], ruleCost:ruleCost}
    end

  if(grid.size>0 and defTime==true) ruleCost=grid.hisRollup(avg,1mo).toRec()["ruleCost"]
  else if(grid.size>0 and defTime==false) ruleCost=grid.hisRollup(sum,1mo).toRec()["ruleCost"]
  else ruleCost=""
  end
  else ruleCost=""
  end

      else ruleCost=""
  end
