report:readAll(rule and not disabled).map row=> do

remark:null
list:sparkDebugRule(row->id)

grid1:null
grid:[].toGrid
if(ok.size>0) ok.each r=>do
grid=addRow(grid,{eq:r->targetRef->dis})
if(grid.size>0) grid1=grid.colToList("eq")
end

nok:list.findAll(v=> v->status=="err" and not v["errTrace"].contains("hisFindPeriods"))
grid2:null
grid4:[].toGrid
if(nok.size>0) nok.each s=>do
grid4=addRow(grid4,{eq:s->targetRef->dis})
if(grid4.size>0) grid2=grid4.colToList("eq")
end

nokh:list.findAll(v=>v->status=="err")
grid3:null
grid5:[].toGrid
if(nokh.size>0) nokh.findAll(v=> v["errTrace"].contains("hisFindPeriods")).each t=>do
grid5=addRow(grid5,{eq:t->targetRef->dis})
if(grid5.size>0) grid3=grid5.colToList("eq")
end

{rule:row->dis,description:row->help,status:status,rule_target:grid1,rule_targetInError:grid2,rule_targetInError_dataissue:grid3,remarks:remark}
end
