(r1:2022-09-14, r2:2022-09-15 /*bmsday*/ ) => do
   point:readAll(eo and (sp or speed) and kind=="Number")
   point.map row=>do
   eq:readById(row->equipRef)
   znTemp:row.hisRead(r1).hisCompare(r2).hisFindAll(v=>v!=na())
   occupancy:[].toGrid
   occupancy=read(occupied and cmd and equipRef==row->equipRef->id,false)
   if (occupancy ==null)
   occupancy=read(run and sensor and equipRef==row->equipRef->id)
   znTemp1:znTemp.hisInterpolate().map row=> do
   {ts:row->ts, v0:if((row["v0"]-row["v0x0"])<0) "less" else if((row["v0"]-row["v0x0"])==0) "equal" else "greater"}
    end


    occHis1:null//occupancy.togetOccupancy(r1)
    if(occupancy!=null and occupancy["kind"]=="Bool") occHis1=occupancy.hisRead(r1..r2).hisFindPeriods(v=>(v!=na() and v==true))
 else if(occupancy!=null and occupancy["kind"]=="Str") occHis1=occupancy.hisRead(r1..r2).hisFindPeriods(matchPointVal(_,"Occupied" ))



   o2:null
   o3:null
   o4:null
    if(occHis1.size>0)
    do

   znTemp2:znTemp1.hisFindPeriods(matchPointVal(_,"greater"))//.foldCol("v0",sum)
   znTemp3:znTemp1.hisFindPeriods(matchPointVal(_,"less"))//.foldCol("v0",sum)
   znTemp4:znTemp1.hisFindPeriods(matchPointVal(_,"equal"))//.foldCol("v0",sum)
   o2=hisPeriodIntersection([znTemp2,occHis1]).foldCol("v0",sum)
   o3=hisPeriodIntersection([znTemp3,occHis1]).foldCol("v0",sum)
   o4=hisPeriodIntersection([znTemp4,occHis1]).foldCol("v0",sum)
   end

   {eoSetpoint:row->dis,type:if(row.has("pressure")) "Pressure" else if(row.has("temp")) "Temperature" else if(row.has("speed")) "Speed" else "Others",equip:eq->dis,higherTimes:o2,lowerTimes:o3, equalTimes:o4}
   end
 end
