//Rule to identify if box flow setpoint is not maintained with supply damper being open
(vav, dates,spdeviation:20,damperopenlimit:95,dpThreshold:50,minTime:30min) => do

//Read VAV flow, setpoint and damper position
 vavFlow: toVavFlow(vav)
 vavFlowSP: toVavFlowSp(vav)
 damperPosition:read(air and damper and discharge and equipRef == vav->id)

//Read AHU status
 ahu:vav.toAHUofVAV()
 ahuStatus:read(discharge and fan and run and sensor and equipRef == ahu->id)

pressSp:ahu.toAhuDischargePressureSp()
press:ahu.toAhuDischargePressure()


//Identify if the damper is fully open
 fullyOpen: hisRead(damperPosition,dates).hisFindPeriods(val=>(val!=na() and val > damperopenlimit))


//periods where flow is less than setpoint by more than spdeviation% and the ahu serving the VAV is on

   ps:hisRead([vavFlow,vavFlowSP,ahuStatus,press,pressSp],dates).map row => do
    if(row["v0"]!=na() and row["v1"]!=na() and row["v2"]!=na() and row["v3"]!=na() and row["v4"]!=na()) {ts: row->ts,v0:((row["v1"] - row["v0"])>(row["v1"]*(spdeviation/100))and(row["v2"]==true)and(abs(row["v3"]-row["v4"])<dpThreshold))}
       end

     flowSPNotMaintained:ps.hisFindPeriods((val,ts,his) =>(val!=na() and val == true))



//Identify fault periods. Intersection of those periods,where damper is fully open and setpoint is still not maintained.
    hisPeriodIntersection([flowSPNotMaintained,fullyOpen]).findAll(ts => ts->v0 > minTime)

end
