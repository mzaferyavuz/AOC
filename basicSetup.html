(systype, cType) => do

 if(systype!="FC" and systype!="EBI" and systype!="N4") throw "Please enter correct value->(FC,EBI or N4)"
 else if (cType!="HBS" and cType!="HCE" and cType!="CBM" and cType!="DEMO" and cType!="CHP" and cType!="TEST") throw "Please enter correct value->(HBS,HCE,CBM,DEMO,CHP orTEST)"
 do

 grid:[].toGrid

 if(read(projMeta)["enableBasicAuth"]==null) read(projMeta).diff({enableBasicAuth}).commit()
 if(read(projMeta)["evalTimeout"]==null) read(projMeta).diff({evalTimeout:3min}).commit()
 if(read(projMeta)["systemType"]==null) read(projMeta).diff({systemType:systype}).commit()
 if(read(projMeta)["customerType"]==null) read(projMeta).diff({customerType:cType}).commit()

  str1:userCur()->email
  if(read(job and dis=="Data Loss Notification",false)==null) diff(null,{job:marker(),jobExpr:("dataLoss(2h,["+"\""+str1+"\""+"])"),dis:"Data Loss Notification",jobSchedule:jobSchedule:"every 1hr"},{add}).commit()
  if(read(job and dis=="Daily Backup",false)==null) diff(null,{job:marker(),jobExpr:"jobFolioSnapshot()",dis:"Daily Backup",jobSchedule:toStr("on sat at "+random(10..23)+":"+random(10..59)+":"+random(10..59)),disabled},{add}).commit()
  if(read(job and dis=="Delete Old Backups",false)==null) diff(null,{job:marker(),jobExpr:"snapshotPurge(160hr)",dis:"Delete Old Backups",jobSchedule:toStr("on sun at "+random(10..23)+":"+random(10..59)+":"+random(10..59))},{add}).commit()
  if(read(job and dis=="Job Failure Alert",false)==null) diff(null,{job:marker(),jobExpr:"jobfailurealert(["+"\""+str1+"\""+"])",dis:"Job Failure Alert",jobSchedule:"every 8hr"},{add}).commit()
  if(read(job and dis=="Points reading NA",false)==null) diff(null,{job:marker(),jobExpr:"findingNAEmail(["+"\""+str1+"\""+"])",dis:"Points reading NA",jobSchedule:toStr("at "+random(10..23)+":"+random(10..59)+":"+random(10..59))},{add}).commit()
  if(read(job and dis=="Auto Tag",false)==null) diff(null,{job:marker(),jobExpr:"autotag()",dis:"Auto Tag",jobSchedule:"every 8hr"},{add}).commit()
  if(read(job and dis=="Rule Type Tag",false)==null) diff(null,{job:marker(),jobExpr:"ruleType()",dis:"Rule Type Tag",jobSchedule:"every 8hr"},{add}).commit()
  if(read(job and dis=="Intermittent Data Loss Notification",false)==null) diff(null,{job:marker(),jobExpr:("intermittentDataLoss(["+"\""+str1+"\""+"])"),dis:"Intermittent Data Loss Notification",jobSchedule:toStr("at "+random(10..23)+":"+random(10..59)+":"+random(10..59))},{add}).commit()
  if(read(job and dis=="Points reading Intermittent NA",false)==null) diff(null,{job:marker(),jobExpr:"intermittentNA(["+"\""+str1+"\""+"])",dis:"Points reading Intermittent NA",jobSchedule:toStr("at "+random(10..23)+":"+random(10..59)+":"+random(10..59))},{add}).commit()

  addSparkGroupings()

  pointCount:readAll(point).size
  if(pointCount>5000)
  commit(diff(read(ext=="rule"),{hotRefreshFreq:1hr}))
  else
  commit(diff(read(ext=="rule"),{hotRefreshFreq:10min}))

  customApp()
  analysisApp()
  end

 end
