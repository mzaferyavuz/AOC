src

(ahu,dates,tempThreshold:2,minTime:30min,maxTime:60min,heatvalveminThreshold:2%,heatvalvemaxThreshold:98%,hotWatertempLim:71,hotWatertempThreshold:2,freq:2,zoneTempMinVal:10,zoneTempMaxVal:35,zoneTempSPMaxVal:28,zoneTempSPMinVal:18,check:false,ruleWorkOrder:"None",priority:"Low",type:"Comfort") => do

   equipWorkOrder:toEquipWorkOrder(ahu)

   occupiedperiods:[].toGrid

   spark1:[].toGrid
   spark2:[].toGrid
   spark:[].toGrid
   hotWaterTempSensor:[].toGrid

   znTemp:ahu.toAhuZoneTemp(false)
   boilerPlant:ahu["boilerPlantRef"]
   if(znTemp==null) znTemp=ahu.toAhuReturnTemp()
   if(readAll(zone and temp and sp and equipRef==ahu->id).size>1) throw "More than one SP"
  else znTempSp:read(zone and temp and sp and equipRef==ahu->id)
   clgElement:read(cool and cmd and equipRef==ahu->id,false)
   heatTime:ahuHeatPeriodslessthan100(ahu,dates,heatvalveminThreshold,heatvalvemaxThreshold)
   maintanancePoint:ahu.tomaintenancePoint()
   occupancyPoint:read(occupied and equipRef == ahu->id,false)

   issueTime:comparepoints(znTempSp,znTemp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))
   ahuOn:ahu.ahuRunPeriods(dates)
   znTempSpHis:znTempSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>zoneTempSPMinVal and v<zoneTempSPMaxVal))
   znTempHis:znTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>zoneTempMinVal and v<zoneTempMaxVal))

    if(boilerPlant==null) spark=hisPeriodIntersection([issueTime,heatTime,ahuOn,znTempHis,znTempSpHis])

   if(boilerPlant!=null ) do
    hotWaterTemp:read(leaving and hot and water and temp and sensor and equipRef==boilerPlant->id,false)
    hotWaterTempSP:read(leaving and hot and water and temp and sp and equipRef==boilerPlant->id,false)
    boileronperiod: boilerPlant.anyboileron(dates)

   if(hotWaterTemp!=null and hotWaterTempSP==null) hotWaterTempSensor=hotWaterTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>hotWatertempLim))
   else if (hotWaterTemp!=null and hotWaterTempSP!=null) do
   hotWaterTempSensor=hisRead([hotWaterTemp,hotWaterTempSP],dates).hisInterpolate().map row => do
          if(row["v0"]!=na() and row["v1"]!=na()) {ts: row->ts,
                                            v0:
                                              if (abs(row["v0"] - row["v1"])<hotWatertempThreshold) "okay"
                                              else "nok"
                                              }
              else {ts: row->ts,v0:na()}
        end.hisFindPeriods(matchPointVal(_,"okay"))
         end


   if(boileronperiod!=null and hotWaterTemp!=null) spark=hisPeriodIntersection([issueTime,heatTime,ahuOn,boileronperiod,hotWaterTempSensor,znTempHis,znTempSpHis])

   else if(boileronperiod!=null and hotWaterTemp==null) spark=hisPeriodIntersection([issueTime,heatTime,ahuOn,boileronperiod,znTempHis,znTempSpHis])

   else if(boileronperiod==null and hotWaterTemp!=null) spark=hisPeriodIntersection([issueTime,heatTime,ahuOn,hotWaterTempSensor,znTempHis,znTempSpHis])

   end

    spark_inter:spark

    if(maintanancePoint!=null)
     do
       maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
       spark_inter= hisPeriodIntersection([spark,maintanceHis])
     end


   if(clgElement!=null and spark_inter.size>0)
     do
       noClgPeriods:ahu.ahuNoCoolPeriods_inter(dates,1)
       spark_inter= hisPeriodIntersection([spark_inter,noClgPeriods])
     end

   if(occupancyPoint!=null and check== true and spark_inter.size>0)
     do
     occupiedperiods=togetOccupancy(ahu,dates)
     spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
     end

   spark1=spark_inter.findAll(ts => ts->v0 > minTime)
   spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

   if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
 else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

 end
