(pack:"Dynamic") => do

  grid:[].toGrid
  line:"_____________________________________________________________"
  ver:readAll(rule).map row=> do
  pinum:null
  pinumber:funcs(ver==2).findAll(v=> v->name==row["ruleFunc"].split("(")[0]).size
  if(pinumber>0) pinum=funcs(ver==2).findAll(v=> v->name==row["ruleFunc"].split("(")[0]).toRec["pi"]
  {ruleName:row->dis,function:(row["ruleFunc"].split("(")[0]),ruleOn:row->ruleOn,piNumber:pinum}
  end

  oldRules:readAll(rule and not disabled).findAll(v=> funcs(ver==1).findAll(a=> a->name==v->ruleFunc.split("(")[0]).size>0).map row=> do
  {ruleName:row->dis,function:row->ruleFunc.split("(")[0],ruleOn:row->ruleOn,targetcount:readAll(parseFilter(row->ruleOn)).size}
  end.sortr("targetcount")

  custom:readAll(rule and not disabled).findAll(v=> funcs(ver==1 or ver==2).findAll(a=> a->name==v->ruleFunc.split("(")[0]).size==0).map row=> do
  {ruleName:row->dis,function:row->ruleFunc.split("(")[0],ruleOn:row->ruleOn,targetcount:readAll(parseFilter(row->ruleOn)).size}
  end.sortr("targetcount")

  if(pack=="Dynamic") missing:funcs(controlsBasic and ver==2).map row=> do
  if(ver.findAll(v=> v->function==row->name).size==0) {ruleName:row->dis,function:row->name,ruleOn:row->ruleOn,targetcount:readAll(parseFilter(row->ruleOn)).size,piNumber:row["pi"]}
  end.sortr("targetcount")

  else if(pack=="Assurance") missing:funcs((controlsBasic or controlsStandard) and ver==2).map row=> do
  if(ver.findAll(v=> v->function==row->name).size==0) {ruleName:row->dis,function:row->name,ruleOn:row->ruleOn,targetcount:readAll(parseFilter(row->ruleOn)).size,piNumber:row["pi"]}
  end.sortr("targetcount")

  else if(pack=="Performance") missing:funcs((controlsBasic or controlsStandard or energy) and ver==2).map row=> do
  if(ver.findAll(v=> v->function==row->name).size==0) {ruleName:row->dis,function:row->name,ruleOn:row->ruleOn,targetcount:readAll(parseFilter(row->ruleOn)).size,piNumber:row["pi"]}
  end.sortr("targetcount")

  preOld:[{ruleName:line,function:line,ruleOn:line,targetcount:line,piNumber:line},{ruleName:"Old version 1 rules in the site "+"("+oldRules.size+" Rules)",function:line,ruleOn:line,targetcount:line,piNumber:line},{ruleName:line,function:line,ruleOn:line,targetcount:line,piNumber:line}].toGrid
  preCustom:[{ruleName:line,function:line,ruleOn:line,targetcount:line,piNumber:line},{ruleName:"Custom rules in the site "+"("+custom.size+" Rules)",function:line,ruleOn:line,targetcount:line,piNumber:line},{ruleName:line,function:line,ruleOn:line,targetcount:line,piNumber:line}].toGrid
  preMissing:[{ruleName:line,function:line,ruleOn:line,targetcount:line,piNumber:line},{ruleName:"Missing Rules as per package "+"("+missing.size+" Rules)",function:line,ruleOn:line,targetcount:line,piNumber:line},{ruleName:line,function:line,ruleOn:line,targetcount:line,piNumber:line}].toGrid

  if(missing.size>0) do
  grid=addRows(grid,preMissing)
  grid=addRows(grid,missing)
  end

  if(oldRules.size>0) do
  grid=addRows(grid, preOld)
  grid=addRows(grid, oldRules)
  end

  if(custom.size>0) do
  grid=addRows(grid,preCustom)
  grid=addRows(grid,custom)
  end

  grid.reorderCols(grid.colNames.moveTo("piNumber", -1))
  end
