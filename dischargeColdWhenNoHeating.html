src

(ahu,dates,tempThreshold:2,minTime:30min,maxTime:60min,freq:2,disTempMinVal:6,disTempMaxVal:60,disTempSPMinVal:6,disTempSPMaxVal:60,check:false,ruleWorkOrder:"None",priority:"Medium",type:"Comfort") => do

  equipWorkOrder:toEquipWorkOrder(ahu)

  occupiedperiods:[].toGrid
  spark:[].toGrid
  spark1:[].toGrid
  spark2:[].toGrid

  heatElement:read(heat and cmd and equipRef == ahu->id,false)
  coolElement:read(cool and cmd and equipRef == ahu->id,false)
  chilledWaterTempSensor:[].toGrid
  occupancyPoint:read(occupied and equipRef == ahu->id,false)

  disTemp:ahu.toAhuDischargeTemp(true)
  if(readAll(discharge and temp and sp and equipRef==ahu->id).size>1) throw "More than one SP"
  else disTempSp:ahu.toAhuDischargeTempSp(true)

  maintanancePoint:ahu.tomaintenancePoint()

  if(heatElement==null and coolElement==null)
  throw "No Heat or Cool Points"

  ahuOn:ahu.ahuRunPeriods(dates)
  disTempHis:disTemp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>disTempMinVal and v<disTempMaxVal))
  disTempSpHis:disTempSp.hisRead(dates).hisFindPeriods(v=>(v!=na() and v>disTempSPMinVal and v<disTempSPMaxVal))
  issueTime:comparepoints(disTempSp,disTemp,dates,tempThreshold).hisFindPeriods(matchPointVal(_,"greater"))

  spark=hisPeriodIntersection([issueTime,ahuOn,disTempHis,disTempSpHis])

   spark_inter:spark

  if(maintanancePoint!=null)
    do
      maintanceHis:maintanancePoint.hisRead(dates).hisFindPeriods(v=>(v!=na() and v==false))
      spark_inter= hisPeriodIntersection([spark,maintanceHis])
    end

  if(heatElement!=null and spark_inter.size>0)
    do
      noHtgPeriods:ahu.ahuNoHeatPeriods_inter(dates,1)
      spark_inter= hisPeriodIntersection([spark_inter,noHtgPeriods])
    end

    if(coolElement!=null and spark_inter.size>0)
    do
      noClgPeriods:ahu.ahuNoCoolPeriods_inter(dates,1)
      spark_inter= hisPeriodIntersection([spark_inter,noClgPeriods])
    end

  if(occupancyPoint!=null and check== true and spark_inter.size>0 )
    do
    occupiedperiods=togetOccupancy(ahu,dates)
    spark_inter= hisPeriodIntersection([spark_inter,occupiedperiods])
    end

  spark1=spark_inter.findAll(ts => ts->v0 > minTime)
  spark2=spark_inter.findAll(ts => ts->v0 > maxTime)

  if(spark1.size>freq) return spark1.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})
else  if(spark2.size>0) return spark2.addMeta({spark_ruleWorkOrder:parseWorkOrder(ruleWorkOrder),spark_equipWorkOrder:parseWorkOrder(equipWorkOrder),spark_priority:parseSparkPriority(priority),spark_type:parseSparkType(type)})

end
