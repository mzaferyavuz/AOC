(dates,emailId,dataLossTime:2hr,naTime:6hr) => do

  tripp:false
  wdIssue:false
  oatDataIssue:false
  pntDataIssue:false
  overrideIssue:false
  energyDataIssue:false
  eoGroupDataLoss:false
  eoGroupDataIssue:false

  sch:read(eo and schedule)
  sw:readAll(eo and switch and sensor)

  schh:sch.hisRead(dates)
  swh:sw.hisRead(dates).hisMap(v=> if(v!=na() and v==true) 1 else if(v!=na() and v==false) 0 else null).hisFoldCols(sum)

  statusS:hisJoin([swh,schh]).hisInterpolate.map rr=> do if(rr["v0"]==0 and rr["v1"]==true) {ts:rr->ts,v0:1} end
  if(statusS.size>0) statusS=statusS.hisFindPeriods(v=>v!=na() and v>0)

  if(statusS.size>0) tripp=true

  if(tripp==true) do // start of tripping

  // switch and strategy and sensor - all should be true

  table:[].toGrid
  issueOver:[].toGrid

  issueOver=sw.map row=> do
    b:null

    rowh:row.hisRead(dates)
    a:hisJoin([rowh,schh]).hisInterpolate.map r=> do if(r["v0"]==false and r["v1"]==true) {ts:r->ts,pnt:row->dis,valP:r["v0"],valS:r["v1"]} end.findAll(v=> v["val"]!=null).sort("ts")
    if(a.size>0) b=a.findAll(v=> v["valP"]!=null).sort("ts").toRec
    if(b!=null) {ts:b->ts,point:b->pnt}
    end

  if(issueOver.size>1) do
    frst:issueOver.findAll(v=> v["pnt"]!=null).sort("ts").toRec
    scnd:issueOver.findAll(v=> v["pnt"]!=null).sort("ts")[1].toRec
    if(frst["ts"]==scnd["ts"] ) table=[{ts:frst["ts"],pnt:frst["pnt"]}].toGrid
    end

  else if (issueOver.size==1) table=issueOver

  if(table.size>0) overrideIssue=true

  /////////////////////////////////////////////////////end of strategy switch

  // ouside and temp - should have data

  oat:readAll(outside and temp and eo)

  oatfault:oat.map row=>do

    nas:pointNaFinder(row,dates,"na",dataLossTime,naTime)
    naI:pointNaFinder(row,dates,"inter",dataLossTime,naTime)
    if(row["hisEnd"]!=null) do lag:(now()-row->hisEnd).to(1hr) end else lag:25hr
    if(lag>dataLossTime) dataloss:true else dataloss:false

    {dis:row->dis,naIinter:naI,naNormal:nas,dataLoss:dataloss,hise:row["hisEnd"],dlag:lag}
    end

  nooadissue:oatfault.findAll(v=> v->dataLoss==true or v->naIinter==true or v->naNormal==true)

  if(nooadissue.size>0) oatDataIssue=true

  /////////////////////////////////////////////////////end of OAT


  // temp and sp and optimized - all should have data
  // pressure and sp and optimized - all should have data

  allp:readAll((temp or pressure) and eo and applied and sp)

  pointfault:allp.map row=>do

  nas:pointNaFinder(row,dates,"na",dataLossTime,naTime)
  naI:pointNaFinder(row,dates,"inter",dataLossTime,naTime)
  if(row["hisEnd"]!=null) do lag:(now()-row->hisEnd).to(1hr) end else lag:25hr
  if(lag>dataLossTime) dataloss:true else dataloss:false

  {dis:row->dis,naIinter:naI,naNormal:nas,dataLoss:dataloss,hise:row["hisEnd"],dlag:lag}
  end

  nopoinissue:pointfault.findAll(v=> v->dataLoss==true or v->naIinter==true or v->naNormal==true)

  if(nopoinissue.size>0) pntDataIssue=true

  /////////////////////////////////////////////////////end of point data issue


  // watchDog and sensor - should be true

  wd:read(watchDog and sensor and eo)

  wdh:wd.hisRead(dates)
  wdissue:hisJoin([wdh,schh]).hisInterpolate.map r=> do if(r["v0"]==0 and r["v1"]==true) {ts:r->ts,status:"Issue"} end
  if(wdissue.size>0) wdissue=wdissue.findAll(v=> v["status"]!=null)
  if(wdissue.size>0) wdIssue=true

  /////////////////////////////////////////////////////end of Watch Dog


  // energy and point - all should have data

  allE:readAll(energy and eo and not outside and not temp and not pressure)

  ePointFault:allE.map row=>do

  nas:pointNaFinder(row,dates,"na",dataLossTime,naTime)
  naI:pointNaFinder(row,dates,"inter",dataLossTime,naTime)
  if(row["hisEnd"]!=null) do lag:(now()-row->hisEnd).to(1hr) end else lag:25hr
  if(lag>dataLossTime) dataloss:true else dataloss:false

  {dis:row->dis,naIinter:naI,naNormal:nas,dataLoss:dataloss,hise:row["hisEnd"],dlag:lag}
  end

  noepoinissue:ePointFault.findAll(v=> v->dataLoss==true or v->naIinter==true or v->naNormal==true)

  if(noepoinissue.size>0) energyDataIssue=true

  /////////////////////////////////////////////////////end of energy data issue


  // eoGroupRef - atleat one zone with demand and temp data

  eog:readAll(eoGroup).map row=>do
  eoGDl:eoGroupNoDL(row,dataLossTime)
  if(eoGDl.size==0) {eoGrp:row->dis}
  end

  if(eog.size>0) eoGroupDataLoss=true

  eogna:readAll(eoGroup).map row=>do
  eoGDissue:eoGroupNoNA(row,dates,dataLossTime,naTime)
  if(eoGDissue.size==0) {eoGrp:row->dis}
  end

  if(eogna.size>0) eoGroupDataIssue=true

  /////////////////////////////////////////////////////end of eoGroup

  // Mail segment

  cause:""
  cause1:""
  cause2:""
  cause3:""
  cause4:""
  cause5:""

  if(overrideIssue==true) do //need to have the point
    record:table.toRec
    casue="EO switch is overrriden " + " " +record
    end

  if(oatDataIssue==true) do
    nooadissue.ioWriteExcel(`io/EO/OAT data issue.xls`)
    record1:nooadissue.colToList("dis")
    casue1="All OAT points were having data issue" + " " +record1
    end

  if(pntDataIssue==true) do //need to have the list
    nopoinissue.ioWriteExcel(`io/EO/SP data issue.xls`)
    record2:nopoinissue.colToList("dis")
    casue2="Point being optimized is having data Issue" + "*" +record2
    end

  if(energyDataIssue==true) do //need to have the point list
    noepoinissue.ioWriteExcel(`io/EO/energy point data issue.xls`)
    record3:noepoinissue.colToList("dis")
    casue3="energy point for the unit being optimized is having data issue" + "*" +record3
    end

  if(wdIssue==true) do
    record4:wdissue.toRec
    casue4="Watchdog point is reading false" + "*" +record4
    end

  if(eoGroupDataLoss==true or eoGroupDataIssue==true) do //need to have EO group list
    if(eog.size>0) record5: "points in data loss " + eog.colToList("eoGrp") else record5:""
    if(eogna.size>0) record6:"points reading na " +eogna.colToList("eoGrp") else record6:""
    casue5="EO group is not having a unit with valid data to not even a single zone" + "*" + record5 + record6
    end

  project:read(projMeta)->dis
  subject:project+" EO Stop Report "+now().date

  body:
   "<html><body>\n"+
     "<b1>EO stopped due to the following reason </b1>\n"+
     "<p> "+cause+" </p>"+
     "<p> "+cause1+" </p>"+
     "<p> "+cause2+" </p>"+
     "<p> "+cause3+" </p>"+
     "<p> "+cause4+" </p>"+
     "<p> "+cause5+" </p>"+
     "<p><small><b> Please do not reply to this email. This is an unmonitored mailbox </b></small></p>"+
   "</body></html>\n"

     "</body></html>\n"

  if(oatDataIssue==true and pntDataIssue==true and energyDataIssue==true) emailSend(emailId,subject,body,[`io/EO/OAT data issue.xls`,`io/EO/SP data issue.xls`,`io/EO/energy point data issue.xls`])
  else if(oatDataIssue==true and pntDataIssue!=true and energyDataIssue==true) emailSend(emailId,subject,body,[`io/EO/OAT data issue.xls`,`io/EO/energy point data issue.xls`])
  else if(oatDataIssue!=true and pntDataIssue==true and energyDataIssue==true) emailSend(emailId,subject,body,[`io/EO/SP data issue.xls`,`io/EO/energy point data issue.xls`])
  else if(oatDataIssue==true and pntDataIssue==true and energyDataIssue!=true) emailSend(emailId,subject,body,[`io/EO/OAT data issue.xls`,`io/EO/SP data issue.xls`])
  else if(oatDataIssue!=true and pntDataIssue!=true and energyDataIssue==true) emailSend(emailId,subject,body,[`io/EO/energy point data issue.xls`])
  else if(oatDataIssue==true and pntDataIssue!=true and energyDataIssue!=true) emailSend(emailId,subject,body,[`io/EO/OAT data issue.xls`])
  else if(oatDataIssue!=true and pntDataIssue==true and energyDataIssue!=true) emailSend(emailId,subject,body,[`io/EO/SP data issue.xls`])
  else if(oatDataIssue!=true and pntDataIssue!=true and energyDataIssue!=true) emailSend(emailId,subject,body)

  end // end of tripping

  end
